רִינָה וְיָעֵל הֵן חָבֵרוֹת טוֹבוֹת עוֹד מֵהַגַן. הַן גַם גָרוֹת בִּשְכֵנוּת זוֹ לַזוֹ. יוֹם יוֹם לְאַחַר הַלִימוּדִים הֵן נִפְגָשוֹת וּמְבָלוֹת בְּיַחַד אֶת שְעוֹת אַחַר הַצָהָרַיִים. יוֹם אֶחַד גִילוּ סוּלַם נִיצַב לְיַד הַקִיר שֶל בַּיִת עָזוּב בַּשְכוּנָה. סַקְרָנוּתָה שֶל יָעֵל הִתְעוֹרֶרָה, וְהִיא אָמְרָה: "בוֹאִי נִרְאֶה לְאֵן מוֹבִיל הַסוּלָם". אָמְרָה, וְהִתְחִילָה לְטָפֵּס עַל הַסוּלָם, וְרִינָה אַחַרֵיהָ.
טִיפְּסוּ וְטִיפְּסוּ עַד שֶהִגִיעוּ לְפֶתַח שֶל עָלִיָית גַג. בְּעָלִיָית הַגַג שָׂרֵרָה אַפְלוּלִית – לֹא הָיָה בָּה חָלוֹן מִלְבַד הַפֶּתַח שֶדַרְכּוֹ נִכְנְסוּ. גִישֵשוּ שְתֵי הֶחֲבֵרוֹת בַּאָפֶלֶה, כְּשֶהֵן אוֹחָזוֹת יָדַיִים. הֵן הִתְיָשְבוּ בַּחֵלֵק הָאָחוֹרִי שֶל הַחֶדֶר, וְדִיבְּרוּ בְּקוֹל חָרִישִי, מְתוֹךְ פַּחַד, שֶמַא מִישֶהוּ נִמְצָא שַם.
לְפֶתַע נִשְמַע רַעַש מוּזָר מְיַרְכֵתֵי הַחֶדֶר. רִינָה נִבְהָלָה מְאוֹד, וּבְצֵעָקָה רָמָה בָּרְחָה וְיָרְדָה בִּמְהִירוּת מְהָסוּלָם. יָעֵל, שֶהָיְתָה אָמִיצָה יוֹתֵר, בָּדְקָה מֵאָין בָּא הָרַעַש הַמוּזָר, וְגִילְתָה גוּר-חָתוּלִים קָטָן. לָקְחָה יָעֵל אֶת הַגוּר לַפֶתַח,וְקָרְאָה כְּשֶהִיא צוֹחֶקֶת:" אֵיזוֹ פַּחְדָנִית אַתְּ רִינָה, זָה בְּסַךְ הַכֹּל גוּר קָטָן, חִזְרִי!". אָבָל רִינָה לֹא חָזְרָה. הִיא נֶעֶלְבָה, כִּי יָעֵל קָרְאָה לָה "פַּחְדָנִית". הִתְקָרְבָה אֶל 
הַסוּלָם וְהֵזִיזָה אוֹתוֹ מִפֶתַח עָלִיָת הַגַג. בִּיקְשָה יָעֵל לָרֶדֶת, אַךְ לֹא יָכְלָה. הִיא הִתְחָנֵנָה לְרִינָה שֶתַחְזִיר אֶת הַסוּלָם לִמְקוֹמוֹ,אַךְ זוֹ סִרְבָה וְעָזְבָה אֶת הַמָקוֹם. יָעֵל נִבְהֲלָה וְלֹא יָדְעָה מָה לַעֲשׂוֹת. רַק לְאַחַר שָעָה אֲרוּכָּה חָזְרָה רִינָה וְהִחְזִירָה אֶת הַסוּלָם לִמְקוֹמוֹ – כְּדֵי שֶיָעֵל תוּכָל לַרֶדֶת. לְאַחַר שֶיָעֵל יָרְדָה וְגוּר – הֶחָתוּלִים בְּיָדֵיהָ, אָמְרָה לְרִינָה בְּכֵעֵס: "יוֹתֵר אֵל תֵדַבְּרִי אֶלַי,אֵינֵנִי חֲבֵרָה שֶלַךְ עוֹד." הִשְתוֹקֶקָה רִינָה לְבָקֶש אֶת סְלִיחָתָה, אַךְ יָעֵל סִירְבָה וְרָצָה לְבֵיתָה, וּמְאַז לֹא דִיבְּרָה אִתָה יוֹתֵר.
בִּכְפָר אֶחָד גָר זָקֵן מֻפְלָג. הַזָּקֵן גָר בְּבַיִת קָטָן יַחַד עִם קְרוֹבַת מִשְׁפָּחָה, 
רָעַת לֵב, אֲשֶׁר מְרֵרָה אֶת חַיָּיו. יוֹם יוֹם עָבַד הַזָּקֵן עֲבוֹדָה קָשָׁה בַּבָּיִת וּבַגַן, וְאִלּוּ הָאִשָּׁה יָשְׁבָה בְּטֵלָה וְדִיבְּרָה אֵלָיו קָשׁוֹת.
וְלַזָּקֵן בַּת יְחִידָה אֲשֶׁר גָּרָה רָחוֹק, מֵעֵבֶר לֶהָרִים. יוֹם אֶחָד, כַּאֲשֶׁר לֹא יָכוֹל הָיָה הַזָּקֵן לִסְבֹּל עוֹד- כָּתַב מִכְתַּב לְבִתּוֹ וְסִפֵּר לָהּ עַל חַיָּיו הַקָּשִׁים. 
כַּעֲבוֹר שָׁבוּעוֹת אֲחָדִים קִבֵּל מִבִּתּוֹ חֲבִילָה קְטַנָּה. בַּחֲבִילָה הָיְתָה פְּנִינָה יְקָרָה וּלְיָדָה מִכְתַּב. וְכָךְ הָיָה כָּתוּב בַּמִכְתָּב: "שְׁמוֹר עַל פְּנִינָה זוֹ הֵיטֵב. אַל תִמְכְּרֶנָּה, אַל תַחְלִיפֶנָה וְאַל תִּתְּנֶנָּה בְּמַתָּנָה. אִם תְּמַלֵּא אֶת בַּקָּשׁוֹתַי, אֲזַי טוֹב יִהְיֶה לְךָ כָּל יְמֵי חַיֶּיךָ."
הִתְפַלֵּא הַזָּקֵן עַל בַּקָּשָׁתָהּ הַמּוּזָרָה שֶׁל בִּתּוֹ, אַךְ הֶחְלִיט לַעֲשׂוֹת כְּבַקָּשָׁתָהּ.
בַּעֶרֶב בָּאָה הַבָּיְתָה קְרוֹבַת הַמִּשְׁפָּחָה הָרָעָה שֶׁל הַזָּקֵן. כַּאֲשֶׁר רָאֲתָה הָאִשָּׁה אֶת הַפְּנִינָה בִידֵי הַזָּקֵן נָצְצוּ עֵינֶיהָ. "שֵׁב נָא וְנוּחַ מֵעֲבוֹדָתְךָ הַקָּשָׁה", אָמְרָה הָאִשָּׁה, וּבְלִבָּהּ חָשְׁבָה שֶּׁתְּקַבֵּל אֶת הַפְּנִינָה בְּמַתָּנָה. אֲבָל הַזָּקֵן שָׁמַר עָלֶיהָ וְלֹא נָתַן אוֹתָהּ לַאִישָׁהּ הָרָעָה.
לְמָּחֳרָת הַיּוֹם, בְּשָׁעָה שֶׁהָאִשָּׁה הַגִּישָׁה לַזָקֵן אוֹכֵל טָעִים אָמְרָה לַזָקֵן: 
"הָבָה נִמְכֹּר אֶת הַפְּנִינָה וּבַכֶּסֶף נִקְנֶה בַּיִת גָּדוֹל וּמְרֻוָּח."
אֲבָל הַזָּקֵן לֹא הִסְכִּים לִמְכּוֹר אֶת הַפְּנִינָה.
כָּךְ עָבְרוּ יָמִים רַבִּים, הָאִשָּׁה טִפְּלָה בַּזָקֵן בִּמְסִירוּת וְהִשְׁגִּיחָה עָלָיו, 
כִּי קִיוְתָה שֶׁיּוֹם יָבוֹא וְהִיא תְּקַבֵּל אֶת הַפְּנִינָה מִיַּד הַזָּקֵן.
בֵּינְתַיִם חַי הַזָּקֵן בְּאֹושֶׁר וּבְכָבוֹד כָּל יְמֵי חַיָּיו, כְּדִבְרֵי בִּתּוֹ הַחָכְמָה
בְּחֹדְשֵי הַסְתַיו אֲנוּ רוֹאִים בַּשָמַיִם הַרְבֵּה צִיפּוֹרִים נוֹדְדוֹת.
אֶרֶץ יִשְׂרָאֶל הִיא אֶרֶץ מַעֲבָר לְהַרְבֶּה צִיפּוֹרִים נוֹדְדוֹת.
צִיפּוֹרִים רָבּוֹת נוֹדְדוֹת דֶרֶךְ אֶרֶץ יִשְׂרָאֶל. הצִיפּוֹרִים נוֹדְדוֹת בְּעִקְבוֹת מְזוֹנָן. 
צִיפּוֹרִים הַחַיוֹת בְּאַרַצוֹת הַצָפוֹן הַקָרוֹת אוֹכְלוֹת בְּעִיקָר חֲרָקִים, גַרְגִירִים וְדָגִים. בָּחורֶף, כְּשֶכָּל הַאֲדָמָה מְכֻסָה שֶלֶג וְקֶרָח, הֵן יִתְקָשוּ לִמְצוֹא לַהֵן אֹכֶל. 
לַכֵן נוֹדְדוֹת הצִיפּוֹרִים לְאַרַצוֹת חָמוֹת יוֹתֶר שֶאֶפְשָר לִמְצוא שָם מָזוֹן. 
שִׁירָז הִגִיעָה חֲגִיגִית לְבֵית-הַסֶפֶר,
הַמוֹרוֹת בַּכְּנִיסָה אָמְרוּ:
" שִׁירָז אַתְּ כָּל-כָּךְ יָפָה הַיוֹם! ",
וְשִׁירָז חִיְכָה וְאָמְרָה:
" תוֹדָה, יֵשׁ לִי הַיוֹם
מְסִבַּת יוֹם-הוּלֶדֶת! "
כְּשֶׁנִכְנְסָה שִׁירָז לַכִּתָה,
אָמְרוּ לָה אֵלָה וְנוֹעָה:
"מַזָל טוֹב שִׁירָז! ",
וְדוֹרוֹנִי חִיֵיךְ וְאָמַר:
" אַתְּ כְּמוֹ מַלְכָּה הַיוֹם! "
בַּמְסבָּה שָׁרוּ כָּל הַיְלָדִים
שִׁירֵי יוֹם-הוּלֶדֶת,
וְנָתְנוּ לָה אֶת הַמַתָנוֹת.
בַּסוֹף אָמְרָה שִׁירָז:
" הָיָה לִי יוֹם נִפְלָא, תוֹדָה. "
"אַבָּא תִקְנֶה לִי מְרִיצָה בַּחֲנוּת הַצַעֲצוּעִים, וְגַם אוֹטוֹ מָנוֹף
"אַבָּא תִקְנֶה לִי מְרִיצָה בַּחֲנוּת הַצַעֲצוּעִים, וְגַם אוֹטוֹ מָנוֹף וּמַשָׂאִית גַם. טוֹב אַבָּא?" כָּךְ אוֹמֵר רוֹן שׁוּב וָשׁוּב, וְחוֹזֵר עַל כָּךְ בְּלִי סוֹף
אֲבָל אַבָּא חוֹשֵׁב לוֹ וּבִכְלָל לֹא שָׂם לֵב. אַחַר כַּךְ הוּא מְדַבֵּר עִם אִמָא וְלֹא מַקְשִׁיב לַבַּקָשָׁה. וּכְשֶׁהוּא עוֹנֶה לְרוֹן סוֹף סוֹף, הוּא אוֹמֵר:
"לֹא עַכְשָׁו, רוֹן. מַה פִּתְאוֹם, רוֹן? אוּלַי תַפְסִיק כְּבָר רוֹן? מַסְפִּיק וְדַי עִם 'אַבָּא תִקְנֶה לִי' הַזֶה כָּל הַזְמַן."
אִילוּ טַלִי, הָאָחוֹת הַגְדוֹלָה, הָיְתָה פֹּה הִיא כְּבָר הָיְתָה אוּלַי מַסְכִּימָה וְקוֹנָה. אֲבָל טַלִי בַּצָבָא. וְאַבָּא כַּנִרְאֶה לֹא מְבִין שֶׁרוֹן רוֹצֶה כָּל כָּךְ מְרִיצָה, וְגַם אוֹטוֹ מָנוֹף וְגַם מַשָׂאִית.
אַבָּא נוֹהֵג בַּמְכוֹנִית, וְאִמָא יוֹשֶׁבֶת לְצִדוֹ, נוֹסְעִים הָעִירָה . בַּעִיר יֵשׁ הַרְבֶּה חֲנוּיוֹת, וּבָהֶן צַעֲצוּעִים. רוֹן רוֹצֶה כְּבָר לְהַגִיעַ
לַחֲנוּיוֹת אֲבָל פִּתְאוֹם עוֹצֵר אָבַּא לִפְנֵי הַרַמְזוֹר – וּמְחַכֶּה. לְפָנָיו וְאַחֲרָיו נֶעֱצָרוֹת עוֹד מְכוֹנִיוֹת, וְכוּלָן מְחַכּוֹת.
אֲבָל לֹא עָבַר רֶגַע – וְכוּלָן, בְּבַת אַחַת, מַתְחִילוֹת שׁוּב לִנְסוֹעַ. אֵיךְ זֶה? לָרַמְזוֹר יֵש פֶּה וְהוּא אוֹמֵר לַמְכוֹנִיוֹת מָתַי לַעֲצוֹר וּמָתַי לִנְסוֹעַ?
רוֹן מַבִּיט וְלא מֵבִין. הִנֵה, בְּצַד הַכְּבִישׁ עוֹמְדִים אֲנָשִׁים – הוֹלְכֵי רֶגֶל, שֶׁלֹא נוֹסְעִים בִּמְכוֹנִיוֹת. לְיָד הַרַמְזוֹר כּוּלָם נֶעֱצָרִים.
אִמָא עִם עֶגְלַת תִינוֹק. סָבְתָא עִם סַל קנִיוֹת. דוֹד אֶחָד וּגְבֶרֶת. כּוּלָם עוֹמְדִים וּמְחַכִּים מוּל הַרַמְזוֹר, כְּאִילוּ הוּא לֹא מַרְשֶה לַהֶם לַעֲבוֹד.
אַחַר-כַּךְ מַתְחִילִים כּוּלָם לָלֶכֶת, חוֹצִים יַחַד אֶת הַכְּבִיש.
אֵיךְ זֶה? לָרַמְזוֹר יֵש פֶּה וְהוּא אוֹמֵר לַאֲנָשִׁים מָתַי לַלֶכֶת וּמָתַי לַעֲמוֹד?
רוֹן לֹא שָׁמַע אוֹתוֹ אוֹמֵר אַף מִילָה אחָת... אֲבָל אוּלָי הַנֶהָגִים וְהוֹלְכֵי הַרֶגֶל שׁוֹמְעִים?
רוֹן מַבִּיט דֶרֶךְ הַחַלוֹן וְרוֹצֶה לְגַלוֹת אֵיךְ נִרְאֶה הַפֶּה שֶׁל הָרַמְזוֹר וְאֵיזֶה קוֹל יֵש לוֹ. אֲבָל הוּא לֹא רוֹאֶה וְלֹא שׁוֹמֵעַ כְּלוּם.
הַמְכוֹנִית נוֹסַעַת הָלְאָה – וְעוֹצֶרֶת בַּרַמְזוֹר הַבָּא. וְאָז שׁוֹאֵל רוֹן אֶת אַבָּא:
"תַגִיד לִי אַבָּא, רַמְזוֹר יוֹדֵעַ לְדַבֵּר? יֵשׁ לוֹ פֶּה?"
"מוּבָן שֶׁרַמְזוֹר לֹא יוֹדֵעַ לְדַבֵּר. אֵין לוֹ פֶּה," עוֹנֶה אבא.
עַכְשָו יֵשׁ לְרוֹן שְׁאֵלָה חֲשׁוּבָה:
"תַגִיד לִי, אַבָּא,"הוּא שׁוֹאֵל "אִם לָרַמְזוֹר אֵין פֶּה, וְהוּא לֹא יוֹדֵעַ לְדַבֵּר, אִיךְ יוֹדְעִים כָּל הַנֶהָגִים מָתַי לַעְצוֹר וּמָתַי לִנְסוֹעַ? וְאֵיך

הַיָרוֹק הוּא סִימָן שֶׁמוּתָר לַעֲבוֹר. אַתָה מְבִין, רוֹן?

אִם אֵין מַעֲבַר חֲצִיָּה ?

וְאַתֶּם רוֹאִים לְמֶרְחָק רַב וְנִרְאִים הֵיטֵב.

חוֹצִים בְּקַו יָשָׁר וּבַהֲלִיכָה מְהִירָה חוֹצִים בִּזְרִיזוּת בַּקַּו הַיָּשָׁר וְהַקָּצָר בְּיוֹתֵר. אָסוּר לַחֲצוֹת בַּאֲלַכְסוֹן .
תּוֹךְ כְּדֵי הַחֲצִיָּה מִסְתַּכְּלִים לְכָל הַכִּוּוּנִים.
אֵיךְ אֶפְשָׁר לַחְצוֹת אֶת הַכְּבִישׁ אִם אֵין מַעֲבַר חֲצִיָּה קָרוֹב וְיֵשׁ הַרְבֵּה מְכוֹנִיוֹת חוֹנוֹת? 
חוֹצִים בֵּין מְכוֹנִיּוֹת חוֹנוֹת רַק כְּשֶׁאֵין אֶפְשָׁרוּת אַחֶרֶת וּמְכוֹנִיּוֹת חוֹנוֹת לְאֹרֶךְ כָּל הָרְחוֹב וְאֵין מַעֲבַר חֲצִיָּה קָרוֹב.

אֶת הַשְּׁכֵנִים שֶׁשָּׁרִים יַחַד כָּל עֵרֶב מֵהַמִּרְפָּסוֹת
בִּמְקוֹם לִהְיוֹת בְּאַיְיפוֹן,
אֶת הַהוֹרִים שֶׁקָּמִים כָּל בּוֹקֶר לַיְּלָדִים וְלֹא לְקַרְיֵירָה,
אֶת הַזּוּגוֹת שֶׁנִּמְצָאִים יַחַד גַּם בְּבִידּוּד,
וְאֵת אֵלּוּ שֶׁמִּתְגַּעְגְּעִים לְחִיבּוּק פָּשׁוּט,
לִלְחִיצַת יָד אֲרֻוכָּה,
כְּבִישִׁים עִם פָּחוֹת פְּקָקִים, אֲוִויר נָקִי עִם פָּחוֹת עָשָׁן ,שַׁבָּתוֹת שְׁקֵטוֹת
וּבָתִּים מְלֵאִים בְּבִילּוּי,
שְׂכֹּל הָעוֹלָם יִידָּבֵק
בָּוִירוּס הַמְּקָרֵב הַזֶּה הַמְּנַקֶּה הַזֶּה הַמְּטַהֵר הַזֶּה,
שֶׁל  יְמֵי הֲקוֹרוֹנָה…
הָיְתָה הַפְסָקָה, וְהַיְלָדִים נִשְׁאֲרוּ בַּפְּרוֹזְדוֹרִים שֶׁל בֵּית הַסֵפֶר, כִּי יָרַד גֶשֶׁם. גוּסְטַב מִילְבַּךְ, אַחַד הַיְלָדִים מִן הַכִּתוֹת הַגְבוֹהוֹת, טִיֵל בְּגַאֲוָה מֵאֲחוֹרֵי קְבוּצָה שֶׁל בְּנֵי שְׁמוֹנֶה. הוּא הָיָה מֵעֵין מְפַקֵחַ עַל הַקְטַנִים, וְאָהַב מְאֹד אֶת הַתַפְקִיד. לְפֶתַע אָמַר: "תַגִידוּ, סִפַּרְתִי לָכֶם אֶת זֶה פַּעַם?"
הַקְטַנִים הִסְתוֹבְבוּ וְשָׁאֲלוּ: "אֶת מָה, גוּסְטַב?"
"נוּ, אֶת הַסִפּוּר עַל הָאַרְיֵה וְסַל הַקְנִיוֹת!" הוּא אָמַר.
לְלֹא מִלָּה נוֹסֶפֶת הִקִיפוּ אוֹתוֹ הַקְטַנִים, וְאֶחָד מָהֵם, הֶחָצוּף מִכֻּלָם, דָחַק בְּגוּסְטַב: "קָדִימָה, בֶּן-אָדָם, תַתְחִיל לְסַפֵּר כְּבָר!" 
"נוּ, טוֹב," אָמַר גוּסְטַב מִילְבַּךְ, "בַּקַיִץ שֶׁעָבָר הָיִיתִי בְּרֶגִיסְוַלְד. בִּתְחִלַת הַקַיִץ רָצִינוּ בִּכְלָל לִנְסֹעַ לְאִיטַלְיָה, אֲבָל הִמְלִיצוּ לְאַבָּא שֶׁלִי בְּחֹם עַל הַכְּפָר רֶגִיסְוַלְד, אָז נָסַעְנוּ לְשָׁם. אַתֶם יוֹדְעִים מָה זֶה כְּפָר? זֶה מָקוֹם מִחוּץ לָעִיר שֶׁגָרִים בּוֹ אֲנָשִׁים, וּמִי שֶׁצוֹעֵד צַעַד גָדוֹל מִדַי מוֹצֵא אֵת עַצְמוֹ מִחוּץ לַכְּפָר, אֲנִי נִשְׁבַּע לָכֶם. אָז, בְּקִצוּר, אִמָא שֶׁלִי נוֹתֶנֶת לִי אֶת סַל הַקְנִיוֹת וְאוֹמֶרֶת לִי לְהָבִיא קִילוֹ דַיְסָה מִפְּרֶנְצְלָאוּ. פְּרֶנְצְלָאוּ נִמְצֵאת מַמָשׁ מוּל הַכְּפָר, קְצָת בַּאֲלַכְסוֹן. הָעִקָר, הִיא נוֹתֶנֶת לִי מַטְבֵּעַ שֶׁל עֲשָׂרָה פְּפֶנִיג ו..."
"דַיְסָה עוֹלָה יוֹתֵר," הֵעִיר אַחַד הַמַאֲזִינִים הַצְעִירִים.
"לֹא נָכוֹן, דַיְסָה לֹא. בִּגְלַל זֶה גַם שָׁלְחוּ אוֹתִי לִפְּרֶנְצְלָאוּ, כִּי יָדוּעַ שֶׁשָׁם הכֹּל זוֹל יוֹתֵר. אוֹפַנַיִם עוֹלִים שָׁם לֹא יוֹתֵר מִשְׁלוֹשָׁה מַרְקִים וְשִׁשִׁים פְּפֶנִיג. אֲנִי לֹא יוֹדֵעַ אִם הַמְחִירִים שָׁם נְמוּכִים כָּל כָּךְ גַם עַכְשָׁו, אֲבָל הָעִקָר הוּא שֶׁבֵּין רֶגִיסְוַלְד לִפְּרֶנְצְלָאוּ מִשְׂתָרֵעַ יַעַר. אֲנִי אוֹמֵר יַעַר וּבְעֶצֶם זֶה יַעַר עַד - ג'וּנְגֶל שָׁלֵם וּבוֹ צוֹמְחִים פֶּרֶא פֶּטֶל, תוּתִים, פִּטְרִיוֹת רַעַל, נְחָשִׁים, חִפּוּשִׁיוֹת מִסְתוֹבְבוֹת בּוֹ חָפְשִׁיוֹת וְכַמוּבָן מְטַפְּסֵי-בָּר מִשְֹתָרְגִים בּוֹ כָּךְ שֶׁבְּקֹשִׁי אֶפְשָׁר לָזוּז..." 
"וּבְטַח יֵשׁ גַם עֵצִים בַּיַעַר הַזֶה," הֵעִיר אַחַד הַיְלָדִים.
"אֲנִי אַפְסִיק לְסַפֵּר לָכֶם אֶת הַסִפּוּר," אִיֵם גוּסְטַב, וּמִיָד הִמְשִׁיךְ:
"אֶת הַיַעַר הַזֶה הָיִיתִי אָמוּר לַחְצוֹת. מִכָּל עֵבֶר נִשְׁמְעוּ שָׁם לַחְשוּשִים. הָעֵץ הָעֲשִׂירִי מִצַד יָמִין יָדַע לְדַבֵּר, וְאָמוּר לִהְיוֹת שָׁם אוֹצָר שָׁלֵם שֶׁל מַטְבְּעוֹת זָהָב מִזְמַנוֹ שֶׁל קוֹלוּמְבּוּס, כֵּן, כֵּן, וְעוֹד סִפּוּרִים מַרְשִׁימִים בְּיוֹתֵר. בַּהַתְחָלָה הַכֹּל הָיָה בְּסֵדֶר," דִוֵחַ גוּסְטַב לַיְלָדִים, "רַק שֶׁבִּזְמַן שֶׁקָטַפְתִי פֶּטֶל, סָטִיתִי מְעַט מִן הַשְׁבִיל הָרָאשִׁי וְשָׁקַעְתִי עַד לְמָתְנִַי בַּבִּצָה. בִּצָה טוֹבְעָנִית. הִמְשַׁכְתִי לִשְׁקֹעַ וְלִשְׁקֹעַ וַאֲנִי אוֹמֵר לָכֶם שֶׁזֶה הָיָה מַגְעִיל בְּיוֹתֵר. אֶת סַל הַקְנִיוֹת שֶׁל הַמִשְׁפָּחָה הַחְזָקָתִי גָבוֹהַ מֵעַל לָרֹאשׁ, כִּי אִם הוּא הָיָה מִתְלַכְלֵךְ, הָיִיתִי נֶאֱלָץ לִסְחֹב אֶת הַדַיְסָה בַּיָד, וְאֶת זֶה מַמָשׁ לֹא רָצִיתִי. הַבֹּץ הָיָה כְּמוֹ קָקָאוֹ סָמִיךְ וַאֲנִי צָרִיךְ לְהַעֲבִיר בּוֹ אֶת הַסַל עִם הַדַיְסָה. בְּקִצוּר הַמַצָב הָיָה דַיְסַת קָקָאוֹ. וְאָז, כְּשֶׁרַק הָרֹאשׁ שֶׁלִי הֵצִיץ מִן הַקָקָאוֹ הַזֶה, הוֹפִיעַ מִן הַשָׁמַיִם אֲוָז, אֲוָז בָּר. תָפַסְתִי אוֹתוֹ בָּרַגְלַיִם וְהוּא הִמְרִיא עַד שֶׁחִלֵץ אוֹתִי וְהֶעֱבִיר אוֹתִי לְקָרַחַת יַעַר יְבֵשָׁה..."
"אֶת הַסִפּוּר הַזֶה בְּדִיוּק קוֹרֵאת אֲחוֹתִי עַכְשָׁו בְּאֵיזֶה סֵפֶר," הֵעִיר אַחַד הַיְלָדִים.
"תִסְתֹם!" גָעַר בּוֹ אַחַד הַחֲבֵרִים שֶׁלוֹ.
"לִקְרֹא זוֹ בִּכְלָל לֹא חָכְמָה," הֵעִיר בְּבוּז גוּסְטַב מִילְבַּךְ וְהִמְשִׁיךְ בַּסִפּוּר: "פִּטְפַּטְנוּ קְצָת, הָאַוָז וַאֲנִי, וְאַחַר כָּךְ הוֹדֵיתִי לוֹ וְהוּא הִמְשִׁיךְ בְּדַרְכּוֹ בְּמַשַׁק כְּנָפַיִם. וּבְעוֹד אֲנִי מְנַסֶה לְסַדֵר קְצָת אֶת הַחֲלִיפָה שֶׁלִי, שֶׁנִרְאֲתָה נוֹרָא וְאָיֹם, אֲנִי שׁוֹמֵעַ מֵאֲחוֹרַי שְׁאָגָה. מַמָשׁ קָרוֹב אֵלַי. תַאֲמִינוּ לִי שֶׁזוֹ הָיְתָה שְׁאָגָה אֲיֻמָה. וְאָז אֲנִי מֵזִיז אֶת הָעֲנָפִים שֶׁל הַצִמְחִיָה וְרוֹאֶה לְפָנַי אַרְיֵה. כֵּן, כֵּן, אַרְיֵה אֲמִתִי שֶׁנִרְאָה כְּמוֹ שְׁנֵי כַּלְבֵי סַן-בֶּרְנַרְד עֲנָקִיִים אֶחָד עַל הַשֵׁנִי. כִּמְעַט וְנִבְהַלְתִי. וְנוֹסָף עַל הַכֹּל גַם שָׁכַחְתִי אֶת רוֹבֶה הָאֲוִיר שֶׁלִי בַּבַּיִת."  
 "זֶה גַם לֹא הָיָה מַמָשׁ עוֹזֵר לְךָ," הֵעִיר מִישֶׁהוּ. 
"בְּכָל אֹפֶן זֶה יוֹתֵר טוֹב מִכְּלוּם," עָנָה גוּסְטַב. מָה הָיִיתִי יָכוֹל כְּבָר לַעֲשׂוֹת? הָאַרְיֵה נָעַץ בִּי אֶת הָעֵינַיִם שֶׁלוֹ, סִלְסֵל סִלְסוּל אֶחָד בַּשָׂפָם וְהֵחֵל לְהִתְקָרֵב אֵלַי בִּצְעָדִים קְפִיצִיִים וּכְשֶׁהוּא מְלַקֵק אֶת שְׂפָתָיו."
דַי, תַפְסִיק!" בִּקֵשׁ אַחַד הַיְלָדִים. 
"הוּא הִתְקָרֵב אֵלַי עוֹד וְעוֹד," תֵאֵר גוּסְטַב, "נִרְאָה עָלָיו שֶׁהוּא עוֹד לֹא אָכַל אֲרוּחַת בֹּקֶר. הוּא הוֹרִיד אֶת הָרֹאשׁ כְּאִלוּ הוּא מְחַפֵּשׂ עַל הָאֲדָמָה מַשֶׁהוּ לֶאֱכֹל. אֲנִי כְּבָר הִרְגַשְׁתִי שֶׁהוּא מַתְחִיל לִלְעֹס אוֹתִי... וְאָז נִזְכַּרְתִי שֶׁיֵשׁ לִי בַּכִּיס כַּמָה בּוּלִים. הוֹצֵאתִי אוֹתָם מֵהַכִּיס, לִקַקְתִי אֶת הַצַד הָאֲחוֹרִי שֶׁלָהֶם, הַדָבִיק. הִתְחַלְתִי לְפַזֵר בּוּלִים לְפָנַי וְלָסֶגֶת בַּהֲלִיכָה לְאָחוֹר, לְאַט לְאַט. הָאַרְיֵה רָטַן וְרִחְרַח אֶת הַבּוּלִים עַד שֶׁאֵלֶה נִדְבְּקוּ לַשָׂפָם שֶׁלוֹ, לִלְחָיָיו, וְכַמָה שֶׁהוּא נִסָה לְהִפָּטֵר מֵהַבּוּלִים הֵם הִתְעַקְשׁוּ לְהִדָבֵק אֵלָיו. בָּאתִי אֵלָיו מִן הַצַד, תָפַסְתִי לוֹ בָּרַעֲמָה. חָבַטְתִי אֶת הָרֹאשׁ שֶׁלוֹ בָּאֲדָמָה וַאֲנִי יָכוֹל לְהַגִיד לָכֶם שֶׁהַפֶּה שֶׁלוֹ הָיָה נָעוּל כְּמוֹ חֲבִילָה בִּסְנִיף הַדֹאַר שֶׁמְחַכָּה שֶׁיִשְׁלְחוּ אוֹתָה לְחוּץ לָאָרֶץ."
"אֲבָל הוּא הֲרֵי יָכוֹל לִקְרֹעַ אֵת הַנְיָר דֶבֶק שֶׁל הַבּוּלִים. לֹא?" כָּךְ צָוַח אַחַד הַקְטַנִים הַמְבֹהָלִים.
"אִלוּ הוּא הָיָה מַצְלִיחַ בָּזֶה, אָז כְּנִרְאֶה שֶׁלֹא הָיִיתִי פֹּה עַכְשָׁו," אָמַר גוּסְטַב, "וְחוּץ מִזֶה: לִפְנֵי שֶׁהָיְתָה לוֹ הִזְדַמְנוּת לְנַסוֹת, זָרַקְתִי אֶת סַל הַקְנִיוֹת שֶׁלָנוּ עַל הָרֹאשׁ שֶׁלוֹ וְזֶה הָיָה כְּמוֹ זְמָם עִם יָדִיוֹת. אָז טִפַּסְתִי עָלָיו וְאָחַזְתִי בַּיָדִיוֹת, יָרַקְתִי לוֹ עַל הָעֹרֶף, בֵּין הָאָזְנַיִם, כְּמוֹ שֶׁעוֹשִׂים עֶגְלוֹנִים כְּשֶׁהֵם רוֹצִים לְזָרֵז אֶת הַסוּסִים שֶׁלָהֶם, אָחַזְתִי חָזָק בַּיָדִיוֹת שֶׁל סַל הַקְנִיוֹת וְעָבַרְנוּ אֶת הַיַעַר בְּטִיסָה. שָׁלוֹשׁ דַקוֹת אַחַר כָּךְ - אֲנִי אוֹמֵר לָכֶם שָׁלוֹשׁ דַקוֹת - הָיִינוּ בִּפְּרֶנְצְלָאוּ. שָׁלוֹשׁ דַקוֹת, אֲנִי יוֹדֵעַ בְּדִיוּק כִּי הֵצַצְתִי בַּשָׁעוֹן.
"וְאָז..." רָצָה אַחַד הַיְלָדִים לָדַעַת.
"מָה זֹאת אוֹמֶרֶת וְאָז..." בִּקֵשׁ גוּסְטַב לְבָרֵר.
"אֲנִי מִתְכַּוֵן לִשְׁאֹל, אֵיךְ בִּכְלָל הִגִיעַ הָאַרְיֵה הַזֶה לַיַעַר?" שָׁאַל הַיֶלֶד.
"אֲהָהָהָ," הִתְעַשֵׁת גוּסְטַב, "אוּלַי הוּא גָר בְּיַעַר."
"בִּפְּרֶנְצְלָאוּ בִּקֵר אָז קִרְקָס נוֹדֵד," נִשְׁמָע לְפֶתַע מֵאֲחוֹרֵיהֶם קוֹל עָמֹק.
כֻּלָם הִסְתוֹבְבוּ.
זֶה הָיָה הַמוֹרֶה אֶבֶּרְלַין, הַמְחַנֵךְ שֶׁל גוּסְטַב מִילְבַּךְ. הוּא הִבִּיט בְּגוּסְטַב בְּמַבָּט רְצִינִי וְאָמַר: "בֹּא אִתִי בְּבַקָשָׁה לַחֲדַר הַמוֹרִים." וְהַשְׁנַיִם הִמְשִׁיכוּ בְּמוֹרַד הַמִסְדְרוֹן עַד לַחֲדַר הַמוֹרִים.
"תַגִיד," שָׁאַל אַחַד הַיְלָדִים, "הוּא הִשִׂיג אֶת הַדַיְסָה שֶׁלוֹ?"
"אֵין לִי מֻשָׂג," עָנָה יֶלֶד אַחֵר, "אֲבָל דָבָר אֶחָד דֵי בָּרוּר..."
"מָה?"
"שֶׁעַכְשָׁו הוּא יְקַבֵּל מָנָה."
יָדִיד שֶׁל סוֹקְרָטֵס בָּא לְבַקֵר אוֹתוֹ.
"הַאִם שָׁמַעְתָ מָה שֶׁמְסַפְּרִים עַל חֲבֵרְךָ"? שָׁאַל הַיָדִיד אֶת סוֹקְרָטֵס.
"לִפְנֵי שֶׁאֶשְׁמַע, אֲנִי רוֹצֶה לָדַעַת אִם סִנַנְתָ אֶת הַשְׁמוּעָה הַזֹאת בִּשְׁלוֹשׁ הַמְסַנְנוֹת", אָמַר סוֹקְרָטֵס.
"אֵילוּ מְסַנְנוֹת"? שָׁאַל הַיָדִיד.
"הַמְסַנֶנֶת הָרִאשׁוֹנָה הִיא הַמְסַנֶנֶת שֶׁל הָאֱמֶת", אָמַר סוֹקְרָטֵס. "הַאִם מָה שֶׁאַתָה רוֹצֶה לְסַפֵּר לִי עַל חֲבֵרִי הוּא אֱמֶת?"
"אֲנִי לֹא בָּטוּחַ שֶׁזֶה אֱמֶת, שָׁמַעְתִי אֶת זֶה מִמִישֶׁהוּ שֶׁשָׁמַע אֶת זֶה מִמִישֶׁהוּ אַחֵר", אָמַר הַיָדִיד. "וּמָה הַמְסַנֶנֶת הַשְׁנִיָה"?
"הַמְסַנֶנֶת הַשְׁנִיָה הִיא הַמְסַנֶנֶת שֶׁל הַטוֹב", אָמַר סוֹקְרָטֵס, "הַאִם מָה שֶׁאַתָה מִתְכַּוֵן לְסַפֵּר לִי עַל חֲבֵרִי הוּא דָבָר טוֹב?"
"הַמְסַנֶנֶת הַשְׁלִישִׁית הִיא הַמְסַנֶנֶת שֶׁל הַנָחוּץ", אָמַר סוֹקְרָטֵס. "הַאִם נָחוּץ שֶׁתְסַפֵּר לִי אֶת הַדָבָר הָרַע שֶׁשָׁמַעְתָ מִמִישֶׁהוּ עַל חֲבֵרִי וְשֶׁאַתָה לֹא בָּטוּחַ שֶׁהוּא אֱמֶת?"
הַיָדִיד חָשַׁב רֶגַע וְאַחַר כָּךְ אָמַר: "לֹא, זֶה לֹא נָחוּץ".
"אִם כָּךְ, אַל תְסַפֵּר לִי", אָמַר סוֹקְרָטֵס.
פַּעַם, לִפְנֵי שָׁנִים רַבּוֹת לַאֲנָשִׁים בִּכְלָל לֹא הָיוּ שְׁמוֹת מִשְׁפָּחָה. אֲנָשִׁים נִקְרְאוּ בַּשֵׁמוֹת הַפְּרָטִיִים שֶׁלָהֶם וְלִפְעָמִים הוֹסִיפוּ לִשְׁמָם גַם אֶת שֵׁם אַבָּא שֶׁלָהֶם. שְׁמַעְתֶם בְּמִקְרֶה עַל אִישׁ שֶׁקָרְאוּ לוֹ אַבְנֵר וּלְאַבָּא שֶׁלוֹ קָרְאוּ נֵר? זֶהוּ אַבְנֵר בֶּן נֵר, וּמְסֻפָּר עָלָיו בַּתָנָ"ךְ. 
בְּמֶשֶׁךְ הַשָׁנִים נוֹסְפוּ לַאֲנָשִׁים שְׁמוֹת מִשְׁפָּחָה.
בַּאֲרָצוֹת שׁוֹנוֹת יֵשׁ שְׁמוֹת מִשְׁפָּחָה שׁוֹנִים.
בַּמַפָּה שֶׁלִפְנֵיכֶם תוּכְלוּ לִמְצֹא מֵידָע מְעַנְיֵן עַל שְׁמוֹת מִשְׁפָּחָה בָּעוֹלָם.
גוֹקוֹ הוּא חָבֵר מִכּוֹכָב אַחֵר. גוֹקוֹ הִגִיעַ לְכַדוּר הָאָרֶץ, וְהוּא רוֹצֶה לְהַכִּיר חֲבֵרִים מִכָּל מִינֵי מְקוֹמוֹת בָּעוֹלָם. הַרְבֵּה יְלָדִים שָׁלְחוּ לוֹ מִכְתָבִים, אֲבָל שָׁכְחוּ לִכְתֹב בְּאֵיזוֹ מְדִינָה הֵם גָרִים. עִזְרוּ לְגוֹקוֹ לְהַשְׁלִים אֶת הַמֵידָע. מִתְחוּ קַו בֵּין שֵׁם הַיֶלֶד וּבֵין שֵׁם הַמְדִינָה שֶׁבָּהּ הוּא גָר.
בַּסְבִיבָה שֶׁלָנוּ יֵשׁ דְבָרִים שֶׁהֵם חֵלֶק מֵעוֹלַם הַטֶבַע וְיֵשׁ דְבָרִים שֶׁעָשָׂה הָאָדָם. 
לִפְנֵיכֶם שְׁתֵי תְמוּנוֹת מְאוֹד מְיֻחָדוֹת.
אֶפְשָׁר לְהִכָּנֵס לַתֹכֶן וּלְסַיֵר בָּאֵזוֹר שֶׁבּוֹ הֵן צֻלְמוּ.
הַתְמוּנוֹת צֻלְמוּ בְּמֵיזָם שֶׁנִקְרָא "גוּגְל סְטְרִיט וִיוּ". צַלָמִים מִכָּל הָעוֹלָם טִיְלוּ וְצִלְמוּ בְּמַצְלֵמוֹת מְיֻחָדוֹת כִּמְעַט כָּל מָקוֹם בָּעוֹלָם.
הַתְמוּנוֹת שֶׁתִּרְאוּ צֻלְמוּ בָּעִיר תֵל אָבִיב.
הָאַחַת צֻלְמָה בַּפָּארְק, הָאַחֶרֶת צֻלְמָה בְּשֶׁטַח בָּנוּי בִּצְפִיפוּת.
הַקְלִיקוּ עִם הָעַכְבָּר עַל הַתְמוּנוֹת וְהַתְחִילוּ לְסַיֵר. מִתְקַדְמִים עִם כָּל לְחִיצָה עַל הָעַכְבָּר.
אָמְרוּ עָלָיו עַל הִלֵּל הַזָּקֵן, שֶׁבְּכָל יוֹם וָיוֹם הָיָה עוֹבֵד וּמְקַבֵּל שָׂכָר.
חֲצִי מִן הַשָּׂכָר שֶׁהִרְוִיחַ הָיָה נוֹתֵן לְשׁוֹמֵר בֵּית הַמִּדְרָשׁ וַחֲצִי הָיָה לוֹקֵחַ לַפַּרְנָסָה שֶׁלוֹ וְשֶׁל בְּנֵי מִשְׁפַּחְתוֹ.
פַּעַם אַחַת לֹא מָצָא עֲבוֹדָה, וְלֹא הָיָה לוֹ כֶּסֶף וְלֹא הִרְשָׁה לוֹ שׁוֹמֵר בֵּית הַמִּדְרָשׁ לְהִכָּנֵס;
עָלָה וְטִפֵּס עַל הַגַּג וְיָשַׁב עַל פִּי הָאֲרֻבָּה, כְּדֵי שֶׁיִּשְׁמַע דִּבְרֵי חֲכָמִים.
אוֹתוֹ הַיּוֹם תְּקוּפַת טֵבֵת הָיְתָה, וְיָרַד עָלָיו שֶׁלֶג מִן הַשָּׁמַיִם.
כְּשֶׁעָלָה עַמּוּד הַשַּׁחַר אָמְרוּ הַחֲכָמִים זֶה לָזֶה:
בְּכָל יוֹם הַבַּיִת מֵאִיר וְאִלּוּ הַיּוֹם אָפֵל, הַאִם יוֹם מְעֻנָּן הוּא הַיּוֹם?
הֵצִיצוּ עֵינֵיהֶם וְרָאוּ דְּמוּת אָדָם בָּאֲרֻבָּה.
עָלוּ וּמָצְאוּ עָלָיו שֶׁלֶג בְּגֹבַהּ רַב.
הוֹרִידוּ אוֹתוֹ מִן הַגַּג וְהוֹשִׁיבוּהוּ מוּל הַמְּדוּרָה.
לִפְנֵיכֶם סִפּוּר הַקָשׁוּר בִּלְמִידָה. גִבּוֹר הַסִפּוּר - הִלֵּל הַזָּקֵן, אָהַב לִלְמֹד תוֹרָה. יוֹם אֶחָד לֹא הָיָה לְהִלֵּל כֶּסֶף לְשַׁלֵם כְּדֵי לְהִכָּנֵס לְבֵית הַמִדְרָשׁ - מָקוֹם שֶׁבּוֹ לוֹמְדִים תוֹרָה. מָה עָשָׂה וּמָה קָרָה? עַל כָּךְ תִקְרְאוּ בַּסִפּוּר שֶׁלִפְנֵיכֶם.

בְּשֶׁקֶט-בְּשֶׁקֶט,
בִּשְׁבִיל שֶׁהִיא
לֹא תִּתְעוֹרֵר.
דַמְיְינוּ שֶׁקָרָה לָכֶם מִקְרֶה דוֹמֶה לַמִקְרֶה שֶׁל הַיֶלֶד מֵהַשִׁיר, וְגַם אַתֶם הֵעַרְתֶם אֶת אִמָא אוֹ אֶת אַבָּא וְהִצְטַעַרְתֶם עַל כָּךְ.
כִּתְבוּ לָהֶם מִכְתָב
בְעוּ בַּשִׁיר אֶת הָאוֹת ש' בְּכָל מָקוֹם שֶׁבּוֹ הִיא מוֹפִיעָה.
קִרְאוּ אֶת הַשִׁיר בְּקוֹל וְהַדְגִישׁוּ בַּקְרִיאָה אֶת הַמִלִים שֶׁבָּהֶן מוֹפִיעָה הָאוֹת ש'.
מָה הַצְלִילִים מַזְכִּירִים לָכֶם?
מָה גִילִיתֶם: מַדוּעַ הַמְשׁוֹרֵר יְהוּדָה אַטְלַס הִשְׁתַמֵשׁ בָּאוֹת ש' הַרְבֵּה פְּעָמִים?
שַׁעֲרוּ, מַדוּעַ הַמְשׁוֹרֵר בָּחַר לִכְתֹב אֶת הַמִילָה "בְּשֶׁקֶט" פַּעֲמַיִם?
הִתְאַמְנוּ יַחַד בִּקְרִיאַת הַשִׁיר זֶה לָזֶה כַּהֲכָנָה לִקְרִיאָה לִפְנֵי הַכִּיתָה.
הַקְלִיטוּ אֶת קְרִיאַתְכֶם וְהַאֲזִינוּ לָה.
בָּעֶרֶב הָיְתָה אֶצְלֵנוּ הַפְסָקַת חַשְׁמַל. אוֹי! הַמְנוֹרוֹת כָּבוּ פִּתְאוֹם.
הַמַחְשֵׁב לֹא פָּעַל, הַטֵלֵוִיזְיָה לֹא פָּעֲלָה. אֲנִי לֹא יָכֹלְתִי לְהָכִין שִׁעוּרִים,
אִמָא לֹא יָכְלָה לִכְתֹב בַּמַחְשֵׁב, אַבָּא לֹא יָכֹֹל לִקְרֹא עִתוֹן, אֲחוֹתִי לֹא יָכְלָה
לִצְפּוֹת בַּטֵלֵוִיזְיָה.
חָשַׁבְתִי: אִי אֶפְשָׁר לַעֲשׂוֹת כְּלוּם בְּלִי חַשְׁמַל, אֲבָל זֶה לֹא הָיָה נָכוֹן.
כֻּלָנוּ הִתְיַשַׁבְנוּ בַּמִטָה שֶׁל הַהוֹרִים, סִפַּרְנוּ סִפּוּרִים מַפְחִידִים בַּחֹשֶׁךְ,
שַׁרְנוּ שִׁירִים מַצְחִיקִים וְהִתְחַמַמְנוּ לָנוּ יַחַד מִתַחַת לַשְׂמִיכָה.
פִּתְאוֹם נִדְלַק הָאוֹר וְשׁוּב הָיָה חַשְׁמַל: הַמַחְשֵׁב וְהַטֵלֵוִיזְיָה פָּעֲלוּ,
וְכָל אֶחָד חָזַר לָעִסוּק שֶׁלוֹ.
וַאֲנִי חָשַׁבְתִי: הַלְוַאי שֶׁגַם מָחָר תִהְיֶה הַפְסָקַת חַשְׁמַל!
קְרָאתֶם שִׁיר וְסִפּוּר עַל יְלָדִים וְהוֹרִים.
בַּחֲרוּ אֶחָד מֵהַנוֹשְׂאִים וְכִתְבוּ עָלָיו:
דְבָרִים שֶׁאַתֶם אוֹהֲבִים לַעֲשׂוֹת עִם בְּנֵי הַמִשְׁפָּחָה.
אֵירוּעַ מְיֻחָד שֶׁקָרָה לָכֶם וּלְמִשְׁפַּחְתְכֶם.
מִקְרֶה שֶׁבּוֹ עֲשִׂיתֶם מַשֶׁהוּ לְמַעַן בֶּן מִשְׁפָּחָה אוֹ בַּת מִשְׁפָּחָה אוֹ שֶׁהֵם עָשׂוּ לְמַעַנְכֶם.
עֲשֶׂרֶת הַדִיבְּרוֹת הֵם רְשִׁימָה שֶׁל עֲשָׂרָה צִיווּיִים. עַל פִּי הַתוֹרָה, אֱלוֹהִים מָסַר אֶת עֲשֶׂרֶת הַדִיבְּרוֹת לְמֹשֶׁה כְּשֶׁהֵם חֲרוּתִים עַל לוּחוֹת אֶבֶן. מֹשֶׁה נָתַן אֶת לוּחוֹת הַבְּרִית לִבְנֵי יִשְׂרָאֵל בְּמַעֲמַד הַר סִינִי, זְמַן קָצָר לְאַחַר שֶׁבְּנֵי יִשְׂרָאֵל יָצְאוּ מִמִצְרַיִם. בְּנֵי יִשְׂרָאֵל הִסְכִּימוּ לְקַבֵּל אֶת הַדִיבְּרוֹת, וּלְקַיֵים אֶת מָה שֶׁכָּתוּב בָּהֶם
הַנַּחֲלִיאֵלִי הוּא צִפּוֹר שִׁיר. שְׁמוֹ נִתַּן לוֹ כִּי הוּא מַרְבֶּה לְהִסְתּוֹבֵב בְּקִרְבַת נַחְלִים וּמְקוֹרוֹת מַיִם. הוּא נֶחְשַׁב לְאֶחָד מִמְּבַשְּׂרֵי הַסְּתָו, כִּי הוּא נוֹדֵד וּמַגִּיעַ אֵלֵינוּ בַּסְּתָו. הוּא נוֹדֵד וּבָא מֵאַרְצוֹת אֵירוֹפָּה. בְמוֹלַדְתוֹ בַּסְּתָו כְּבָר קַר וּבְאַרְצֵנוּ עֲדַיִן חָם. הַנַּחֲלִיאֵלִי נוֹדֵד בְּלַהֲקוֹת בַּלֵּילוֹת וּבְמֶשֶׁךְ הַיּוֹם הוּא נָח וְאוֹכֵל.
הַנַחֲלִיאֵלִי מִתְעוֹפֵף, עוֹלֵה וְיוֹרֵד, יוֹרֵד וְעוֹלֵה. הוּא רָץ עַל פְּנֵי הַקַרְקַע כְּשֵזְנַבוֹ הַאָרוֹךְ מִתְנוֹעֵעַ בְּלִי הֶרֶף. תְּנוּעוֹת הַזָּנָב מְשַׁמְּשׁוֹת לוֹ גַּם לְנִוּוּט וְגַם לְקֶשֶׁר בֵּין נַחֲלִיאֵלִי אֶחָד לַשֵּׁנִי.
לַנַּחֲלִיאֵלִי מָקוֹר אָרֹךְ וְדַק. מָקוֹר זֶה מַתְאִים לְצַיִד שֶׁל חֲרָקִים קְטַנִּים כְּמָזוֹן. וּכְשֶׁהַנַּחֲלִיאֵלִי רָעֵב , הוּא מִתְרוֹצֵץ עַל פְּנֵי הַקַרְקַע לְחַפֵּשׂ מָזוֹן. מִדֵי פַּעַם הוּא מִתְעוֹפֵף בַּאֲוִיר כְּדֵי לָצוּד לוֹ זְבוּבִים וְיָתוּשִים.
לְנַחֲלִיאֵלִי רַגְלַיִם בְּעָלֵי אַרְבַּע אֶצְבָּעוֹת אֲרֻכּוֹת שֶׁנּוֹעֲדוּ לִתְנוּעָה מְהִירָה עַל פְּנֵי הַקַּרְקַע.
הַנַחֲלִיאֵלִי בָּא אֵלֵינוּ מִדֵי שָנָה, אַךְ הוּא לֹא בּוֹנֶה כָּאן את קֵינוֹ. רַק כְּשֶׁיָּשׁוּב הַנַּחֲלִיאֵלִי לְאַרְצוֹ, לִקְרַאת הָאָבִיב, הוּא יְחַפֵּשׂ לוֹ מָקוֹם לְיַד מַיִם כְּדֵי לִבְנוֹת אֶת הַקֵּן. שָׁנִי בְּנֵי הַזּוּג בּוֹנִים אֶת הַקֵּן וּלְאַחַר שֶׁהַנְּקֵבָה מַטִּילָה אֶת הַבֵּיצִים, הַזָּכָר עוֹזֵר בְּגִדּוּלָם.
מִשֶּׁבָּא הַאָבִיב מִתְאַסְפִים הַנַחֲלִיאֵלִים לְמַחֲנֵה גָדוֹל וְנוֹדְדִים חֲזָרָה אֵל מוֹלַדְתָם אֲשֶׁר בַּצָפוֹן.
אִישׁ לֹא יֹאמַר עַל נִצָּה וְעַל דִּיצָה שֶׁהֵן אֲחָיוֹת; כָּל כָּךְ רַב הַהֶבְדֵּל בֵּינֵיהֶן. נִצָּה בַּת חָמֵשׁ שָׁנִים וְדִיצָה בַּת אַרְבַּע. לְנִצָּה שְׂעָרוֹת שְׁחוֹרוֹת וְהֵן סְדוּרוֹת לָהּ תַּלְתַּלִים וְעֵינֶיהָ שְׁחוֹרוֹת מִשְּׁחוֹר; וּלְדִיצָה עֲטֶרֶת זָהָב לְרֹאשָׁהּ וְעֵינֶיהָ תְּכֻלּוֹת כִּתְכֵלֶת הַשָּׁמַיִם לָטֹהַר. נִצָּה דַּקַּת גֵּו וּתְמִירָה כְּעֹפֶר הָאַיָּלִים, וְדִיצָה עֲגַלְגֹּלֶת וּזְרִיזָה כְּחֻלְדַּת הַסְּנָאִים.
גַּם בְּאָפְיָן שׁוֹנוֹת הָאֲחָיוֹת הַקְּטַנּוֹת. נִצָּה שְׁלֵוָה מְאֹד וְאוֹהֶבֶת סֵדֶר וְנִקָּיוֹן. הַבֻּבָּה שֶׁלָּהּ מְקֻשֶּׁטֶת תָּמִיד, כְּמוֹ לִפְנֵי טִיּוּל; וְהַצַּעֲצוּעִים שְׁלֵמִים וּמַבְרִיקִים, כְּאִלּוּ הוּבְאוּ זֶה עַתָּה מֵהֶחָנוּת.
אַךְ דִּיצָה מְלֵאָה תְּנוּעָה כְּפַרְפָּר. אֵין הִיא יוֹדַעַת מְנוּחָה: אוֹ שֶׁהִיא צוֹחֶקֶת וְשָׁרָה, אוֹ שֶׁהִיא קוֹפֶצֶת, מְרַחֶפֶת וּמִתְגַּלְגֶּלֶת מֵחֶדֶר לְחֶדֶר וּמְמַלְּאָה אֶת הַבַּיִת דִּיצָה וְחֶדְוָה. אֵין הִיא נוֹתֶנֶת דַּעְתָּהּ הַרְבֵּה עַל הַצַּעֲצוּעִים שֶׁלָּהּ. לְאַחַר שֶׁשִּׂחֲקָה בָּהֶם זְמַן מָה, מַנִּיחָה הִיא אוֹתָם וְאֵינָהּ מִתְעַנְיֶנֶת בָּהֶם עוֹד.
כָּל הַיּוֹם מְשַׂחֲקוֹת שְׁתֵּי הָאֲחָיוֹת יַחַד בְּשָׁלוֹם, מְפַטְפְּטוֹת וְהוֹגוֹת כִּשְׁתֵּי יוֹנִים. אוּלָם יֵשׁ וּבֵין שְׁתֵּי הַיּוֹנִים נוֹפֵל סִכְסוּךְ קָטָן וְאָז הֵן נֶהְפָּכוֹת לִשְׁתֵּי חֲתַלְתּוּלוֹת רַגְזָנִיּוֹת. הַגּוֹרֶמֶת לָרִיב הִיא עַל פִּי רֹב דִּיצָה. אוֹהֶבֶת הִיא לְקַנְטֵר אֶת אֲחוֹתָהּ לֹא מִתּוֹךְ רֹעַ לֵב, חָלִילָה, אֶלָּא מִתּוֹךְ שׁוֹבְבוּת סְתָם. כְּשֶׁנִּצָּה מַצִּיעָה לְשַׂחֵק בְּמַחְבּוֹאִים, אוֹמֶרֶת דִּיצָה:
“דַּוְקָא לֹא! נְשַׁחֵק בְּקֻבִּיּוֹת!”
וּכְשֶׁאוֹמֶרֶת נִצָּה לְשַׂחֵק בְּקֻבִּיּוֹת, מִתְנַגֶּדֶת לָזֶה דִּיצָה וְאוֹמֶרֶת:
“דַּוְקָא לֹא! נְשַׂחֵק בְּמַחְבּוֹאִים!”
“דַּוְקָנִית אַחַת!” אוֹמֶרֶת נִצָּה וְנַעֲשֵׂית בְּרֹגֶז. “גַּאַוְתָנִית אַחַת!” אוֹמֶרֶת דִּיצָה וּמִתְנַפַּחַת.
אִמָּא מִתְעָרֶבֶת לִפְעָמִים בְּרִיב בְּנוֹתֶיהָ.
“נִצָּה!” אוֹמֶרֶת הִיא, “הֲלֹא אַתְּ הַגְּדוֹלָה, עָלַיִךְ אֵפוֹא לְוַתֵּר לָהּ לְיַלְדֹּנֶת זוֹ”.
“אֲנִי מְוַתֶּרֶת כָּל פַּעַם”, מִתְרָעֶמֶת נִצָּה, “אַךְ הִיא דַּוְקָנִית גְּדוֹלָה”.
“בּוּשָׁה גְּדוֹלָה וּכְלִמָּה!” אוֹמֶרֶת אִמָּא, “שְׁתֵּי אֲחָיוֹת נֶחְמָדוֹת וְאֵינָן יְכוֹלוֹת לִחְיוֹת בְּשָׁלוֹם”.
אַךְ מִי שֶׁחוֹשֵׁב, שֶׁאֵין הָאֲחָיוֹת אוֹהֲבוֹת אִשָּׁה אֶת רְעוּתָהּ, טוֹעָה טָעוּת גְּדוֹלָה. יְנַסֶּה נָא מִי שֶׁהוּא לֵאמֹר מִלָּה רָעָה עַל דִּיצָה, מִיָּד תַּעֲמֹד נִצָּה לִימִינָה וְתוֹדִיעַ:
“דִּיצָה הִיא יַלְדֹּנֶת חֲבִיבָה וַחֲמוּדָה מְאֹד”.
וְאִם מִי שֶׁהוּא יְדַבֵּר רֵעוּת בְּנִצָּה, מִיָּד תָּגֵן עָלֶיהָ דִּיצָה וְתֹאמַר:
“נִצָּה שֶׁלִּי מְתוּקָה מִכָּל מֹתֶק שֶׁבָּעוֹלָם”.
יוֹם אֶחָד חָלְתָה נִצָּה וְנָפְלָה לְמִשְׁכָּב. הָיָה לָהּ חֹם גָּבוֹהַּ וְהָרוֹפֵא אָמַר, שֶׁבְּמֶשֶׁךְ יָמִים אֲחָדִים אֵין הַיְלָדוֹת צְרִיכוֹת לְהִמָּצֵא בְּחֶדֶר אֶחָד.
דִּיצָה הָיְתָה עֲצוּבָה מְאֹד וּבִקְּשָׁה מֵאֵת אִמָּא, שֶׁתִּתֵּן לָהּ לְכָל הַפָּחוֹת לִרְאוֹת אֶת אֲחוֹתָהּ מֵרָחוֹק.
“הֲלֹא אַתֶּן רָבוֹת כָּל הַיָּמִים”, אָמְרָה לָהּ הָאֵם.
“זֶה לֹא כְלוּם”, הֵשִׁיבָה דִּיצָה, “זֶה סְתָם כָּךְ, בִּצְחוֹק. אֲנִי אוֹהֶבֶת אֶת נִצָּה שֶׁלִּי”.
וּשְׁתֵּי פְּנִינִים נָשְׁרוּ מֵעֵינֶיהָ הַתְּכֻלּוֹת.
“טוֹב”, אָמְרָה אִמָּא, “אֲנִי אַקְרִיב אוֹתָךְ לַדֶּלֶת וְתוּכְלִי לִרְאוֹת אוֹתָהּ, אַךְ אַל תְּדַבְּרִי אִתָּהּ: הִיא זְקוּקָה לִמְנוּחָה”.
דִּיצָה שָׂמְחָה מְאֹד וְשָׁלְחָה לַאֲחוֹתָהּ הַחוֹלָה נְשִׁיקַת אֲוִיר וּבַת-צְחוֹק מְתוּקָה.
כָּל יְמֵי מַחֲלָתָהּ שֶׁל נִצָּה הָיְתָה דִּיצָה שְׁקֵטָה וּרְצִינִית וְהִתְהַלְּכָה עַל קְצוֹת אֶצְבְּעוֹתֶיהָ שֶׁלֹּא לְהַרְבּוֹת שֶׁאוֹן. בְּכָל בֹּקֶר הָיְתָה קוֹטֶפֶת שׁוֹשַׁנָּה בַּגִּנָּה וְשׁוֹלַחַת אוֹתָהּ עַל-יְדֵי אִמָּא לַאֲחוֹתָהּ.
כְּשֶׁיָּרְדָה נִצָּה מִמִּטָּתָהּ, הָיְתָה דִּיצָה מְאֻשָּׁרָה מְאֹד. כְּדֵי לְשַּׂמֵחַ מְאֹד אֶת אֲחוֹתָהּ הָאֲהוּבָה עָשְׂתָה לְפָנֶיהָ כָּל מִינֵי מוֹפְתִים: קָפְצָה וְרָקְדָה וְהִתְגַּלְגְּלָה כְּכַדּוּר וְשָׁרָה שִׁירִים. וּכְשֶׁרָאֲתָה בַּת-צְחוֹק עַל פְּנֵי נִצָּה, צָחֲקָה בִּצְחוֹקָהּ הַמְצַלְצֵל וְאָמְרָה: “כַּמָּה טוֹב שֶׁאַתְּ כְּבָר בְּרִיאָה, נִצָּה! דַּי, מֵהַיּוֹם וָהָלְאָה לֹא נָרִיב עוֹד. מַה שֶּׁתְּבַקְּשִׁי אֶעֱשֶׂה, אֲבָל רְאִי, אַל תֶּחֱלִי עוֹד פָּעַם”. “טוֹב”, הִבְטִיחָה נִצָּה. וּשְׁתֵּי הָאֲחָיוֹת הַקְּטַנּוֹת חַיּוֹת שׁוּב כִּשְׁתֵּי יוֹנִים חֲבִיבוֹת. אַךְ אִם דִּיצָה שׁוֹכַחַת פַּעַם אֶת הַבְטָחָתָהּ וּמַתְחִילָה לְהִתְעַקֵּשׁ, אוֹמֶרֶת לָהּ נִצָּה:
“דִּיצָה, אַל תִּהְיִי יַלְדָּה רָעָה, פֶּן אֶחֱלֶה עוֹד פָּעַם”. מִיָּד נִכְנַעַת דִּיצָה וְאוֹמֶרֶת: “זֶה רַק בִּצְחוֹק, נִצָּה! רַק בִּצְחוֹק!”
