א געשעפט
א געשעפט (בְּעִבְרִית: הָעִסְקָה) הוּא שְׁמוֹ שֶׁל סֶרֶט פְּעֻלָּה בְּיִידִישׁ מִשְּׁנַת 2005, שֶׁהוּפַק, צֻלַּם, וּבֻיַּם עַל יְדֵי יְהוּדִים חֲרֵדִים בִּמְנֻסֵּי, נְיוּ יוֹרְק. זֶהוּ הַסֶּרֶט הָרִאשׁוֹן בְּיִידִישׁ שֶׁיּוֹצֵא מִקְּרַב חוּגִים חֲרֵדִיִּים, וְאַחַד הַסְּרָטִים הָרִאשׁוֹנִים שֶׁהוּפְקוּ בְּיִידִישׁ מֵאָז שְׁנוֹת הַ-40.
הַסֶּרֶט צֻלַּם בְּאַרְצוֹת הַבְּרִית בְּרָמַת הֲפָקָה גְּבוֹהָה יַחֲסִית, וְזָכָה שֵׁם לַתְּהוּדָה. כְּמוֹ כָּל סֶרֶט חֲרֵדִי טִיפּוּסִי, צֶוֶת הַשַּׂחְקָנִים בְּסֶרֶט מֻרְכַּב מַגְבְּרִים בִּלְבַד.
"א געשעפט" הוּא סֶרֶט שֹׁד, הַמְּגוֹלֵל אֶת סִפּוּרוֹ שֶׁל פֶּרֶץ, סוֹחֵר שֶׁחַיָּיו מִשְׂתָּרְעִים עַל קַו הַתֶּפֶר בֵּין הָרֶקַע הַחֲרֵדִי שֶׁלּוֹ לְבֵין חַיֵּי עוֹלָם תַּחְתּוֹן. פָּרַץ מוּכָן לְהִשְׁתַּמֵּשׁ בְּאַלִּימוּת כְּדֵי לְנַהֵל אֶת עֲסָקָיו הַמְּפֻקְפָּקִים, וְהַסֶּרֶט עוֹקֵב אַחַר הַקַּרְיֶרָה הַפְּלִילִית שֶׁלּוֹ. בַּמַּאי הַסֶּרֶט, חָסִיד ויז'ניץ, צִיֵּן שֶׁהוּא הֻשְׁפַּע מִכַּמָּה סִרְטֵי מֵינְסְטְרִים אָמֶרִיקָאִיִּים, כּוֹלֵל "לֵב אַמִּיץ" שֶׁל הַבַּמַּאי מָל גיבסון. עוֹרֵךְ הַסֶּרֶט, רולנד מילמן, עֵרֶךְ מִסְפַּר סִרְטֵי יִידִישׁ (סֶרֶט כֹּשֶׁר "שְׁוַיְץ" וְסֶרֶט בִּשּׁוּל "נוּ שְׁמַלְץ"), וּלְפִיו הַמִּשְׂחָק בַּסֶּרֶט "לֹא נִפְלָא, אֲבָל הַצִּלּוּם לֹא רַע".
סְבִיב הַשִּׁוּוּק הַוִּירָלִי וְהָאִינְטֶרְנֶטִי שֶׁל הַסֶּרֶט, שֶׁהֻפְנָה בְּעִקָּר לְקָהָל חֲרֵדִי, הִתְעוֹרֵר פֻּלְמוּס כַּאֲשֶׁר רַבָּנִים אָסְרוּ לִצְפּוֹת בּוֹ בְּשֶׁל חֲשָׁשָׁם מַחְדִּירָה שֶׁל הַשְׁפָּעוֹת הַקּוֹלְנוֹעַ וְהַתֵּאַטְרוֹן לַצִּבּוּר הַחֲרֵדִי. הַסּוֹצְיוֹלוֹג חַיִּים וַקְסְמָן מֵאוּנִיבֶרְסִיטַת ראטגרס שֵׂעָר כִּי הָעֻבְדָּה שֶׁהַסֶּרֶט נַעֲשָׂה בְּיִידִישׁ הוֹסִיפָה לַחֲשָׁשׁוֹת, מִשּׁוּם שֶׁ"אֲנָשִׁים עֲלוּלִים לַחְשֹׁב אִם זֶה בְּיִידִישׁ, זֶה עוֹשֶׂה אֶת זֶה כָּשֵׁר". עַל-פִּי וַקְסְמָן, הַהוֹפָעָה שֶׁל סֶרֶט בְּיִידִישׁ הָיָה חֵלֶק מִשִּׁנּוּי מַשְׁמָעוּתִי בָּעוֹלָם הַחֲרֵדִי אֲשֶׁר הֵחֵל לְאַמֵּץ מְגֻוָּן שֶׁל מוּצְרֵי תַּרְבּוּת, כּוֹלֵל מוּזִיקָה, סִפֹּרֶת, וַאֲפִלּוּ תֵּאַטְרוֹן, שֶׁהַדּוֹרוֹת הַקּוֹדְמִים לָהֶם נִמְנְעוּ מֵהֶם.[1]
הַסֶּרֶט מָשַׁךְ גַּם תְּשׂוּמַת לֵב בְּחוּגִים יְהוּדִיִּים חִלּוֹנִיִּים בַּעֲלֵי עִנְיָן בְּיִידִישׁ, מִשּׁוּם שֶׁמְּדֻבָּר בַּמִּסְמָךְ בְּיִידִישׁ בֶּן זְמַנֵּנוּ.[2] הוּא הֻקְרַן בֵּין הַיֶּתֶר בְּ"אַשְׁכְּנַז פֶסְטִיבָל" בְּטוֹרוֹנְטוֹ וּבְפֶסְטִיבַל הַיִּידִישׁ בְּאוּנִיבֶרְסִיטַת אָרִיזוֹנָה.[3] בִּשְׁנַת 2009 יָצָא הַסֶּרֶט בְּיִשְׂרָאֵל תַּחַת הַשֵּׁם "הָעִסְקָה", וְצוֹרֵף לוֹ דִּבּוּב בְּעִבְרִית.
