אֶפְרַיִם אָשְׁרִי
הָרַב אֶפְרַיִם אָשְׁרִי (מְבֻטָּא, וּלְעִתִּים אַף נִכְתַּב כְּאָשְׁרִי; ה'תרע"ד, 1914 - 28 בְּסֶפְּטֶמְבֶּר 2003) הָיָה תַּלְמִיד חָכָם, פּוֹסֵק וְרֹאשׁ יְשִׁיבָה. עָסַק רַבּוֹת בַּשּׁוֹאָה, וְחִבֵּר סֵפֶר שו"ת שֶׁעָסַק בִּשְׁאֵלוֹת הִלְכָתִיּוֹת שֶׁנָּגְעוּ לַשּׁוֹאָה, בַּשֵּׁם מִמַּעֲמַקִּים.
בִּיוֹגְרַפְיָה
אָשְׁרִי נוֹלַד בקופישוק שֶׁבְּלִיטָא לְרַב דֹּב אָשְׁרִי. הִתְחַנֵּךְ בִּישִׁיבַת סלובודקה, בִּישִׁיבַת פוניבז' וּבְתַלְמוּד תּוֹרָה בקלם. בִּשְׁנַת ה'תרצ"ד (1934) נִסְמַךְ לַהוֹרָאָה בִּידֵי רַבָּהּ שֶׁל קוּבָּנֶה, הָרַב אַבְרָהָם דֹּב כהנא שַׁפִּירָא וּבִידֵי רַבִּי לייב רוּבִּין מווילקומיר, רַבֵּי עֶזְרָא אלטשולר, ר' אַבְרָהָם זְאֵב הלר ממריאמפולה ור' בָּרוּךְ הורוביץ מאלקסוט. לְאַחַר מִכֵּן שִׁמֵּשׁ כְּרָב בסלובודקה וְלִמֵּד בְּבֵית מִדְרָשׁ מְקוֹמִי.[1]
בְּמֶשֶׁךְ מִלְחֶמֶת הָעוֹלָם הַשְּׁנִיָּה שָׁהָה הָרַב אָשְׁרִי בְּגֶטוֹ קוּבָּנֶה, שֵׁם שִׁמֵּשׁ יַד יְמִינוֹ שֶׁל רַב הָעִיר - הָרַב אַבְרָהָם דֹּב כהנא שַׁפִּירָא - וְעָמַד בְּרֹאשׁ הָאִרְגּוּן "תִּפְאֶרֶת בַּחוּרִים" בְּגֶטוֹ קוּבָּנֶה. בִּתְקוּפָה זוֹ רֹב רַבָּנֵי הַגֶּטוֹ נִרְצְחוּ בִּידֵי הַנָּאצִים וְהַלִּיטָאִים, וְהָרַב שַׁפִּירָא עַצְמוֹ כְּבָר הָיָה חוֹלֶה וְחַלָּשׁ וְנִבְצָר מִמֶּנּוּ לְתַפְקֵד כְּפוֹסֵק בְּאֹפֶן מָלֵא, וְעַל כֵּן הָרַב שַׁפִּירָא בִּקֵּשׁ מֵהָרַב אָשְׁרִי שֶׁיַּעֲנֶה עַל שְׁאֵלוֹת הִלְכָתִיּוֹת בִּמְקוֹמוֹ. אֶת הַדַּפִּים בָּהֶם נִכְתְּבוּ הַתְּשׁוּבוֹת שֶׁמַּר הָרַב אָשְׁרִי טְמוּנִים בָּאֲדָמָה, וְרַק לְאַחַר הַמִּלְחָמָה קִבֵּץ אוֹתָם לְסַפֵּר שְׁאֵלוֹת וּתְשׁוּבוֹת "מִמַּעֲמַקִּים".
לְאַחַר הַמִּלְחָמָה הָרַב אָשְׁרִי נוֹתָר הָרַב הַיָּחִיד בַּקּוּבָּנֶה, וְהִמְשִׁיךְ שֵׁם בִּפְעִילוּת צִבּוּרִית לְמַעַן הַפְּלִיטִים. מְאֻחָר יוֹתֵר הוּא הִגֵּר לְאוֹסְטְרִיָּה וְהֵקִים יְשִׁיבָה לְיַד זלצבורג, וּמִשָּׁם לְרוֹמָא, שֵׁם שִׁמֵּשׁ כְּרֹאשׁ יְשִׁיבַת "מֵאוֹר הַגּוֹלָה". לְאַחַר תְּקוּפָה זוֹ הַיֶּגֶר לְאַרְצוֹת הַבְּרִית, שָׁם גַּם פִּרְסֵם אֶת סְפָרָיו. בְּמֶשֶׁךְ עֶשְׂרוֹת שָׁנִים הִנְהִיג אֶת קְהִלַּת בֵּית הַמִּדְרָשׁ הַגָּדוֹל בַּלּוּאָר איסט סַיָּד שֶׁל מנהטן בִּנְיוּ יוֹרְק.
הָרַב אָשְׁרִי נִפְטַר בב' תִּשְׁרֵי שְׁנַת ה'תשס"ד, 2003, בִּנְיוּ יוֹרְק. לְאַחַר הַלְוָיָה מְכֻבֶּדֶת בחו"ל, הוּא הָעֳלָה לִקְבוּרָה בִּירוּשָׁלַיִם.
פְּסִיקוֹתָיו
הַגֶּטוֹ זִמֵּן לְרַב אָשְׁרִי שְׁאֵלוֹת לֹא שִׁגְרָתִיּוֹת, וְכֵן הֵבִיא נוֹשְׂאִים תֵּאוֹרֶטִיִּים שׁוֹנִים לִידֵי מַשְׁמָעוּת מַעֲשִׂית. כָּךְ לְדֻגְמָה הוּא נִשְׁאַל הַאִם נִתָּן לְבָרֵךְ בְּבִרְכוֹת הַשַּׁחַר בִּרְכַּת "שֶׁלֹּא עַשְּׁנִי עָבַד", מִכֵּיוָן שֶׁהַצְהָרָה זוֹ עָמְדָה בַּסְּתִירָה לְמַצַּב הַיְּהוּדִים בַּגֶּטוֹ. בִּתְשׁוּבָתוֹ כּוֹתֵב הָרַב אָשְׁרִי: "עִקַּר הַבְּרָכָה הַזֹּאת נַתֵּקְנָה לֹא עַל עַבְדוּת הַגּוּף, אֶלָּא עַל עַבְדוּת הַנֶּפֶשׁ... וְאַדְּרַבָּה, דַּוְקָא כָּעֵת חוֹבָה עָלֵינוּ לְבֶרֶךְ בְּרָכָה זֹאת לְמַעַן יַכִּירוּ אוֹיְבֵינוּ וּמִנְּדָנוֹ, שֶׁלַּמְרוֹת שֶׁאָנוּ נְתוּנִים בְּיָדָם לַעֲשׂוֹת בָּנוּ כִּרְצוֹנָם הָרַע, בְּכָל זֹאת רוֹאִים אָנוּ אֶת עַצְמֵנוּ כִּבְנֵי חוֹרִין הַנְּתוּנִים בַּמָּצוֹר וּבַשִּׁבְיָה, וּקְרוֹבָה יְשׁוּעָתֵנוּ לָבוֹא."[2]
עִנְיָן הִלְכָתִי נוֹסָף שֶׁנּוֹבֵעַ מֵהַשּׁוֹאָה, הוּא הַאִם יֵשׁ לוֹמַר קַדִּישׁ עַל גּוֹי שֶׁנֶּחְשַׁב לַחֲסִיד אֻמּוֹת הָעוֹלָם. הָרַב אָשְׁרִי כּוֹתֵב כִּי הַדָּבָר אֵינוֹ רַק מֻתָּר, אֶלָּא גַּם בַּעַל עֵרֶךְ סֶגּוֹלִי - מִצְוָה.[3]
