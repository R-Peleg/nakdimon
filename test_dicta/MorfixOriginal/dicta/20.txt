אַדְרִיכָלוּת קְלָסִית
אַדְרִיכָלוּת קְלָסִית הוּא שֵׁם כּוֹלֵל לְסִגְנוֹנוֹת אַדְרִיכָלִיִּים שֶׁאִפְיְנוּ אֶת אֵזוֹר הַיָּם הַתִּיכוֹן בָּעֵת הָעַתִּיקָה בְּמֶשֶׁךְ תְּקוּפָה שֶׁל כְּאֶלֶף שָׁנָה.
הָאַדְרִיכָלוּת הַקְּלָסִית מִתְיַחֶסֶת לַסִּגְנוֹן הָאַדְרִיכָלִי אֲשֶׁר פּוֹתֵחַ בְּיָוָן הֶעֱתִיקָה כְּבָר בַּתְּקוּפָה הָאַרְכָאִית, הָאַדְרִיכָלוּת הַהֶלֶנִיסְטִית שֶׁאִפְיְנָה אֶת מִזְרַח הַיָּם הַתִּיכוֹן וְאֶת אֵזוֹר הַמִּזְרָח הַתִּיכוֹן אֲשֶׁר נִכְבְּשׁוּ עַל יְדֵי אָלֶכְּסַנְדֶּר מוֹקְדּוֹן וְאֶת הָאַדְרִיכָלוּת הָרוֹמִית עַד לִנְפִילַת הָאִימְפֶּרְיָה הָרוֹמִית.
הָאַדְרִיכָלוּת הַקְּלָסִית הִוְּתָה בָּסִיס לְכָל סִגְנוֹנוֹת הַבְּנִיָּה בְּתַרְבּוּת הַמַּעֲרָב עַד הַמֵּאָה הַ-20. הַמִּבְנִים הַקְּלָסִיִּים עֲלֵיהֶם מְדֻבָּר הֵם בְּעִקַּר מִקְדָּשִׁים וְכֵן גַּם מוֹסְדוֹת שִׁלְטוֹן וְתֵאַטְרָאוֹת אֲשֶׁר לֹא הָיוּ קַיָּמִים עַד אוֹתָהּ הָעֵת.
הָאַדְרִיכָלוּת בַּתְּקוּפָה הַקְּלָסִית שֶׁל יָוָן הָעַתִּיקָה (הַמֵּאָה הַ-5 לפנה"ס) כְּבָר הָיְתָה בְּרוּרָה וּמְזֻקֶּקֶת. הַסִּגְנוֹן הִגְדִּיר כְּלָלִים מְאוֹד בְּרוּרִים לַבְּנִיָּה אֲשֶׁר הִתְבַּסְּסוּ עַל סֵדֶר וְסִימֶטְרְיָה. נֻסְּחוּ סְדָרִים אֲשֶׁר הִגְדִּירוּ אֶת הַפְּרוֹפּוֹרְצְיוֹת הַמְּדֻיָּקוֹת וְאֶת הַתְאָמַת האורנומנטיקה (קִשּׁוּטִים) לַמִּבְנֶה.
בְּמֵאָה הַ-4 לפנה"ס כָּבַשׁ אָלֶכְּסַנְדֶּר מוֹקְדּוֹן שְׁטָחִים עֲצוּמִים מֵהַבַּלְקָן וְעַד הֹדּוּ. בְּמִסְגֶּרֶת הֲפָצַת הַתַּרְבּוּת הַהֶלֶנִיסְטִית נִבְנוּ עָרִים וּמִבְנִים בַּסִּגְנוֹן הַיְּוָנִי הַקְּלָסִי. שְׂרִידִים אַרְכֵאוֹלוֹגִיִּים שֶׁל תֵּאַטְרָאוֹת נִמְצָאִים בְּכָל רַחֲבֵי הַמִּזְרָח הַתִּיכוֹן וְעַד לְאַפְגָּנִיסְטָן.
הָאַדְרִיכָלוּת הָרוֹמִית אִמְּצָה אֶת הַסִּגְנוֹן וְהִמְשִׁיכָה לְשַׁכְלְלוֹ וּלְפָתְחוֹ עַד הַמֵּאָה ה-5 לַסְּפִירָה.
