אֶלְדָּד פֶּלֶד
אֶלְדָּד פֶּלֶד הוּא קְצִין צה"ל בַּמִּלּוּאִים בְּדַרְגַּת אַלּוּף מִשְׁנֶה.
שֵׁרוּת צְבָאִי
עִם גִּיּוּסוֹ לצה"ל שֻׁבַּץ פֶּלֶד בַּחֲטִיבַת גִּבְעָתִי וְעָבַר מַסְלוּל הַכְשָׁרָה כְּלוֹחֵם, קוּרְס מְ"כְּיַם חי"ר וְקוּרְס קְצִינֵי חי"ר. עִם סִיּוּם הַקּוּרְס שִׁמֵּשׁ כִּמְפַקֵּד מַחְלָקָה וּפְלֻגָּה בַּחֲטִיבָה. בַּהֶמְשֵׁךְ שִׁמֵּשׁ כִּמְפַקֵּד פלחה"ן גִּבְעָתִי וְכִמְפַקֵּד סַיֶּרֶת גִּבְעָתִי. בְּשָׁנִים 1999-2001 שִׁמֵּשׁ כִּמְפַקֵּד גְּדוּד שָׁקֵד וְלָקַח חֵלֶק בַּפְּעִילוּת הַמִּבְצָעִית כְּנֶגֶד הַטֶּרוֹר הַפָלַסְטִינִי בִּרְצוּעַת עַזָּה וּבַלְּחִימָה בִּדְרוֹם לְבָנוֹן. לְאַחַר מִכֵּן שִׁמֵּשׁ כִּמְפַקֵּד בא"ח גִּבְעָתִי[1] וְכִסְגַן מְפַקֵּד הַחֲטִיבָה[2]. בְּשָׁנִים 2006-2009 שִׁמֵּשׁ כִּמְפַקֵּד מַעֲרַךְ מַגָּל[3]. לְאַחַר שִׁחְרוּרוֹ פָּנָה לַעֲבֹד כְּעוֹרֵךְ דִּין[4].
