אֹפֶן חִתּוּךְ
אֹפֶן חִתּוּךְ הוּא מִדַּת הַחֲסִימָה שֶׁל מַעֲבַר הָאֲוִיר בִּבְסִיס הַחִתּוּךְ שֶׁהוּא הַמָּקוֹם בּוֹ מִסְפַּר אֵיבְרֵי הֲגִיָּה סוֹתְמִים בְּאֹפֶן מָלֵא אוֹ חַלְּקִי אֶת חֲלַל הַפֶּה אוֹ הַגָּרוֹן וְגוֹרְמִים לַהֲפָקַת הָעִצּוּרִים כְּתוֹצָאָה מִתְּזוּזָתָם שֶׁל אֵיבְרֵי הֲגִיָּה מְסֻיָּמִים כְּלַפֵּי אֲחֵרִים וּמִזְּרִימַת הָאֲוִיר בֵּינֵיהֶם.
מְאַפְיְנֵי אֹפֶן הַחִתּוּךְ
מִדַּת הַקִּרְבָה וסונורנטיות
מִדַּת הַקִּרְבָה הִיא מִדַּת הַחֲסִימָה שֶׁל מַעֲבַר הָאֲוִיר בֵּין אֵיבְרֵי הַהֲגִיָּה בַּחֲלַל הַפֶּה וְהַגָּרוֹן, כַּאֲשֶׁר יֵשׁ סְתִימָה מְלֵאָה שֶׁל חֲלַל הַפֶּה אוֹ הַגָּרוֹן, הַחֲסִימָה שֶׁל מַעֲבַר הָאֲוִיר הִיא מֻחְלֶטֶת וּמַקְסִימָלִית, אַךְ כַּאֲשֶׁר יֵשׁ סְתִימָה חֶלְקִית שֶׁל מַעֲבַר הָאֲוִיר, הַחֲסִימָה שֶׁל מַעֲבַר הָאֲוִיר הִיא קְטַנָּה וּמִינִימָלִית.
עִם זֹאת, מִדַּת הסונורנטיות הִיא עָצְמַת הַקּוֹל הַנִּקְבַּעַת עַל פִּי מִדַּת הַתְּנוּדָה שֶׁל מֵיתְרֵי הַקּוֹל, כַּאֲשֶׁר עָצְמַת הַקּוֹל נְמוּכָה, מֵיתְרֵי הַקּוֹל רְפוּיִים וְאֵינָם רוֹטְטִים, אַךְ כַּאֲשֶׁר עָצְמַת הַקּוֹל גְּבוֹהָה, מֵיתְרֵי הַקּוֹל רוֹטְטִים.
סוֹתֵם הוּא עִצּוּר שֶׁבְּמַהֲלַךְ הֲפָקָתוֹ נוֹצֶרֶת חֲסִימָה מֻחְלֶטֶת וּמַקְסִימָלִית שֶׁל מַעֲבַר הָאֲוִיר בֵּין אֵיבְרֵי הַהֲגִיָּה שֶׁמִּמֶּנָּה נוֹבֵעַ עִקַּר צְלִיל הָעִצּוּר וְלָכֵן מֵיתְרֵי הַקּוֹל רְפוּיִים וְאֵינָם רוֹטְטִים בְּמַהֲלַךְ הֲגִיָּתוֹ, מִדַּת הסונורנטיות שֶׁלּוֹ נְמוּכָה וְהוּא גַּם בִּלְתִּי סונורנטי.
בִּלְתִּי סוֹתֵם הוּא עִצּוּר שֶׁבְּמַהֲלַךְ הֲפָקָתוֹ נוֹצֶרֶת חֲסִימָה קְטַנָּה וּמִינִימָלִית שֶׁל מַעֲבַר אֲוִיר בֵּין אֵיבְרֵי הַהֲגִיָּה וְלָכֵן עִקַּר צְלִיל הָעִצּוּר נוֹבֵעַ מֵרְטִיטַת מֵיתְרֵי הַקּוֹל, מִדַּת הסונורנטיות שֶׁלּוֹ גְּבוֹהָה וְהוּא גַּם סונורנטי.
מִדַּת הסונורנטיות מֻשְׁפַּעַת מִמִּדַּת הַקִּרְבָה, כַּאֲשֶׁר הַחֲסִימָה שֶׁל מַעֲבַר הָאֲוִיר קְטַנָּה מֵיתְרֵי הַקּוֹל רוֹטְטִים וְכַאֲשֶׁר הַחֲסִימָה שֶׁל מַעֲבַר הָאֲוִיר הִיא מֻחְלֶטֶת מֵיתְרֵי הַקּוֹל רְפוּיִים וְלָכֵן עִצּוּר סוֹתֵם הוּא עִצּוּר בִּלְתִּי סונורנטי וְעִצּוּר בִּלְתִּי סוֹתֵם הוּא עִצּוּר סונורנטי.
צִדִּיּוֹת
צִדִּיּוֹת הִיא הָאֵזוֹר בַּחֲלַל הַפֶּה בּוֹ עוֹבֵר זֶרֶם הָאֲוִיר יַחֲסִית לְמַצַּב הַלָּשׁוֹן, הַצִּדִּיּוֹת יוֹצֶרֶת הַבְחָנָה בֵּין שִׁנִּי מַצָּבִים בָּהֶם הַלָּשׁוֹן עֲשׂוּיָה לְהִמָּצֵא:
מֶרְכָּזִיִּים הֵם עִצּוּרִים שֶׁבְּמַהֲלַךְ הֲפָקָתָם זָרַם הָאֲוִיר עוֹבֵר מֵעַל מֶרְכַּז הַלָּשׁוֹן. רֹב הָעִצּוּרִים הֵם מֶרְכָּזִיִּים.
צִדִּיִּים הֵם עִצּוּרִים שֶׁבְּמַהֲלַךְ הֲפָקָתָם צִדֵּי הַלָּשׁוֹן יוֹרְדִים וּמְאַפְשְׁרִים אֶת מַעֲבַר זֶרֶם הָאֲוִיר מִצִּדִּי הַלָּשׁוֹן.
כַּאֲשֶׁר לֹא מְצֻיָּן לְיַד שֵׁם הָעִצּוּר שֶׁהוּא צִדִּי, סִימָן שֶׁהוּא מֶרְכָּזִי.
מִיּוּן אוֹפַנֵּי הַחִתּוּךְ
עִצּוּרִים רֵאָתִיִּים
עִצּוּרִים רֵאָתִיִּים הֵם עִצּוּרִים שֶׁמּוּפָקִים בְּאֶמְצָעוּת הַמַּנְגָּנוֹן הָרֵאָתִי מִחוּצָן שֶׁבּוֹ זְרִימַת הָאֲוִיר הַחוּצָה מְוַסֶּתֶת בְּאֶמְצָעוּת תְּנוּעַת הַסַּרְעֶפֶת הָעוֹלֶה לְמַעְלָה, יוֹצֶרֶת לַחַץ חִיּוּבִי בְּבֵית הֶחָזֶה, מְכַוֶּצֶת אֶת הָרֵאוֹת וְגוֹרֶמֶת לְשִׁחְרוּר הָאֲוִיר מִתּוֹכָן הַחוּצָה דֶּרֶךְ חֲלַל הַפֶּה אוֹ הַאַף. עִצּוּרִים אֵלֶּה הֵם הַנְּפוֹצִים בְּיוֹתֵר וְיֵשׁ לָהֶם אוֹפַנֵּי חִתּוּךְ רַבִּים וְשׁוֹנִים שֶׁחֶלְקָם סוֹתְמִים וְחֶלְקָם בִּלְתִּי סוֹתְמִים.
עִצּוּרִים בִּלְתִּי סוֹתְמִים
עִצּוּרִים בִּלְתִּי סוֹתְמִים הֵם עִצּוּרִים שֶׁבְּמַהֲלַךְ הֲפָקָתָם זְרִימַת הָאֲוִיר אֵינָהּ נֶחְסֶמֶת וְנוֹצַר מִרְוָח מַסְפִּיק לְמַעֲבַר הָאֲוִיר בֵּין אֵיבְרֵי הַהֲגִיָּה. כָּל הָעִצּוּרִים הַלָּלוּ הֵם רֵאָתִיִּים.
עִצּוּרִים מְקֹרָבִים הֵם עִצּוּרִים שֶׁבְּמַהֲלַךְ הֲפָקָתָם הָאֲוִיר זוֹרֵם דֶּרֶךְ מְרֻוָּח שֶׁאֵינוֹ מֵבִיא לִיצִירַת צְלִיל חִכּוּךְ. בַּהֲגִיַּת הָעִצּוּרִים הַמְּקֹרָבִים מֵיתְרֵי הַקּוֹל רוֹטְטִים בְּעָצְמָה רַבָּה, וְכִמְעַט שֶׁלֹּא נִשְׁמַע "רַעַשׁ אָקוּסְטִי". כָּל הָעִצּוּרִים הַמְּקֹרָבִים מִלְּבַד הָעִצּוּרִים הַמְּקֹרָבִים צִדִּיִּים הֵם מֶרְכָּזִיִּים.
חֲצָאֵי תְּנוּעוֹת (מְקֹרָבִים) הֵם עִצּוּרִים שֶׁיֵּשׁ לָהֶם זִקָּה לִתְנוּעָה מְסֻיֶּמֶת וְשֶׁבְּמַהֲלַךְ הֲפָקָתָם זָרַם הָאֲוִיר עוֹבֵר מֵעַל מֶרְכַּז הַלָּשׁוֹן.
עִצּוּרִים מְקֹרָבִים נְזִילִים (חוֹכְכִים-מְקֹרָבִים) הֵם עִצּוּרִים שֶׁאֵין לָהֶם זִקָּה לִתְנוּעָה מְסֻיֶּמֶת וְשֶׁלֹּא מִשְׁתַּיְּכִים לַחֲצָאֵי הַתְּנוּעוֹת. הֵם נֶחְתָּכִים בְּאֹפֶן מְקֹרָב עִם חִכּוּךְ מִזְעָרִי מֻרְגָּשׁ. 
עִצּוּרִים מְקֹרָבִים נְזִילִים (חוֹכְכִים-מְקֹרָבִים) צִדִּיִּים הֵם עִצּוּרִים שֶׁבְּמַהֲלַךְ הֲפָקָתָם זָרַם הָאֲוִיר עוֹבֵר מִצַּד אֶחָד אוֹ מִשְּׁנֵי צִדֵּי הַלָּשׁוֹן. גַּם הֵם נֶחְתָּכִים בְּאֹפֶן מְקֹרָב עִם חִכּוּךְ מִזְעָרִי מֻרְגָּשׁ.
עִצּוּרִים חוֹכְכִים הֵם עִצּוּרִים שֶׁבְּמַהֲלַךְ הֲפָקָתָם הָאֲוִיר זוֹרֵם דֶּרֶךְ מִרְוָח צָר יַחֲסִית הַמִּשְׁתַּחְרֵר עִם צְלִיל שֶׁל חִכּוּךְ. מִבְּחִינָה אָקוּסְטִית, נוֹצַר "רַעַשׁ" לְפֶרֶק זְמַן אָרֹךְ. כָּל הָעִצּוּרִים הַחוֹכְכִים מִלְּבַד הָעִצּוּרִים הַחוֹכְכִים צִדִּיִּים הֵם מֶרְכָּזִיִּים.
עִצּוּרִים חוֹכְכִים שׁוֹרְקִים הֵם עִצּוּרִים הַמּוּפָקִים בְּאֵזוֹר הַמַּכְתֵּשׁ וּבָתַר הַמַּכְתֵּשׁ כָּךְ שֶׁקִּדְמַת הַלָּשׁוֹן מְתַעֶלֶת אֶת זֶרֶם הָאֲוִיר הָעוֹבֵר מֵעַל מֶרְכָּזָהּ אֶל הַשִּׁנַּיִם הָעֶלְיוֹנוֹת.
עִצּוּרִים חוֹכְכִים בִּלְתִּי-שׁוֹרְקִים הֵם עִצּוּרִים שֶׁאֵינָם מוּפָקִים בְּאֵזוֹר הַמַּכְתֵּשׁ וּבָתַר הַמַּכְתֵּשׁ וְשֶׁלֹּא מִשְׁתַּיְּכִים לָעִצּוּרִים הַשּׁוֹרְקִים.
עִצּוּרִים חוֹכְכִים בִּלְתִּי-שׁוֹרְקִים צִדִּיִּים הֵם עִצּוּרִים שֶׁבְּמַהֲלַךְ הֲפָקָתָם זָרַם הָאֲוִיר עוֹבֵר מִצַּד אֶחָד אוֹ מִשְּׁנֵי צִדֵּי הַלָּשׁוֹן.
עִצּוּרִים מְחֻכָּכִים הֵם עִצּוּרִים שֶׁבְּמַהֲלַךְ הֲפָקָתָם זְרִימַת הָאֲוִיר נֶחְסֶמֶת בְּאֹפֶן מֻחְלָט וּמִשְׁתַּחְרֶרֶת לְאֹרֶךְ זְמַן תּוֹךְ יְצִירַת צְלִיל חִכּוּךְ. מִבְּחִינָה אָקוּסְטִית, נוֹצַר "רַעַשׁ" חָזָק לְפֶרֶק זְמַן קָצָר, וּמִיָּד לְאַחֲרָיו "רַעַשׁ" חַלָּשׁ יוֹתֵר לְפֶרֶק זְמַן אָרֹךְ יוֹתֵר. כָּל הָעִצּוּרִים הַמְּחֻכָּכִים מִלְּבַד הָעִצּוּרִים הַמְּחֻכָּכִים צִדִּיִּים הֵם מֶרְכָּזִיִּים.
עִצּוּרִים מְחֻכָּכִים שׁוֹרְקִים הֵם עִצּוּרִים הַמּוּפָקִים בְּאֵזוֹר הַמַּכְתֵּשׁ וּבָתַר הַמַּכְתֵּשׁ כָּךְ שֶׁקִּדְמַת הַלָּשׁוֹן מְתַעֶלֶת אֶת זֶרֶם הָאֲוִיר הָעוֹבֵר מֵעַל מֶרְכָּזָהּ אֶל הַשִּׁנַּיִם הָעֶלְיוֹנוֹת.
עִצּוּרִים מְחֻכָּכִים בִּלְתִּי-שׁוֹרְקִים הֵם עִצּוּרִים שֶׁאֵינָם מוּפָקִים בְּאֵזוֹר הַמַּכְתֵּשׁ וּבָתַר הַמַּכְתֵּשׁ וְשֶׁלֹּא מִשְׁתַּיְּכִים לָעִצּוּרִים הַשּׁוֹרְקִים.
עִצּוּרִים מְחֻכָּכִים בִּלְתִּי-שׁוֹרְקִים צִדִּיִּים הֵם עִצּוּרִים שֶׁבְּמַהֲלַךְ הֲפָקָתָם זָרַם הָאֲוִיר עוֹבֵר מִצַּד אֶחָד אוֹ מִשְּׁנֵי צִדֵּי הַלָּשׁוֹן
