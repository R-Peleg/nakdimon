אוזבקיסטן בַּמִּשְׂחָקִים האולימפיים
אוזבקיסטן מִשְׁתַּתֶּפֶת בַּמִּשְׂחָקִים האולימפיים מֵאָז אולימפיאדת אטלנטה (1996). עַד 1991 הָיְיתָה אוזבקיסטן חָלַק מִבְּרִית הַמּוֹעָצוֹת, וספורטאיה הִתְחָרוּ בַּמִּסְגֶּרֶת הַמִּשְׁלַחַת הסובייטית. בְּאוֹתָהּ שָׁנָה הִתְפָּרְקָה בְּרִית הַמּוֹעָצוֹת ל-15 מְדִינוֹת נִפְרָדוֹת, ו-12 מָהֵן, בָּהֶן אוזבקיסטן, שָׁלְחוּ מִשְׁלַחַת מְאוּחֶדֶת לאולימפיאדת בַּרְצֶלוֹנָה (1992). בְּאַרְבַּע האולימפיאדות בָּהֶן הִשְׁתַּתְּפָה אוזבקיסטן, צָבְרוּ הַסְּפּוֹרְטָאִים האוזבקים 32 מֵדַלְיוֹת, מֵהֶן 9 מֵדַלְיוֹת זָהָב. הָעֲנָפִים הַבּוֹלְטִים בְּיוֹתֵר הֵם הֵיאָבְקוּת ואיגרוף.
הִיסְטוֹרְיָה
הַמִּשְׁלַחַת הָעַצְמָאִית הָרִאשׁוֹנָה שֶׁל אוזבקיסטן, באולימפיאדת אטלנטה, כָּלְלָה 71 סְפּוֹרְטָאִים. שְׁנִיִּים מֵהֵם זָכוּ בַּמֵּדַלְיוֹת הָרִאשׁוֹנוֹת בתולדותיה: הג'ודוקא ארמן בגדסרוב זָכָה בְּמֵדַלְיַית כֶּסֶף והמתאגרף כַּרִים טולגנוב זָכָה בְּמֵדַלְיַית אָרָד. באולימפיאדת סידני (2000) נָטְלוּ חֵלֶק 70 סְפּוֹרְטָאִים אוזבקים. המתאגרף מחמטקודיר עבדולאייב זָכָה בְּמֵדַלְיַית הַזָּהָב הָרִאשׁוֹנָה בַּתּוֹלָדוֹת אוזבקיסטן (בְּמִשְׂחֲקֵי הַקַּיִץ). שֵׁנִי מתאגרפים נוֹסָפִים, סרגיי מיחאילוב ורוסטם סאידוב, זָכוּ בְּמֵדַלְיוֹת אָרָד, ומתאגרף רְבִיעִי, רוסלאן צ'גאייב, הֶעְפִּיל לְשָׁלָב רֶבַע גָּמַר. בַּהֵיאָבְקוּת בְּלַט ארתור טאימזוב, שֶׁזָּכָה בְּמֵדַלְיַית כֶּסֶף.
גַּם באולימפיאדת אַתוּנָה (2004) הִשְׁתַּתְּפוּ 70 סְפּוֹרְטָאִים אוזבקים. טאימזוב זָכָה בְּמֵדַלְיַית זָהָב, וְכָמוֹהוּ גַּם הַמִּתְאַבֵּק אלכסנדר דוקטורשווילי. מִתְאַבֵּק שְׁלִישִׁי, מגומד איברגימוב זָכָה בְּמֵדַלְיַית כֶּסֶף. והמתאגרפים בחודירג'ון סולטונוב ואוטקירבק חיידרוב זָכוּ בְּמֵדַלְיוֹת אָרָד, וּשְׁלוֹשָׁה מתאגרפים אוזבקים נוֹסָפִים, טולאשבוי דוניורוב, דילשוד מחמודוב ושרזוד חוסאנוב, הֶעְפִּילוּ לְשָׁלָב רֶבַע הגמר. בַּהֲרָמַת מִשְׁקוֹלוֹת בָּלַט אלכסנדר אורינוב, שֶׁסִּיֵּים בַּמָּקוֹם הַשְּׁמִינִי. עַל הִשְׁתַּתְּפוּתָהּ שֶׁל אוזבקיסטן בְּמִשְׂחָקִים אַלָּה העיבה פְּסִילָתָהּ שֶׁל מיידה הַפַּטִּישׁ אוֹלְגָה שצ'וקינה, בְּשֶׁל שִׁימּוּשׁ בַּסַּמִים.
לאולימפיאדת בייג'ינג (2008) שָׁלְחָה אוזבקיסטן מִשְׁלַחַת קָטְנָה יוֹתֵר, בַּת 58 סְפּוֹרְטָאִים. הַמִּתְאַבֵּק ארתור טאימזוב זָכָה בְּמֵדַלְיַית זָהָב שְׁנִיָּיה בִּרְצִיפוּת, ועמיתו סוסלן טיגייב זָכָה בְּמֵדַלְיַית כֶּסֶף, אַךְ זוֹ נשללה מַמְּנוּ ב-2016 בְּשֶׁל הָרְשָׁעָה באמרוץ. דילשוד מנסורוב סִיֵּים בַּמָּקוֹם הַחֲמִישִׁי וְקָורְבָּן קורבנוב בַּמָּקוֹם הַשְּׁבִיעִי. הג'ודוקא עבדולו טנגרייב זָכָה בְּמֵדַלְיַית כֶּסֶף, בְּעוֹד שרישוד סובירוב זָכָה בְּמֵדַלְיַית אָרָד. ג'ודוקא נוֹסַף, מיראלי שריפוב, סִיֵּים בַּמָּקוֹם הַחֲמִישִׁי. בְּנוֹסָף, שֵׁנִי מִתְעַמְּלִים אוזבקים זָכוּ לָרִאשׁוֹנָה בְּמֵדַלְיוֹת אָרָד - אנטון פוקין בַּתַּרְגִּיל הַמַּקְבִּילִים ויקטרינה חִילְּקוּ, שֶׁהָיְיתָה לַסְּפּוֹרְטָאִית האוזבקית הָרִאשׁוֹנָה, וְהַיְּחִידָה עֵד כֹּה, שֶׁזּוֹכֶה בַּמֵּדַלְיָה האולימפית בְּמִשְׂחֲקֵי הַקַּיִץ - בטרמפולינה. חוֹתֵר הֵקֵאנוּ וְאֵדִים מנקוב הָיָה קָרוֹב לִזְכִיָּיה רִאשׁוֹנָה בַּמֵּדַלְיָה גַּם בְּעָנָף זֶה, אַךְ סִיֵּים בַּמָּקוֹם הָרְבִיעִי במקצה הספרינט ל-1,000 מָטָר. בֶּעָנָף הטאקוונדו בָּלַט אקמאל אירגשב, שֶׁסִּיֵּים בַּמָּקוֹם הַחֲמִישִׁי. המתאגרפים האוזבקים לֹא הִצְלִיחוּ בְּמִשְׂחָקִים אֵלּוּ לִזְכּוֹת בְּמֵדַלְיוֹת, אַךְ שְׁנִיִּים מָהֵם, דילשוד מחמודוב ואלשוד רסולוב, הֶעְפִּילוּ לְשָׁלָב רֶבַע הגמר.
באולימפיאדת לוֹנְדוֹן (2012) הִשְׁתַּתְּפוּ 54 סְפּוֹרְטָאִים אוזבקים. טאימזוב זָכָה בְּמֵדַלְיַית זָהָב שְׁלִישִׁית בִּרְצִיפוֹת, שֶׁהָיְיתָה גַּם מֵדַלְיַית הַזָּהָב הַיְּחִידָה לִזְכוּתָהּ שֶׁל הַמִּשְׁלַחַת. מִתְאַבֵּק נוֹסָף, סוסלאן טיגייב, זָכָה בְּמֵדַלְיַית אָרָד, אַךְ זוֹ נשללה מַמְּנוּ חֳודָשִׁים סְפוּרִים לְאַחֵר הַמְּשַׂחֲקִים, לְאַחֵר שֶׁנִּמְצָא כִּי נֵטֶל סַמִים. שְׁנֵי סְפּוֹרְטָאִים אוזבקים נוֹסָפִים שֶׁזָּכוּ בַּמֵּדַלְיוֹת הָיוּ הג'ודוקא רישוד סובירוב והמתאגרף אבוס אטויב, שֶׁזָּכוּ בְּמֵדַלְיוֹת אָרָד. אַרְבָּעָה מתאגרפים נוֹסָפִים הוּדְּחוּ בְּשָׁלָב רֶבַע הגמר. סְפּוֹרְטָאִים אוזבקים נוֹסָפִים שֶׁבָּלְטוּ בְּלוֹנְדוֹן כָּלְלוּ אֶת חוֹתֵר הֵקֵאנוּ וְאֵדִים מנקוב (מָקוֹם רְבִיעִי במקצה ל-1000 מֶטְרִים), מֵרִים הַמִּשְׁקוֹלוֹת רוסלן נורודינוב (מָקוֹם הָרְבִיעִי), רוֹכֵב הָאוֹפַנַּיִים סרגיי לגוטין (מָקוֹם חֲמִישִׁי בְּמֵרוֹץ אוֹפַנַּי כְּבִישׁ), הג'ודוקא רמזידין (מָקוֹם חֲמִישִׁי), הַמִּתְאַבֵּק קָורְבָּן קורבנוב (מָקוֹם חֲמִישִׁי), מֵרִים הַמִּשְׁקוֹלוֹת איוואן אפרמוב (מָקוֹם חֲמִישִׁי); האתלטית סבטלנה רדזיוויל (מָקוֹם שְׁבִיעִי בַּקְּפִיצָה לַגּוֹבַהּ) והג'ודוקא דילשוד צ'ורייב (מָקוֹם שְׁבִיעִי).
אולימפיאדת ריו דה ז'ניירו (2016) הָיְיתָה הַמּוּצְלַחַת בְּיוֹתֵר בַּתּוֹלָדוֹת אוזבקיסטן, עִם 13 מֵדַלְיוֹת, מֵהֶן 4 מֵדַלְיוֹת זָהָב, הֶישֵּׂג דּוֹמֶה לְהֶישֵּׂגֵי הַמְּשַׁלְּחוֹת האוזבקיות בַּשָּׁלוֹשׁ האולימפיאדות הַקּוֹדְמוֹת גַּם יַחַד. האיגרוף הָיָה הֶעָנָף הַבּוֹלֵט בְּיוֹתֵר. המתאגרפים הִתְחָרוּ בְּכָל עֶשֶׂר קָטֵגוֹרְיוֹת הַמִּשְׁקָל לִגְבָרִים, וּבַשֶּׁבַע מֵהֶן זָכוּ בַּמֵּדַלְיוֹת. חסנבוי דוסמטוב, שאחובידין זוירוב ופזלידין גאיבנזארוב זָכוּ בְּמֵדַלְיוֹת זָהָב, שחראם גיאסוב ובקטמיר מליקוזייב זָכוּ בְּמֵדַלְיוֹת כֶּסֶף, ומורוג'ון אחמדלייב ורוסטם טולגנוב זָכוּ בְּמֵדַלְיוֹת אָרָד. בְּנוֹסָף, סִיְּימוּ הורשיד טוז'יבייב ובחודיר ז'לולוב בַּמָּקוֹם הַחֲמִישִׁי. בְּמֵדַלְיַית הַזָּהָב הָרְבִיעִית לַזְּכוּת האוזבקים זָכָה מֵרִים הַמִּשְׁקוֹלוֹת רוסלאן נורודינוב. איוואן אפרמוב סִיֵּים בַּמָּקוֹם הַחֲמִישִׁי ורוסטם ג'נגבייב בַּמָּקוֹם הַשִּׁישִּׁי. בַּהֵיאָבְקוּת זָכוּ שְׁלוֹשָׁה אוזבקים בְּמֵדַלְיוֹת אָרָד: איחטיור נברוזוב, מגומד איברגימוב ואלמורט טסמורדוב. בקזוד עבדורחמונוב סִיֵּים בַּמָּקוֹם הַחֲמִישִׁי. הג'ודוקא רישוד סובירוב זָכָה בְּמֵדַלְיַית אָרָד שְׁלִישִׁית בִּרְצִיפוֹת, וְגַם דיורבק אורוזבוייב זָכָה בְּמֵדַלְיַית אָרָד. עבדולו טנגרייב סִיֵּים בַּמָּקוֹם הַחֲמִישִׁי. סְפּוֹרְטָאִים אוזבקים נוֹסָפִים שֶׁבָּלְטוּ בריו דה ז'ניירו כָּלְלוּ אֶת לוֹחֵם הטאקוונדו דמיטרי שוקין שֶׁסִּיֵּים בַּמָּקוֹם הַחֲמִישִׁי, אַתְּ חוֹתֵר הֵקֵאנוּ גראסים קוצ'נב שֶׁסִּיֵּים בַּמָּקוֹם הַשְּׁבִיעִי במקצה ל-1000 מָטָר וּבַמָּקוֹם הַשְּׁמִינִי במקצה הַזּוּגוֹת ל-1000 מָטָר, יַחַד עִם סריק מירבקוב, וְאֶת הַמִּתְעַמֶּלֶת הוותיקה אוקסנה צ'וסוביטינה בַּת ה-41, שֶׁסִּיְּימָה בַּמָּקוֹם הַשְּׁבִיעִי בַּתַּרְגִּיל הַקְּפִיצוֹת.
מִשְׂחֲקֵי הַחוֹרֶף
אולימפיאדת הַחוֹרֶף הָרִאשׁוֹנָה בָּהּ הִשְׁתַּתְּפָה אוזבקיסטן הָיְיתָה אולימפיאדת לילהאמר (1994). הָיְיתָה זוֹ אולימפיאדת הַחוֹרֶף הַיְּחִידָה עַד כֹּה בָּהּ זָכָה סְפּוֹרְטַאי אוזבקי בַּמֵּדַלְיָה - לִינָה צ'ריאזובה זָכְתָה בְּמֵדַלְיַית זָהָב בסקי חָופְשִׁי. פָּרֵט לְהֶישֵּׂג זֶה, הַהֶישֵּׂג הַגָּבוֹהַּ הַיָּחִיד הַנּוֹסָף שֶׁל הַמִּשְׁלַחַת האוזבקית נִרְשַׁם באולימפיאדת נאגנו (1998), אָז דּוֹרְגָה הַמַּחְלִיקָה הָאָומָּנוּתִית טטיאנה מלינינה בַּמָּקוֹם הַשְּׁמִינִי.
