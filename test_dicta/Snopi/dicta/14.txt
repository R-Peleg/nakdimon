אלכס היגינס
אלכסנדר גורדון "אלכס" היגינס (18 בַּמֶּרֶץ 1949 - 24 בְּיוּלִי 2010), יָדוּעַ גַּם בַּכִּינּוּי הוריקן היגנס, הָיָה שַׂחְקָן סנוקר מקצועני צָפוּן אירי. היגינס זָכָה פַּעֲמִיִּים בָּאַלִּיפוּת הָעוֹלָם בסנוקר בַּשָּׁנִים 1972 ו-1982, וְהָיָה פַּעֲמַיִים סְגָן אַלּוּף הָעוֹלָם בסנוקר בַּשָּׁנִים 1976 ו-1980. בְּנוֹסָף, היגינס זָכָה בָּאַלִּיפוּת הָעוֹלָם בסנוקר לַזּוּגוֹת, יַחַד עִם ג'ימי וייט הָאַנְגְּלִי בִּשְׁנַת 1984, וְכֵן זָכָה שָׁלוֹשׁ פְּעָמִים בִּרְצִיפוֹת בַּגָּבִיעַ הָעוֹלָם בסנוקר לְנִבְחָרוֹת, עִם נִבְחַרְתָּ אירלנד בַּשָּׁנִים 1985, 1986 ו-1987. בְּשִׂיאוֹ דּוֹרַג היגינס בַּמָּקוֹם הַשֵּׁנִי בְּעוֹלָם בְּדֵירוּג שַׂחְקָנַי הסנוקר הָעוֹלָמִי. זָכָה לַכִּינּוּי "הוריקן" בִּזְכוּת סִגְנוֹן מִשְׂחָקוֹ הַמָּהִיר.
קַרְיֵירָה
היגינס הֵחֵל לְשַׂחֵק סנוקר מקצועני בְּעוֹנַת 1971/1972, בַּגִּיל 22. הוּא זָכָה בָּאַלִּיפוּת הָעוֹלָם בסנוקר כְּבָר בְּנִיסְיוֹנוֹ הָרִאשׁוֹן בִּשְׁנַת 1972, בַּגִּיל 22, לְאַחֵר שֶׁנִּיצֵּחַ בגמר אֶת ג'ון ספנסר הָאַנְגְּלִי בַּתּוֹצָאָה 32–37, וּבַכָּךְ נִהְיֶה לָאַלּוּף הָעוֹלָם בסנוקר הִצְעִיר בְּיוֹתֵר אִי פַּעַם (שִׂיא שֶׁנִּשְׁבָּר כַּעֲבוֹר 18 שָׁנִים עַל יָדִי סטיבן הנדרי כְּשֶׁזָּכָה בָּאַלִּיפוּת הָעוֹלָם בַּגִּיל 21). לְאַחַר מִכֶּן הִגִּיעַ היגינס לִגְמֹר אַלִּיפוּת הָעוֹלָם בַּשָּׁנִים 1976 ו-1980, אַךְ הִפְסִיד. בִּשְׁנַת 1982, זָכָה היגינס בַּפַּעַם הַשְּׁנִיָּיה בָּאַלִּיפוּת הָעוֹלָם, לְאַחֵר שֶׁנִּיצֵּחַ בגמר אֶת ריי רירדון הולשי בַּתּוֹצָאָה 15–18, וְאִם לֹא הָיָה נִקְנָס בְּהוֹרָדַת נְקוּדּוֹת דֵּירוּג עָקַב עבירות מִשְׁמַעַת, הָיָה מְדוֹרָג בְּעוֹנַת 1982/1983 בַּמָּקוֹם הָרִאשׁוֹן בָּעוֹלָם.
היגינס, עִם סִגְנוֹן מִשְׂחָקוֹ, שֶׁהָיָה מַלְהִיב, הָיָה אַחְרָאִי בְּמִידָּה רַבָּה לַעֲלִיָּיה בַּפּוֹפּוּלָרִיּוּת שֶׁל עָנָף הסנוקר בְּסוֹף שְׁנוֹת הַשְּׂבֵעִים. הוּא הָיָה שַׂחְקָן מָהִיר מְאוֹד, וּבְרַאֲיוֹנוֹת הוּא אַף הִסְבִּיר שֶׁהַסִּיבָּה לַכָּךְ הָיְיתָה פָּשׁוּט חוֹסֶר סַבְלָנוּת. רָבִים מַחֲשִׁיבִים אוֹתוֹ לְאֶחָד מִשַּׂחְקָנַי הסנוקר הַטּוֹבִים בְּיוֹתֵר בְּכָל הַזְּמַנִּים.
עִם זֹאת, נוֹדָע גַּם בְּאוֹפִי וּמֶזֶג בְּעָיָיתִיִּים. עַל פִּי הָעֶרְכּוֹת, כ-50 תְּלוּנוֹת רִשְׁמִיּוֹת הוּגְּשׁוּ נֶגֶד אלכס היגינס עַל יְדֵי הִתְאַחֲדוּת הסנוקר הָעוֹלָמִית - יוֹתֵר מִכָּל שַׂחְקָן סנוקר אַחֵר בַּהִיסְטוֹרְיָה שֶׁל הַמִּשְׂחָק.[1]
בְּמַהֲלַךְ הַקַּרְיֵירָה, זָכָה היגינס ב-30 תְּאַרִים מקצועניים (26 תָּואֳרִי יְחִידִים ו-4 תְּאַרִים קבוצתיים), בָּהֶם אַלִּיפוּת בְּרִיטַנְיָה בִּשְׁנַת 1983, טורניר המאסטרס בַּשָּׁנִים 1978 ו-1981, והמאסטרס האירי בִּשְׁנַת 1989, בּוֹ נִיצֵּחַ בַּגִּיל 40 אֶת סטיבן הנדרי הִצְעִיר.
