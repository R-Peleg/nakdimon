אֲנִי נִכְנָס לַחֶדֶר וּמִתְיַשֵּׁב עַל הַמִּטָּה וּפוֹתֵחַ אֶת קֻפְסַת הַצְּבָעִים וְאֶת חוֹבֶרֶת הַצְּבִיעָה וּמוֹצִיא אֶת הָאָדֹם הַמְּסַחְרֵר מֵהַקֻּפְסָה וּמַתְחִיל לִקְרֹא.
בִּתְחִלַּת הַחוֹבֶרֶת מִסּוֹפֵר הַסִּפּוּר שֶׁל ג'ו.
ג'ו הוּא שַׁתְיָן שֶׁמְּאַבֵּד אֶת אִשְׁתּוֹ, אֶת הָעֲבוֹדָה שֶׁלּוֹ, אֶת כָּל כַּסְפּוֹ, וּבְסוֹפוֹ שֶׁל דָּבָר מַגִּיעַ לָרְחוֹב וְשָׁם הוּא שׁוֹתֶה יַיִן זוֹל יָשָׁר מֵהַבַּקְבּוּק.
לְאוֹרֵךְ כָּל הַתְּקוּפָה הַזֹּאת מְסָרֵב ג'ו לְהוֹדוֹת שֶׁיֵּשׁ לוֹ בְּעָיָה וְשֶׁהוּא אִבֵּד שְׁלִיטָה.
הַסִּפּוּר מְסֻפָּר בְּמִילִים פְּשׁוּטוֹת וּבְצִיּוּרִים פְּשׁוּטִים, רִשּׁוּמִים בְּקַו שֶׁל דְּמוּיוֹת וּמְקוֹמוֹת, וְאֶת אֵלֶּה צָרִיךְ לְמַלֵּא בַּצֶּבַע.
אִם אֲנִי מֵבִין נָכוֹן, הָרַעְיוֹן הַבְּסִיסִי הוּא שֶׁבִּזְמַן שֶׁאֲנִי מַשְׁקִיעַ זְמַן בִּצְבִיעַת הַתְּמוּנוֹת, אֲנִי אָמוּר לְהַפְנִים אֶת הַסִּפּוּר הַזְּוָעָתִי שֶׁל ג'ו וּלְשַׁיֵּךְ אֶת הַזְּוָעָה הַזֹּאת לְמַצָּבִים שׁוֹנִים בַּחַיִּים שֶׁלִּי.
אִם ג'ו אִבֵּד שְׁלִיטָה, הֲרֵי גַּם אֲנִי אִבַּדְתִּי שְׁלִיטָה.
אִם ג'ו מַגִּיעַ לָרְחוֹב, הֲרֵי כְּדַאי לִי לְהִזָּהֵר, אַחֶרֶת גַּם אֲנִי אֶהְיֶה הוֹמְלֶס.
סוֹף הַסִּפּוּר שֶׁל ג'ו אוֹפְּטִימִי לְמַדַּי: הוּא מוֹדֶה שֶׁאִבֵּד שְׁלִיטָה וּמִצְטָרֵף לְאַלְכּוֹהוֹלִיסְטִים אָנוֹנִימִיִּים.
אַחַר כָּךְ יֵשׁ שְׁאֵלוֹן וּבוֹ עֶשְׂרִים וְשֶׁבַע שְׁאֵלוֹת שֶׁעוֹסְקוֹת בְּהֶרְגֵּלֵי הַשְּׁתִיָּה שֶׁל הַנִּבְדָּק.
הַשְּׁאֵלוֹת פְּשׁוּטוֹת, וְכֻלָּן מְחַיְּבוֹת תְּשׁוּבוֹת שֶׁל כֵּן וְלֹא.
הַאִם יֵשׁ מַצָּבִים שֶׁאֵינְךָ מַרְגִּישׁ בָּהֶם בְּנוֹחַ אִם אֵין בְּנִמְצָא אַלְכּוֹהוֹל? כֵּן.
הַאִם קָרָה פַּעַם שֶׁהִתְעוֹרַרְתָּ בַּבּוֹקֵר לְאַחַר לַיְלָה שֶׁל שְׁתִיָּה וְלֹא זָכַרְתָּ אֵיפֹה הָיִיתָ בָּעֶרֶב הַקּוֹדֵם? כֵּן.
כְּשֶׁאַתָּה פִּכֵּחַ, הַאִם אַתָּה מִתְחָרֵט עַל דְּבָרִים שֶׁעָשִׂיתָ בִּהְיוֹתְךָ שִׁכּוֹר? כֵּן.
הַאִם הַיָּדַיִם שֶׁלְּךָ רוֹעֲדוֹת בַּבּוֹקֵר שֶׁלְּאַחַר לַיְלָה שֶׁל שְׁתִיָּה? כֵּן.
הַאִם קוֹרֶה שֶׁאַתָּה שָׁתוּי בְּמֶשֶׁךְ כַּמָּה יָמִים רְצוּפִים? כֵּן, כֵּן, כֵּן, כֵּן.
אֲנִי עוֹנֶה בְּחִיּוּב לְכָל הַשְּׁאֵלוֹת, וְעַל פִּי הַמַּפְתֵּחַ שֶׁבְּסוֹף הַשְּׁאֵלוֹן, פֵּרוּשׁ הַדָּבָר שֶׁאֲנִי נִמְצָא בַּשְּׁלַבִּים הָאַחֲרוֹנִים שֶׁל אַלְכּוֹהוֹלִיזְם כְרוֹנִי וּמְסֻכָּן.
מָה אַתֶּם סָחִים.
