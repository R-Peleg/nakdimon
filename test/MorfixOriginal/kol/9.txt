חַיָּלִים, צָבָא וּבִטְחוֹן
הַפּוֹרְטָל מֶרְכַּז מֵידָע עַל זְכֻיּוֹת חַיָּלֵי צה"ל בְּשֵׁרוּת סָדִיר (חוֹבָה וְקֶבַע) וּמִלּוּאִים, זְכֻיּוֹת הַחַיָּלִים לְאַחַר שִׁחְרוּרָם וּזְכֻיּוֹתֵיהֶם שֶׁל נְכֵי צה"ל.
בְּנוֹסָף, הַפּוֹרְטָל כּוֹלֵל מֵידָע רִאשׁוֹנִי לְגַבֵּי זְכֻיּוֹתֵיהֶם שֶׁל בְּנֵי הַמִּשְׁפָּחוֹת הַשַּׁכּוּלוֹת. אֶת הַמֵּידָע הַמַּקִּיף נִתָּן לִמְצֹא בַּאֲתַר מִשְׁפָּחוֹת וְהַנְצָחָה שֶׁל מִשְׂרַד הַבִּטָּחוֹן.
חַיָּלִים בְּשֵׁרוּת חוֹבָה:
-הַתְאָמוֹת לְבַעֲלִי לְקוּיוֹת לְמִידָה בַּמִּבְחָנִים בְּלִשְׁכוֹת הַגִּיּוּס
-פִּצּוּיֵי פִּטּוּרִים לְעוֹבֵד שֶׁהִתְפַּטֵּר עֵקֶב גִּיּוּס לצה"ל
-אִסּוּר אַפְלָיָה שֶׁל חַיָּלִים בִּמְקוֹמוֹת צִבּוּרִיִּים
-חַיָּלִים וְחַיָּלוֹת בְּשֵׁרוּת חוֹבָה (צה"ל)
-הִתְנַדְּבוּת לְצָבָא וְאֻכְלוּסִיּוֹת מְיֻחָדוֹת
-חַיָּלִים בּוֹדְדִים
חַיָּלִים בְּשֵׁרוּת קֶבַע:
-בִּטּוּחַ בְּרִיאוּת לְחַיָּלֵי קֶבַע
-הַגָּשַׁת עֲרָר לְוַעֲדַת עֲרָר לַחַיָּלִים בְּשֵׁרוּת קֶבַע
-הִשְׁתַּתְּפוּת צה"ל בְּהוֹצָאוֹת קַיְטָנָה לְיַלְדֵי חַיָּלִים בְּשֵׁרוּת קֶבַע
-תְּבִיעָה לַהַכָּרָה בַּפְּגִיעָה בָּעֲבוֹדָה לַחַיָּלִים בְּשֵׁרוּת קֶבַע
-תְּבִיעָה לַהַכָּרָה בְּלִקּוּי רְפוּאִי אוֹ מַחֲלַת מִקְצוֹעַ לַחַיָּלִים בְּשֵׁרוּת קֶבַע
-אִסּוּר אַפְלָיָה שֶׁל חַיָּלִים בִּמְקוֹמוֹת צִבּוּרִיִּים
-אִסּוּר פִּטּוּרֵי חַיֶּלֶת בַּהֵרָיוֹן מִשֵּׁרוּת קֶבַע
-אִסּוּר פִּטּוּרֵי חַיָּל/ת מִשֵּׁרוּת קֶבַע בְּמַהֲלַךְ חֻפְשַׁת לֵדָה אוֹ לְאַחֲרֶיהָ
-אִסּוּר פִּטּוּרֵי חַיָּל/ת מִשֵּׁרוּת קֶבַע בְּמַהֲלַךְ טִפּוּלֵי פּוֹרִיּוּת
-פָּטוּר לִמְשָׁרְתֵי קֶבַע מֵהֲלִיכֵי מִכְרָז לַתַּפְקִידִים בְּשֵׁרוּת הַמְּדִינָה
לוֹחֲמִים:
-קַבָּלַת תְּעוּדַת לוֹחֵם
-זְכֻיּוֹת וַהֲטָבוֹת לְלוֹחֲמֵי חֹד בַּאֲתַר צה"ל
-חַיָּלִים מְשֻׁחְרָרִים וְחַיָּלֵי מִלּוּאִים
-אִסּוּר אַפְלָיָה שֶׁל חַיָּלִים בִּמְקוֹמוֹת צִבּוּרִיִּים
-רִשּׁוּם לְקֻפַּת חוֹלִים לְאַחַר הַשִּׁחְרוּר מצה"ל
-דְּמֵי לֵדָה לְחַיָּלוֹת מְשֻׁחְרָרוֹת שֶׁהֵחֵלּוּ לַעֲבֹד
-זְכֻיּוֹת חַיָּלִים מְשֻׁחְרָרִים
-מִלּוּאִים
-גִּמְלַאי צה"ל
-הַגְדָּלַת קִצְבַּת גִּמְלַאי צה"ל לְנִצּוֹלֵי שׁוֹאָה
-דְּחִיַּת שֵׁרוּת
-מְכִינָה קְדַם צְבָאִית וּשְׁנַת שֵׁרוּת
נְכֵי צה"ל וְנִפְגָּעִים:
-נְכֵי צה"ל
-תְּבִיעָה לַהַכָּרָה בַּפְּגִיעָה בָּעֲבוֹדָה לַחַיָּלִים בְּשֵׁרוּת קֶבַע
-תְּבִיעָה לַהַכָּרָה בְּלִקּוּי רְפוּאִי אוֹ מַחֲלַת מִקְצוֹעַ לַחַיָּלִים בְּשֵׁרוּת קֶבַע
-תְּבִיעָה לַהַכָּרָה בַּפְּגִיעָה בָּעֲבוֹדָה לְאַנְשֵׁי כּוֹחוֹת הַבִּטָּחוֹן
-תְּבִיעָה לַהַכָּרָה בְּלִקּוּי רְפוּאִי אוֹ מַחֲלַת מִקְצוֹעַ לְאַנְשֵׁי כּוֹחוֹת הַבִּטָּחוֹן
-חוֹבֶרֶת מֵידָע לִפְצֹעַ (בְּשָׂפוֹת שׁוֹנוֹת) בַּאֲתַר אֲגַף שִׁקּוּם נָכִים בְּמִשְׂרַד הַבִּטָּחוֹן
מִשְׁפָּחוֹת שַׁכּוּלוֹת:
-הֲנָחָה בָּאַרְנוֹנָה לְמִשְׁפָּחָה שַׁכּוּלָה
-הֲנָחָה בְּמַס רְכִישָׁה לִבְנֵי מִשְׁפָּחוֹת שַׁכּוּלוֹת
-מִמּוּן אִבְחוּנִי לְקוּיוֹת לְמִידָה לִיתוֹמִים בְּנֵי מִשְׁפָּחוֹת שַׁכּוּלוֹת
-מִמּוּן שִׁעוּרֵי הוֹרָאָה מְתַקֶּנֶת לִיתוֹמִים בְּנֵי מִשְׁפָּחוֹת שַׁכּוּלוֹת
-מִמּוּן לִמּוּדֵי תְּעוּדָה מִקְצוֹעִית וְתֹאַר רִאשׁוֹן לְאַחִים שַׁכּוּלִים
-הֵעָדְרוּת מֵהָעֲבוֹדָה בְּיוֹם הַזִּכָּרוֹן
-הֵעָדְרוּת מִתַּעֲסוּקָה לְמִשְׁתַּקֵּם בְּיוֹם הַזִּכָּרוֹן
-שְׁעוֹת סִעוּד לְנִצּוֹלֵי שׁוֹאָה הַזַּכָּאִים לְ"עֶזְרָה לַזּוּלָת" מִמִּשְׂרַד הַבִּטָּחוֹן
-אִסּוּר פִּטּוּרִים בְּגִיל פְּרִישָׁה שֶׁל אַלְמְנוּת צה"ל וְהוֹרִים שַׁכּוּלִים שֶׁיַּלְדָּם נִפְטַר עֵקֶב שֵׁרוּת צְבָאִי
פְּדוּיֵי שֶׁבִי:
-קִצְבָּה לִמְשָׁרְתִים בְּכוֹחוֹת הַבִּטָּחוֹן שֶׁנָּפְלוּ בַּשֶּׁבִי
-תַּשְׁלוּם לַחֲבֵרִים בְּאִרְגּוּנֵי מַחְתֶּרֶת שֶׁנֶּאֶסְרוּ בִּתְקוּפַת הַמַּנְדָּט הַבְּרִיטִי
-קִצְבָּה לְמִתְנַדְּבֵי הַצָּבָא הַבְּרִיטִי שֶׁנָּפְלוּ בַּשֶּׁבִי הַנָּאצִי
מֶרְכְּזֵי שֵׁרוּת צִבּוּרִיִּים:
-מוֹקְדִים
