פּוֹרְטָל מְהַוֶּה שְׂעַר כְּנִיסָה לְכָל נוֹשֵׂא הַהוֹצָאָה לַפֹּעַל וּגְבִיָּה.
מַעֲרֶכֶת הַהוֹצָאָה לַפֹּעַל פּוֹעֶלֶת עַל-פִּי חֹק לַאֲכִיפַת פִּסְקֵי דִּין (כּוֹלֵל פִּסְקֵי דִּין לְתַשְׁלוּם מְזוֹנוֹת) מַשְׁכַּנְתָּאוֹת, שְׁטָרוֹת, הַחְלָטוֹת שֶׁנִּתְּנוּ עַל-יְדֵי גּוּפִים שׁוֹנִים וּתְבִיעוֹת בִּסְכוּם קָצוּב (עַד 75,000 שְׁקָלִים). בְּמִסְגֶּרֶת הַחֹק מֻסְדָּרִים אֶמְצָעֵי הָאֲכִיפָה הַשּׁוֹנִים אוֹתָם נִתָּן לִנְקֹט נֶגֶד הַחַיָּב. אֶמְצָעֵי הָאֲכִיפָה בַּהוֹצָאָה לַפֹּעַל נִנְקָטִים לְפִי בַּקָּשָׁה שֶׁל הַזּוֹכֶה (הָאָדָם שֶׁפּוֹתֵחַ אֶת תִּיק הַהוֹצָאָה לַפֹּעַל) וּמַתַּן אִשּׁוּר שֶׁל הַגּוֹרֵם הַמֻּסְמָךְ. הַגּוֹרֵם הַמֻּסְמָךְ יָכוֹל לִהְיוֹת רַשַּׁם הַהוֹצָאָה לַפֹּעַל, שֶׁהוּא הַגּוֹרֵם הַשִּׁפּוּטִי בְּלִשְׁכוֹת הַהוֹצָאָה לַפֹּעַל, אוֹ מְנַהֵל לִשְׁכַּת הַהוֹצָאָה לַפֹּעַל, שֶׁהוּא הַגּוֹרֵם הַמִּנְהָלִי הַבָּכִיר בְּיוֹתֵר בְּלִשְׁכוֹת הַהוֹצָאָה לַפֹּעַל.
עַל הַגּוֹרֵם הַמֻּסְמָךְ לָדוּן בַּבַּקָּשׁוֹת וְלָתֵת הַחְלָטָה תּוֹךְ שְׁקִילַת מִכְלוֹל הַשִּׁקּוּלִים הָרֶלֶוַנְטִיִּים וְתוֹךְ אִזּוּן מַתְמִיד בֵּין זְכֻיּוֹתָיו שֶׁל הַזּוֹכֶה לְבֵין זְכֻיּוֹתָיו שֶׁל הַחַיָּב.
