הַמְּעַרְעֶרֶת הוֹסִיפָה וְטָעֲנָה כִּי הַמֵּשִׁיב הַפָּר אֶת חוֹבַת תֹּם הַלֵּב בְּנִהוּל מַשָּׂא וּמַתָּן, וְלָכֵן יֵשׁ לְקַיֵּם אֶת הַהֶסְכֵּם, אַךְ יַחַד עִם זֹאת לִקְבֹּעַ כִּי מַגִּיעִים לָהּ סְכוּמִים נוֹסָפִים שֶׁעַל הַמֵּשִׁיב לְהַעֲבִיר לָהּ לְאַחַר שֶׁיֵּעָרֵךְ אִזּוּן מַשְׁאַבִּים מָלֵא, שֶׁיִּתְיַחֵס לִכְלָל הָרְכוּשׁ.
הַמְּעַרְעֶרֶת עָתְרָה לְכָךְ, שֶׁבֵּית הַמִּשְׁפָּט יַצְהִיר כִּי הַמֵּשִׁיב הִטְעָה אוֹתָהּ, הוֹלִיךְ אוֹתָהּ שׁוֹלָל, הִצִּיג מַצַּג שָׁוְא בַּיַּחַס לִרְכוּשׁוֹ וּנְכָסָיו לִפְנֵי שֶׁנִּכְרַת הַהֶסְכֵּם וְהֵפֵר אֶת הַחוֹבוֹת הַמֻּגְבָּרוֹת הַמּוּטָלוֹת עָלָיו.
מִכָּאן עֲתִירָתָהּ, כִּי יִקָּבַע שֶׁהַוִּתּוּרִים שֶׁעָשְׂתָה אֵינָם חָלִים, וּבְכָל מִקְרֶה אֵינָם חוֹסְמִים אוֹתָהּ מִקַּבָּלַת מְלוֹא הַכְּסָפִים הַמַּגִּיעִים לָהּ מִמִּכְלוֹל הָרְכוּשׁ, שֶׁהָיָה לְמֵשִׁיב בְּמוֹעֵד עֲרִיכַת הַהֶסְכֵּם.
לְטַעֲנָתָהּ, עַל הַמֵּשִׁיב לִמְסֹר לָהּ אֶת הַפְּרָטִים הָרֶלֶבַנְטִיִּים בַּיַּחַס לַהֲלִיךְ שֶׁהִתְקַיֵּם, לְטַעֲנָתָהּ, בְּמַס הַהַכְנָסָה וְיֵשׁ לְחַיְּבוֹ לְשַׁלֵּם לָהּ מַחֲצִית מִכָּל הַזְּכֻיּוֹת שֶׁיִּתְבָּרְרוּ בַּהֲלִיךְ, לְאַחַר שֶׁהַמֵּשִׁיב יַחְשֹׂף אֶת מְלוֹא הָרְכוּשׁ שֶׁהָיָה לוֹ בַּמּוֹעֵד הַקּוֹבֵעַ.
עוֹד טָעֲנָה הַמְּעַרְעֶרֶת כִּי לֹא הָיְתָה לָהּ אֶפְשָׁרוּת לַעֲרֹךְ בֵּרוּר עֻבְדָּתִי בַּיַּחַס לְנִכְסֵי הַמֵּשִׁיב, כַּאֲשֶׁר זֶה רִמָּה בְּהַצְלָחָה אֶת שִׁלְטוֹנוֹת הַמַּס, הֲרֵי יֵשׁ לְהָנִיחַ כִּי גַּם הִיא לֹא הָיְתָה מַצְלִיחָה.
לְטַעֲנָתָהּ, הִיא לֹא יָדְעָה כִּי יֵשׁ לְמֵשִׁיב רְכוּשׁ רַב, אֶלָּא רַק חָשְׁדָה בְּכָךְ, וְזֹאת שָׁעָה שֶׁהַמֵּשִׁיב לֹא סִפֵּק לָהּ מֵידָע אֶלָּא הֶעֱלִים זֹאת מִמֶּנָּה כְּמוֹ מֵרָשֻׁיּוֹת הַמַּס.
