בְּמוֹעֵד מְאֻחָר יוֹתֵר שֶׁלֹּא פֹּרַט חָתְמוּ בְּנוֹת הַזּוּג עַל הֶסְכֵּם שִׁיּוּךְ חוֹזִי עִם הַקִּבּוּץ לְצֹרֶךְ בְּנִיַּת בֵּית מְגוּרִים עַל מִשְׁבֶּצֶת קַרְקַע בַּשֶּׁטַח שֶׁל 400 מ"ר בְּמִסְגֶּרֶת הַהַרְחָבָה שֶׁל הַקִּבּוּץ, תּוֹךְ מִמּוּשׁ זְכֻיּוֹת בְּנִיָּה שֶׁל חֲבֵר קִבּוּץ.
הַתּוֹבַעַת הִצְטָרְפָה שׁוּב לַקִּבּוּץ בַּמַּעֲמָד שֶׁל "חֲבֵרָה חֲדָשָׁה".
בְּנִיַּת הַבַּיִת הֵחֵלָּה בִּשְׁנַת 2009, וְאֵלָיו נִכְנְסוּ בְּנוֹת הַזּוּג בִּשְׁנַת 2011 (וּלְהַלָּן: "הַבַּיִת").
לְטַעֲנַת הַתּוֹבַעַת יֵשׁ לְפָרֵק אֶת הַשִּׁתּוּף בַּבַּיִת וּלְאַזֵּן אֶת הַמַּשְׁאַבִּים בֵּין בְּנוֹת הַזּוּג בְּהֶתְאֵם לְהוֹרָאוֹת הַהֶסְכֵּם, וּלְהָשִׁיב לְהוֹרֶיהָ אֶת יִתְרַת הַהַלְוָאָה שֶׁנָּתְנוּ לִבְנוֹת הַזּוּג לְצֹרֶךְ בְּנִיַּת הַבַּיִת בְּסַךְ שֶׁל 1,200,000₪.
הַתּוֹבְעִים טוֹעֲנִים אַף הֵם לְהַלְוָאָה בְּסַךְ הָאָמוּר, וּמֵאַחַר שֶׁבִּתָּם הַתּוֹבַעַת אֵינָהּ חוֹלֶקֶת עַל הֶחְזֵר הַהַלְוָאָה, תּוֹבְעִים מֵהַנִּתְבַּעַת בִּלְבַד אֶת חֶלְקָהּ בַּהַלְוָאָה, בְּקִזּוּז הַסְּכוּמִים שֶׁכְּבָר שֻׁלְּמוּ עַל יְדֵי בְּנוֹת הַזּוּג עַל חֶשְׁבּוֹן הַחֲזָרָה.
מִנֶּגֶד, טוֹעֶנֶת הַנִּתְבַּעַת כִּי הִיא לֹא מַכִּירָה בְּתָקְפּוֹ שֶׁל הַהֶסְכֵּם וְעַל כֵּן אֵין לִפְעֹל עַל פִּיו.
כֵּן לִדְבָרֶיהָ הַכְּסָפִים שֶׁנִּתְּנוּ מֵהוֹרֵי הַתּוֹבַעַת נָתְנוּ בְּתוֹרַת מַתָּנָה וְלֹא בְּתוֹרַת הַלְוָאָה, וְעַל כֵּן אֵין הִיא חַיֶּבֶת בָּהּ.
עוֹד טוֹעֶנֶת הַנִּתְבַּעַת כִּי הַסִּכּוּם בֵּין בְּנוֹת הַזּוּג הָיָה כִּי כְּנֶגֶד מִמּוּשׁ זְכוּת הַשִּׁיּוּךְ שֶׁל הַנִּתְבַּעַת וּוִתּוּר עַל בֵּיתָהּ בַּקִּבּוּץ לָקְחָה עַל עַצְמָהּ הַתּוֹבַעַת לָשֵׂאת בְּהוֹצָאוֹת בְּנִיַּת הַבַּיִת.
עַל כֵּן לְטַעֲנָתָהּ בְּמִסְגֶּרֶת פֵּרוּק הַשִּׁתּוּף בַּבַּיִת, וּבְמִדָּה וְתִרְכֹּשׁ הַתּוֹבַעַת אֶת זְכֻיּוֹתֶיהָ "יֵשׁ לְהָשִׁיב לְתוֹבַעַת אֶת זְכוּת הַשִּׁיּוּךְ" (סְעִיף 32 לַסִּכּוּמִים).
