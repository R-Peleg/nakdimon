וְאִלּוּ בְּנוֹגֵעַ לַנַּחֲלָה קָבְעָה הַמְּנוּחָה כִּי הִיא לֹא תִּמְכֹּר אֶת הַנַּחֲלָה בִּימֵי חַיֶּיהָ וְעַל כֵּן צִוְּתָה הִיא, כִּי מְנַהֵל הָעִזָּבוֹן, אוֹתוֹ מִנְּתָה הִיא, יִמְכֹּר אֶת הַנַּחֲלָה בְּאֹפֶן מִיָּדִי בְּיוֹם מוֹתָהּ וִיחַלֵּק בְּאוֹתוֹ אֹפֶן אֶת הַתְּמוּרָה מִמֶּנָּה לְאַחַר תַּשְׁלוּם מְלוֹא חוֹבוֹת, הִתְחַיְּבֻיּוֹת, וְהוֹצָאוֹת בַּגִּנָּה.
הַמְּנוּחָה אַף הִסְבִּירָה בַּצַּוָּאָה כִּי לַמִּתְנַגֵּד הִיא מוֹתִירָה 50% וְזֹאת עַל מְנַת לְפַצּוֹת אוֹתוֹ בְּאֹפֶן מָלֵא וְסוֹפִי בְּגִין מְלוֹא טַעֲנוֹתָיו וּדְרִישׁוֹתָיו בְּעִנְיַן הַנַּחֲלָה, לְרַבּוֹת טַעֲנָתוֹ כְּ"בֵּן מַמְשִׁיךְ".
לִשְׁתֵּי בְּנוֹתֶיהָ, מִבַּקָּשׁוֹת הַקִּיּוּם, הוֹתִירָה 25% לְכָל אַחַת.
בְּיוֹם 12.4.24, כְּחֹדֶשׁ לְאַחַר מוֹת בַּעֲלָהּ הַמָּנוֹחַ וְכִשְׁנָתַיִם לִפְנֵי מוֹתָהּ, עָרְכָה הַמְּנוּחָה, אֶת הַצַּוָּאָה.
בְּצַוָּאָה זוֹ, בִּטְּלָה הַמְּנוּחָה אֶת הַצַּוָּאָה הַמֻּקְדֶּמֶת בְּאֹפֶן מְפֹרָשׁ, וְאַף נִשְּׁלָה אֶת הַמִּתְנַגֵּד מֵעִזְבוֹנָהּ תּוֹךְ שֶׁהִיא מְנַמֶּקֶת וּמַסְבִּירָה כִּי הַשִּׁנּוּי בַּצַּוָּאָה נוֹבֵעַ לְנֹכַח הִתְנַכְּרוּתוֹ שֶׁל הַמִּתְנַגֵּד וְהַהֲלִיךְ הַמִּשְׁפָּטִי אֲשֶׁר הִסְתַּיֵּם בפס"ד בְּיוֹם 2011.09.04.
לְפִיכָךְ, הַמְּנוּחָה הוֹרִישָׁה אֶת כָּל רְכוּשָׁהּ לִשְׁתֵּי בְּנוֹתֶיהָ -מִבַּקָּשׁוֹת הַקִּיּוּם כַּאֲשֶׁר כָּל אַחַת מֵהֶן קִבְּלָה 50%.
