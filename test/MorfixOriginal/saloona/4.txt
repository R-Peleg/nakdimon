בְּאַפְּרִיל 2016 אֻבְחַן בְּנִי בָּאָרִי, אָז פָּעוֹט בֶּן שָׁנָה וָחֵצִי עַל הַסְּפֶּקְטְרוּם הָאוֹטִיסְטִי.
לִבְאֵרִי 4 אֲחָיוֹת (בְּנֵיהֶן אֲחוֹת תְּאוֹמָה), הִתְגּוֹרַרְנוּ בְּקִבּוּץ אֵילוֹן סָמוּךְ לַגְּבוּל, הִגַּעְנוּ לַגָּלִיל עִם חֲלוֹם הִתְיַשְּׁבוּת,
בְּמַטָּרָה לְגַדֵּל אֶת יְלָדֵינוּ בְּחָסוּת הַחִנּוּךְ הַקִּבּוּצִי, הַשִּׁוְיוֹנִי, עַל עֶרְכֵי אַהֲבַת הַמּוֹלֶדֶת, בַּטֶּבַע.
הַכֹּל הָיָה נִפְלָא, עַד לְאִבְחוּנוֹ שֶׁל בְּאֵרִי.
עִם קַבָּלַת אִבְחוּנוֹ שֶׁל בְּאֵרִי עַל הַסְּפֶּקְטְרוּם הָאוֹטִיסְטִי בְּתִפְקוּד גָּבוֹהַּ, קִבַּלְנוּ הַחְלָטָה, בָּאָרִי יִהְיֶה מְשֻׁלָּב בַּחֲבֵרָה, הוּא יִגְדַּל לִהְיוֹת מִבּוֹגֵר יַצְרָנִי, וְהוּא גַּם מְסֻגָּל.
לִבְאֵרִי הִצְטָרְפָה מְתַוֶּכֶת, בַּיּוֹם הַשֵּׁנִי לַעֲבוֹדָתָהּ הִיא הִתְקַשְּׁרָה אֵלַי בּוֹכָה: " הֵם לֹא נוֹתְנִים לִי לְהוֹצִיא אֶת בְּאֵרִי מֵהַכְּלוּב (עֲגָלוּל) בַּטִּיּוּל", זֶהוּ הַמִּשְׁפָּט שֶׁשִּׁנָּה אֶת חַיַּי...
בַּפְּנִיָּה לַגּוֹרְמִים הַמִּקְצוֹעִיִּים נֶאֱמַר לִי מְפֹרָשׁוֹת שֶׁהַפְרָדָה שֶׁל בְּאֵרִי מֵחֲבֵרָיו תִּגְרֹם לוֹ נֶזֶק, אַךְ בַּקִּבּוּץ הִתְקַבְּלָה הַחְלָטָה - בָּאָרִי יוֹבֵל בַּכְּלוּב בִּזְמַן טִיּוּל.
בִּמְדִינַת יִשְׂרָאֵל חוֹקַק חֹק שִׁוְיוֹן זְכֻיּוֹת לָאֲנָשִׁים עִם מֻגְבָּלוֹת שֶׁאוֹסֵר אַפְלָיָה שֶׁל אָדָם עִם מֻגְבָּלוּת - לְדַאֲבוֹנִי בַּקִּבּוּץ לֹא אִפְשְׁרוּ לַחֹק לְהִכָּנֵס בַּשַּׁעַר הַצָּהֹב בְּמִקְרֶה שֶׁל בְּאֵרִי.
וְגֵרְשׁוּ אֶת מִשְׁפַּחְתֵּנוּ (סִיְּמוּ לָנוּ אֶת חוֹזֵה הַשְּׂכִירוּת) פָּתְחוּ לָנוּ אֶת הַשַּׁעַר הַחוּצָה.
בַּשְּׁנָתַיִם הָאַחֲרוֹנוֹת, אֲנִי מְנַהֶלֶת מַאֲבָק נֶגֶד אַפְלָיַת יְלָדִים עִם מֻגְבָּלוֹת, בַּכְּנֶסֶת, בַּתִּקְשֹׁרֶת, בָּרְשָׁתוֹת הַחֶבְרָתִיּוֹת,
בְּאֵרִי שֶׁלִּי יָצָא מְ"הַכְּלוּב" לְתָא טַיָּס שֶׁל מְטוֹס קְרָב - כִּי נִדְרֶשֶׁת רַק אִמָּא אַחַת שֶׁמַּאֲמִינָה בְּיֶלֶד אֶחָד בִּכְדֵי שֶׁהוּא יִגַּע בַּשָּׁמַיִם.,
בְּאָרִי מְשֻׁלָּב בַּגַּן "רָגִיל" וְהוּא מַתָּנָה מְבֹרֶכֶת לַגַּן וְלַחֲבֵרָה, הוּא יוֹדֵעַ לְלַמֵּד אֶת חֲבֵרָיו עֲרָכִים שֶׁאַף מְבֻגָּר לֹא יוֹדֵעַ לְלָמֵד, וְהוּא מַתָּנָה עֲנָקִית.
