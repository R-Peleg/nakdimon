חֵלֶק גָּדוֹל מֵהַמִּגְבָּלוֹת שֶׁאַתָּה מַצִּיג - הֵן עִנְיָן הִתְפַּתְּחוּתִי: כֵּיוָן שֶׁאַתָּה בְּרָמָה רוּחָנִית נְמוּכָה, אֵינְךָ חָשׁ בַּפְּגִיעָה שֶׁהַפְּעֻלּוֹת גּוֹרְמוֹת, וְלָכֵן חוֹשֵׁשׁ מִשִּׁנּוּי שֶׁיִּגְרֹם לַהַפְחָתָה מֵהַהֲנָאָה הַמְּלַוֶּה אוֹתָן.
תּוּכַל בְּהֶחְלֵט לְהִצְטָרֵף לְבֵית כְּנֶסֶת, וּלְהִתְנַסּוּת הַדְרָגָתִית בַּתְּפִלָּה, שְׁמִירַת שַׁבָּת, שְׁמִירַת כַּשְׁרוּת וכו',
וְכָךְ רָמָתֵךְ הָרוּחָנִית תַּעֲלֶה, וְתָחוּשׁ בְּעַצְמְךָ בַּנֶּזֶק מִפְּעֻלּוֹת מְסֻיָּמוֹת,
כָּךְ שֶׁתָּבִין יוֹתֵר, וְהַשִּׁנּוּי יִנְבַּע מִמְּךָ - וְלֹא מִתְּחוּשַׁת כֹּרַח חִיצוֹנִי.
חָשׁוּב לְצַיֵּן, חֵלֶק מֵהַדְּבָרִים שֶׁצִּיַּנְתָּ הֵם חֻמְרוֹת מְסֻיָּמוֹת, שֶׁאֲנָשִׁים לוֹקְחִים עַל עַצְמָם (שִׁירַת נָשִׁים), תּוֹךְ הֲבָנָה שֶׁל מַצָּבָם הָאִישִׁי (הַנָּאֶה מִנִּיתָ מִשִּׁירָתָהּ שֶׁל אִשָּׁה נְשׂוּאָה לְמָשָׁל, שֶׁעֲלוּלָה לְעוֹרֵר בְּךָ מַחְשְׁבוֹת זִמָּה כְּלַפֶּיהָ, וְאוּלַי אַף לְמַמְּשָׁן), חָלָק (עִסּוּק בָּאֶרוֹטִיקָה) הוּא עַצְמוֹ אִסּוּר.
וְחָלָק (לָצוּם בְּלִי לִשְׁתּוֹת מַיִם בְּיוֹם חַם) הוּא בְּעִקַּר מִגְבָּלָה חֲשִׁיבָתִית.
הַיּוֹם כִּמְעַט תָּמִיד נִמְצָאִים בְּמִתְחָם מְמֻזָּג, וְנִתָּן לְאַרְגֵּן שְׁהוּת כָּזוֹ כְּשֶׁיּוֹדְעִים מֵרֹאשׁ שֶׁמְּדֻבָּר בְּיוֹם צוֹם,
אוֹ לַחֲלוּפִין - לָלֶכֶת לְנוֹחַ אוֹ לַעֲצֹר אֶת הַצּוֹם, אִם אַתָּה חָשׁ שֶׁאִמַּצְתָּ עַצְמְךָ יוֹתֵר מִדַּי.
