שֵׁירוּת זֶה מְסַפֵּק מֵידָע לְגַבֵּי תַּהֲלִיךְ הַנְפָקַת דַּרְכּוֹן וְחִידּוּשׁ דַּרְכּוֹן.
כָּאן גַּם תּוּכְלוּ לְשַׁלֵּם עֲבוּר שֵׁירוּתֵי הַנְפָקַת הַדַּרָכּוֹן וְחִידּוּשׁוֹ.
חוֹבָה לְהַצִּיג תְּעוּדַת זֶהוּת בְּקַבָּלַת הַשֵּׁירוּת בְּלִשְׁכָּה.

תְּנָאִים לְקִיּוּם הַבְּחִינָה.
יֵשׁ לְהַחְתִּים אֶת הָרְשַׁ"ל 18 בְּאוֹפֶן אִישִׁי בְּמִשְׂרַד הָרִישּׁוּי טֶרֶם בִּיצּוּעַ מִבְחַן עִיּוּנֵי (תֵּיאוֹרְיָה).
יֵשׁ לְהֵירָשֵׁם לְמִּבְחַן הָעִיּוּנִי מֵרֹאשׁ. בְּמַהֲלַךְ הָרִישּׁוּם, תִּתְבַּקְּשׁוּ לְשַׁלֵּם עֲבוּר אִגֶרַת מִבְחָן.

רִישׁוּם לְמַעֲרֶכֶת הַהַזְדַּהוּת הַמֶּמְשַׁלְתִּית.

מַעֲרֶכֶת הַהַזְדַּהוּת הַמֶּמְשַׁלְתִּית מַעֲנִיקָה גִּישָׁה מְאוּבְטַחַת, מְהִירָה וְנוֹחָה לְשֵׁירוּתִים מֶמְשַׁלְתִּיִּים מְקַוְונִים.

עֲלֵיכֶם לְהֵירָשֵׁם לַמַּעֲרֶכֶת בְּאוֹפֶן חַד פַּעֲמִי.
לְּאַחַר הָרִישּׁוּם תּוּכְלוּ לְהִיכָּנֵס לַמַּעֲרֶכֶת בְּאֶמְצָעוּת שָׁם הַמִּשְׁתַּמֵּשׁ וְהַסִּיסְמָה שֶׁלָּכֶם.

ישׁ לְמַלֵּא אֶת טוֹפֶּס הַבַּקָּשָׁה לְהֶחְזֵר כַּסְפִּי וְלִשְׁלוֹחַ אוֹתוֹ דֶּרֶךְ הָאִינְטֶרְנֶט.

לְאַחַר מִשְׁלוֹחַ הַטּוֹפֶס, תִּתְקַבֵּל הוֹדָעָה אוֹטוֹמָטִית לִכְתוֹבֶת הַדּוֹאַר הָאֵלֶקְטְרוֹנִי עִם פְּרָטֵי הַפְּנִיָּיה.

רִישׁוּם פְּטִירָה לְמִי שֶׁנִּפְטַר בַּחוּץ לְאֶרֶץ רִישּׁוּםְ

בְּעַת פְּתִיחַת הַתִּיק בְּתַחֲנַת הַמִּשְׁטָרָה, מְקַבֵּל הַנִּפְגָּע שָׁם מִשְׁתַּמֵּשׁ וְסִיסְּמָה הַמְּאַפְשְׁרִים לוֹ לְהִיכָּנֵס לַמַּעֲרֶכֶת בְּכָל עֵת וּלְהִתְעַדְכֵן בַּמַצָּב הַתִּיק וּבְמַצָּב הֶחָשׁוּד אוֹ הַנֶּאֱשָׁם.
לִכְּנִיסָה לְמַעֲרֶכְת

ישׁ לְמַלֵּא וְלִשְׁלוֹחַ אֶת טוֹפֶּס הַבַּקָּשָׁה לְעִיּוּן בְּחוֹמֶר הַחֲקִירָה בְּנוֹשֵׂא דּוֹחוֹת תְּנוּעָה בְּאוֹפֶן מְקוּוָן בְּצֵירוּף הַמִּסְמָכִים הַנִּדְרָשִׁים.

הַשִּׁירוּת מְאַפְשֵׁר לְנֶאֱשְׁמִים בַּעֲבֵירוֹת תְּנוּעָה, שֶׁהוּגַּשׁ נֶגְדָּם דּוּ"חַ תְּנוּעָה, לְעַיֵּין בְּחוֹמֶר הַחֲקִירָה הַמִּתְנַהֶלֶת בְּעִנְיָינָם וּלְקַבֵּל הֶעָתָּק מִמֶּנּוּ.

הַקְצָבָה נוֹעֲדָה לְסַיֵּיעַ לְמִשְׁפָּחָה בְּהוֹצָאוֹת גִּידּוּל הַיְּלָדִים, וְהִיא מְשׁוּלֶּמֶת לְלֹא קֶשֶׁר לְהַכְנָסוֹת הַמִּשְׁפָּחָה.

תְּעוּדַת הַזְּכָּאוּת הִיא הַמִּסְמָךְ הַמְּאֻשֵׁר אֶת הַזְּכָּאוּת וּמְפָרֵט אֶת רָמַת הַסִּיּוּעַ, הַתְּנָאִים וְהַמִּגְבָּלוֹת שֶׁנִּקְבְּעוּ.
תוֹקְפָה הוּא שָׁנָה מִיּוֹם הַגָּשַׁת הַבַּקָּשָׁה.

אם אֵין בְּאֶפְשָׁרוּתֵנוּ לְסַפֵּק הַרְצָאָה כְּפִי שֶׁבִּיקַּשְׁתֶּם, נִתְאַם אֶתְכֶם מוֹעֵד אוֹ מְקוֹם חֲלוּפִיִּים.

הַגֶּשֶׁת הָעֶמְדָּה עַל יְדֵי בֶּן מִשְׁפָּחָה שֶׁל הַנִּפְגָּע מִדַּרְגָּה רִאשׁוֹנָה בְּמִקְרִים בָּהֶם הָעֲבֵירָה גָּרְמָה לְמוֹתוֹ.

בְּהֶתְאֵם לְסוּג הָעֲבֵירָה מִמֶּנָּה נִפְגַּעְתָּם, בְּאֶפְשָׁרוּתְכֶם לְקַבֵּל מֵידָע עַל הַהֲלִיךְ הַפְּלִילִי וְכֵן לְקַבֵּל הוֹדָעָה עַל זְכוּתְכֶם לְהַבִּיעַ עֶמְדָּה בְּמַהֲלַךְ הַהֲלִיךְ הַמִּשְׁפָּטִי בְּאֶמְצָעוּת מַעֲרֶכֶת נִפְגְּעֵי עֲבֵירָה (מָנ"ע) שֶׁל הַמִּשְׁטָרָה.

ישׁ לְהַדְפִּיס אֶת טוֹפֶּס הַבַּקָּשָׁה לְאִישּׁוּר חוֹזֵר עַל וִיתּוּר עַל אֶזְרָחוּת יִשְׂרְאֵלִית וּלְמַלֵּא אוֹתוֹ בִּכְתַב יָד.

לְאַחַר שֶׁמִּילְּאתֶם אֶת הַטּוֹפֶס יֵשׁ לְהַגִּיעַ בְּאוֹפֶן אִישִׁי בְּצֵירוּף הַמִּסְמָכִים הַנִּדְרָשִׁים לְלִשְׁכַּת רָשׁוּת הָאוּכְלוּסִין אוֹ לִנְצִיגוּת יִשְׂרָאֵל בַּחוּץ לָאָרֶץ.

ישׁ לְהוֹרִיד אֶת טוֹפֶּס הַבַּקָּשָׁה לְמַתַּן אִישּׁוּר עַל אֶזְרָחוּת אֶרֶץ יִשְׂרְאֵלִית (פָּלַשְׂתִּינָאִית) לְהַדְפִּיס וּלְמַלֵּא אֶת הַפְּרָטִים בִּכְתַב יָד.

לְאַחַר שְׁמִּילְּאתֶם אֶת הַטּוֹפֶס יֵשׁ לְהַגִּיעַ בְּאוֹפֶן אִישִׁי בְּצֵירוּף הַמִּסְמָכִים הַנִּדְרָשִׁים לְלִשְׁכַּת רָשׁוּת הָאוּכְלוּסִין וְהַהֲגִּירָה הַקְּרוֹבָה לְמְקוֹם מְגוּרֵיכֶם.

ישׁ לְהַדְפִּיס וּלְמַלֵּא אֶת הַפְּרָטִים הַנִּדְרָשִׁים בְּטוֹפֶס הַבַּקָּשָׁה לְאִישּׁוּר עַל אִי רְכִישַׁת אֶזְרָחוּת יִשְׂרְאֵלִית, לְצָרֵף אֶת הַמִּסְמָכִים הַנִּדְרָשִׁים וּלְהֲגִּישָׁם בִּנְצִיגוּת יִשְׂרָאֵל בְּחוּ"ל אוֹ בְּאַחַת מְלַשְׁכוֹת רָשׁוּת הָאוּכְלוּסִין וְהַהֲגִירָה.

שִׁירוּת זֶה מְאַפְשֵׁר לְהַגִּישׁ בַּקָּשָׁה לְאִישּׁוּר הַמֵּעִיד עַל מַעֲמָד שֶׁל "לֹא אֶזְרָח" וְעַל מַעֲמָד שֶׁהָייה בְּיִשְׂרָאֵל.

הַשִּׁירוּת נִיתָּן לְמִי שֶׁהוּגְּשָׁה בַּעֲבוּרָם בְּעֵבֶר הַצְהָרַת אִי-רָצוֹן לִהְיוֹת אֶזְרָחִים יִשְׂרְאֵלִים וְעוֹד לֹא עָבְרָה שָׁנָה מִתוֹם שֵׁירוּתָם הַצְּבָאִי אוֹ שֶׁהֵם בְּגִיל 18 עַד 22 (לְפִי הַתַּאֲרִיךְ הַמְּאוּחָר יוֹתֵר)

ישׁ לְהַדְפִּיס וּלְמַלֵּא אֶת טוֹפֶּס הוֹדָעַת וִיתּוּר אוֹ בִּיטּוּל - הַצְהָרַת אִי רָצוֹן לִהְיוֹת אֶזְרָח יִשְׂרְאֵלִי, אוֹ לְמַלֵּא טוֹפֶס מְקוּוָן שֶׁל הוֹדָעַת וִיתּוּר אוֹ בִּיטּוּל – הַצְהָרַת אִי רָצוֹן לִהְיוֹת אֶזְרָח יִשְׂרְאֵלִי (הַטּוֹפֶס נִפְתַּח בְּדַפְדְּפָן אֶקְסְפְלוֹרְר בִּלְבַד) וּלְהַדְפִּיסוֹ

עַל מַגִּישׁ הַהַצְהָרָה לִבְדּוֹק כִּי הַמִּבְנִים שֶׁבַּדִּיוּוּחַ הֵם פְּשׁוּטִים מִבְּחִינָה הַנְדָּסִית, וְעוֹמְדִים בִּדְרִישׁוֹת הַתַּקָּנוֹת.

רְשָׁאִים לְהַגִּישׁ אֶת הַבַּקָּשָׁה בְּנֵי מִשְׁפָּחָה שֶׁל יְהוּדִים שֶׁנִּכְנְסוּ לָאָרֶץ לִפְנֵי הַ-19 בְּמַארְס 1970, וְקִיבְּלוּ רִישְׁיוֹן לִישִׁיבַת קֶבַע.

ישׁ לְהוֹרִיד אֶת טוֹפֶּס הַהוֹדָעָה עַל רָצוֹן לְקַבֵּל אֶזְרָחוּת מִכּוֹחַ הַשָּׁבוֹת, לְהַדְפִּיס וּלְמַלֵּא אֶת הַפְּרָטִים בִּכְתַב יָד.

לְאַחַר שֶׁמִּילְּאתֶם אֶת הַטּוֹפֶס יֵשׁ לְהַגִּיעַ בְּאוֹפֶן אִישִׁי בְּצֵירוּף הַמִּסְמָכִים הַנִּדְרָשִׁים לְלִשְׁכַּת רָשׁוּת הָאוּכְלוּסִין וְהַהֲגִּירָה הַקְּרוֹבָה לְמְקוֹם מְגוּרֵיכָם.

שִׁירוּת זֶה מְאַפְשֵׁר לְהַגִּישׁ בַּקָּשָׁה לְהַעֲנָקַת אֶזְרָחוּת יִשְׂרְאֵלִית לִמְבַקְּשִׁים מִתַּחַת לְגִיל 18, שֶׁנּוֹלְדוּ בַּחוּץ לָאָרֶץ לְאֶזְרָחִים יִשְׂרְאֵלִים וְשֶׁאֵינָם זַכָּאִים לְאֶזְרָחוּת מִכּוֹחַ לְיָדָה.

ישׁ לְמַלֵּא אֶת טוֹפֶּס הַבַּקָּשָׁה לְהַעֲנָקַת אֶזְרָחוּת יִשְׂרְאֵלִית לְמְבַקֵּשׁ מִתַּחַת גִּיל 18 תּוֹשַׁב חוּ"ל לְהַדְפִּיס וּלְמַלֵּא אֶת הַפְּרָטִים בִּכְתַב יָד.

לְאַחַר שֶׁמִּילְּאתֶם אֶת הַטּוֹפֶס יֵשׁ לְהַגִּישׁ בְּצֵירוּף הַמִּסְמָכִים הַנִּדְרָשִׁים בְּלִשְׁכַּת רְשׁוּת אוּכְלוּסִין וְהַהֲגִירָה אוֹ בִּנְצִיגוּת יִשְׂרָאֵל בִּמְקוֹם מְגוּרֵיכֶם בַּחוּץ לָאָרֶץ.

ישׁ לְהַדְפִּיס אֶת טוֹפֶס הַבַּקָּשָׁה לְנַלְוֶה לְאַנְשֵׁי דָּת (אָ/3) וְלַסְטוּדֶנְטִים (אָ/2) - אָ/4, לְמַלֵּא אֶת הַפְּרָטִים וּלְצָרֵף אֵלָיו אֶת הַמִּמִסְמְכִים הַדְּרוּשִׁים

לַמִתְגּוֹרְרִים בַּחוּץ לָאָרֶץ נִיתָּן לְהַגִּישׁ אֶת הַמִּסְמָכִים בִּנְצִיגוּת יִשְׂרָאֵל הַקְּרוֹבָה לְמָקוֹם מִגוֹרִיךְ

הַשֵּׁירוּת נִיתָּן לִסְטוּדֶנְטִים וּלְתַלְמִידֵי יְשִׁיבוֹת שֶׁאֵינָם תּוֹשָׁבֵי יִשְׂרְאֵלֶ

לְאַחַר שְׁמִּילְּאתֶם אֶת הַטּוֹפֶס לְאַשְׁרַת כְּנִיסָה לִסְטוּדֶנְט אוֹ לְתַלְמִיד יְשִׁיבָה יֵשׁ לְהַגִּיעַ בְּאוֹפֶן אִישִׁי, בְּצֵירוּף הַמִּסְמָכִים הַנִּדְרָשִׁים, לְלִשְׁכַּת רָשׁוּת הָאוּכְלוּסִין וְהַהֲגִּירָה הַקְּרוֹבָה לְמְקוֹם מְגוּרֵיכָם אוֹ לִנְצִיגוּת יִשְׂרָאֵל הַסְּמוּכָה לְמְקוֹם מְגוּרֵיכָם.

בְּמְקְרָה שֶׁל הָאֲרָכֶת רִישָּׁיוֹן לְשָׁנָה נוֹסֶפֶת יֵשׁ לְהַמְצִיא גִּילְיוֹן צִיּוֹנִים

אם הַמּוּזְמָנִים נְשׂוּאִים, יֵשׁ לְצָרֵף אִישּׁוּרִים עַל מַצָּבָם הָאִישִׁי כּוֹלֵל פְּרָטֵי בְּנִי אוֹ בְּנוֹת הַזּוּגֶ

אִישׁוּר תַּשְׁלוּם הָאַגְרָה

חוֹבָה נוֹכְחוּת אִישִׁית שֶׁל נְצִיגֵי הַמּוֹסָד הָאָקָדֵמִי הַמַּזְמִין

רְשָׁאִים לְהַגִּישׁ אֶת הַבַּקָּשָׁה אֶזְרָחִים אוֹ תּוֹשָׁבֵי קָבְעַ

בְּמְקְרִים שֶׁבָּהֶם מְדִינַת מוֹצָאָם אוֹ אֶזְרָחוּתָם שֶׁל הַמּוּזְמָנִים מוּגְדֶּרֶת  עַל-יְדֵי גּוֹרְמֵי הַבִּיטָּחוֹן כִּמְדִינַת אִבְחוּן אוֹ סִיכּוּן, יֵשׁ לְמַלֵּא טוֹפֶס קוֹרוֹת חַיִּים מְקוּצָר שֶׁיּוֹעֲבֵר לְאִישּׁוּר בִּיטְחוֹנִי.

תּוֹקֶף הַדַּרְכּוֹן הַזָּר שֶׁל הַמּוּזְמָנִים יִהְיֶה לִתְקוּפָה שֶׁל שִׁישָּׁה חוֹדָשִׁים מֵעֵבֶר לִתְקוּפַת הָרִישְׁיוֹן הַמְּבוּקָּשׁ.

מִשְׂרַד הַתַּיָּירוּת זִיהָה פּוֹטֶנְצְיָאל גָּבוֹהַּ לִתְנוּעַת תַּיָּירוּת יוֹמִית שֶׁל נוֹפְשִׁים זָרִים הַמַּגִּיעִים לְבִיקּוּר בְּסִינַי אוֹ בְּטוּרְקִיָּה, וּמְעוּנְיָינִים בְּבִיקּוּר יוֹמִי בְּיִשְׂרָאֵל.

הֵחֵל מִפֶבְּרוּאָר 2007 הַנְּהִיגָה רָשׁוּת הָאוּכְלוּסִין וְהַהֲגִּירָה הֲקָלוֹת מְיוּחָדוֹת לְהַגָּשַׁת בַּקָּשׁוֹת אַשְׁרַת תַּיֵּיר קְבוּצָתִית לְתַיָּירוּת יוֹמִית שֶׁל נוֹפְשִׁים מִטּוּרְקִיָּה וּמִסִינֵי הַמְּבַקְּשִׁים לְבַקֵּר בְּיִשְׂרָאֵל.

בְּרוֹב הַמִּקְרִים מְדוּבָּר בַּבִיקּוּרִים בִּלְתִּי מְתוּכְנָנִים הַיְּכוֹלִים לְהַוּוֹת סִיּוּר הֶיכֵּרוּת שֶׁיְּעוֹרֵר עִנְיָין וְרָצוֹן בְּקֶרֶב הַתַּיָּירִים, לְהַגִּיעַ שֵׁנִית לְבִיקּוּר מַקִּיף יוֹתֵר.

סוּכְנֵי נְסִיעוֹת הַמְּבַקְּשִׁים לְהָבִיא קְבוּצוֹת מִמְּדִינוֹת מוּגְבָּלוֹת, יֵשׁ לְצַיֵּין זֹאת בְּטוֹפֶס הַבַּקְשָׁה

עַל הַסּוֹכְנִים מַגִּישִׁי הַבַּקָּשָׁה בְּיִשְׂרָאֵל, לַעֲרוֹךְ בְּדִיקוֹת מַקִּיפוֹת לְגַבֵּי הַתַּיָּירִים.

הגַּעַת הַקְּבוּצָה לְאָרֶץ מוּתָנִית בְּקַבָּלַת אַשְׁרַת כְּנִיסָה בְּכָפוּף לְמַסְלוּל הַטִּיּוּל שֶׁהוּצַּג בְּבַקָּשָׁה, כְּנִיסַת חַבְרֵי הַקְּבוּצָה בְּיַחַד, שֶׁהָייה בְּאָרֶץ כַּקְבוּצָה וִיצִיאָתָם מֵהָאָרֶץ בְּיַחַד בְּתוֹם תְּקוּפַת הָרִישְׁיוֹן.

רְשָׁאִים לְהַגִּישׁ אֶת הַבַּקָּשָׁה תּוֹשָׁבִים זָרִים הַשּׁוֹהִים בְּיִשְׂרָאֵל וּמְבַקְּשִׁים לְהַאֲרִיךְ אֶת שֶׁהִיִּיתָם

דְּרְכוֹן תָּקַף לַתְקוּפָה הָעוֹלָה עַל שִׁישָּׁה חוֹדָשִׁים מֵעֵבֶר לִשְׁהוֹת הַמְּבוּקֶשְׁת

הַזְמֶנֶת תּוֹר - יֵשׁ לִזְמַן תּוֹר לְמַחְלֶקֶת אֲשְׁרוֹת וְלִבְחוֹר אֶת הַלִּשְׁכָּה הָרְצוּיה

תַיּירִים שֶׁבְּעֵת הַזְמַנְתֶּם הוּפְקְדָה עֲרָבוֹת בִּנְקָאִית לְהַבְטָחַת יְצִיאָתָם, יֵשׁ לְבַקֵּשׁ אֶת הַאֲרָכַת תּוֹקֶף הָעֲרָבוֹת בְּעֵת מַתַּן הָאֲרֻכָּה.

הַמַּזְמִינִים תּוֹר יְקַבְּלוּ עֲדִיפוּת בְּקַבָּלַת שֵׁירוּת בַּלִשְׁכָּה.

בְכֵּל בַּקָּשָׁה לִתְעוּדַת מַעֲבָר, יֵשׁ לִבְדּוֹק אֶת הַתְּנָאִים הַנִּדְרָשִׁים בְּהֶתְאֵם לְגִיל הַמְּבַקֵּשׁ.

בְּכְּדֵי לְקַבֵּל אֶת הַשֵּׁירוּת יֵשׁ לְהַזְמִין תּוֹרְ

תְּעוּדַת הַמַּעֲבָר תּוֹעֵבַר לַבֵית הַלָּקוּחַ תּוֹךְ כְּ-3 שָׁבוּעוֹת.

נוֹכְחוֹת אִישִׁית חוֹבָה לִמְבַקְּשֵׁי הַשֵּׁירוּת וּלְהוֹרֵיהֶם

מִקְרָה שֶׁל אַבְדָן, גְּנֵיבָה אוֹ הַשְׁחָתָה שֶׁל מִסְמָךְ נְסִיעָה, בְּעֵת הַשְּׁהִיָּיה בַּחוּץ לָאָרֶץ, יֵשׁ לְהַגִּיעַ לִנְצִיגוּת יִשְׂרָאֵל הַקְּרוֹבָה לְשֵׁם הַנְּפָקֶת מִסְמַךְ נְסִיעָה חָדָשׁ.

תּוֹקְפוֹ שֶׁל דַּרְכּוֹן, שֶׁאֵינוֹ בְּיוֹמֶטְרִי, נִיתָּן לְהַאֲרָכָה בִּתְנָאֵי שֶׁמִּיּוֹם נִיפּוֹקוֹ וְעַד תּוֹם תְּקוּפַת הַהַאֲרָכָה לֹא חָלְפוּ יוֹתֵר מִ-5 שָׁנִים.

ישׁ לְהַגִּישׁ אֶת הַבַּקָּשָׁה לְמַעֲמַד בְּתוֹךְ שְׁלוֹשָׁה חוֹדָשִׁים מִמּוֹעֵד הַלֵּידָה.

ישׁ לְהַגִּיעַ עִם כָּל הַמִּסְמָכִים הַמִּסְמָכִים הַדְּרוּשִׁים לְלִשְׁכַּת רָשׁוּת הָאוּכְלוּסִין וְהַהֲגִּירָה הַקְּרוֹבָה לְמְקוֹם מִגוֹרֵיכָם

הַגֶּשֶׁת הַבַּקָּשָׁה תֵּיעָשֶׂה בְּנוֹכְחוּת שְׁנֵי הַהוֹרִים.

הַמְבַקְּשִׁים יוּפְנוּ לְלִשְׁכַּת הַנִּישּׂוּאִין בָּהּ הוּנְפְּקָה הַתְּעוּדָה.

ישׁ לְמַלֵּא אֶת טוֹפֶּס הַבַּקָּשָׁה הַמְּקוּוָׂן לְאִיתּוּר תְּעוּדַת נִישּׂוּאִין עַל פִּי הַהַנְחָיוֹת.

ישׁ לְהַדְפִּיס אֶת טוֹפֶּס הַבַּקָּשָׁה הַיְּדָנִי לְאִיתּוּר תְּעוּדַת נִישּׂוּאִין, וּלְמַלֵּא בִּכְתַב יָד אֶת הַפְּרָטִים הַנִּדְרָשִׁים.

ישׁ לְצָרֵף לְטוֹפֶּס זֶה אִישּׁוּר רוֹאָה חֶשְׁבּוֹן עַל הַנְהָלַת חֶשְׁבּוֹנוֹת לְפִי הַשִּׁיטָה הַכְּפוּלָה הַכּוֹלֶלֶת גַּם כִּרְטֶסַת מְלַאי, חוֹמְרֵי גֶּלֶם / חֲלָקִים וּמוּצָרִים מוּגְמָרִים.

אָם יִימָּצֵא כִּי הַבַּקָּשָׁה מוּלָאָה שֶׁלֹּא כַּנִּדְרָשׁ, הִיא תּוּחְזַר לְמַגִּישׁ הַבַּקָּשָׁה וְלֹא תְּטוּפָּל

ישׁ לְצַיֵּין כִּי בְּמִקְרִים מְסוּיָּימִים הַמְּבַקֵּשׁ יִידָּרֵשׁ לְהַפְקִיד עַרָבִים כִּתְנַאֵי לִתְעוּדָה.

נוֹכְחוּת אִישִׁית חוֹבָה.

נִיתָן לְבַקֵּשׁ תְּעוּדָה דּוּ לְשׁוֹנִית בְּשָׂפוֹת עִבְרִית אוֹ אַנְגְּלִית.

לְאַחַר שֶׁמִּילְּאתֶם אֶת הַטּוֹפֶס יֵשׁ לְהַגִּישׁ בְּאוֹפֶן אִישִׁי בְּצֵירוּף הַמִּסְמָכִים הַנִּדְרָשִׁים בְּלִשְׁכַּת רָשׁוּת הָאוּכְלוּסִין וְהַהֲגִּירָה הַקְּרוֹבָה לְמְקוֹם מְגוּרֵיכֶם.

הַשִּׁירוּת נִיתָּן לִבְנִי אוֹ לִבְנוֹת הַזּוּג שֶׁל הַנִּפְטָרִים, לְיַלְדֵיהֶם אוֹ לְהוֹרֵיהֶם.

תְּעוּדַת פְּטִירָה נִיתֶּנֶת עַל שָׁם מִי שֶׁנִּפְטְרוּ בָּאָרֶץ בִּלְבַד.

בְּהַגֶּשֶׁת בַּקָּשָׁה בְּלִשְׁכַּת הָרָשׁוּת, הִתְעוּדָה תּוּדְפֵּס וְתִימָסֵר בִּמָקוֹם בְּמִידָּה וְהַנִּפְטָר עוֹדְכָן בְּמִרְשָׁם הָאוּכְלוּסִין.

שֵׁנֵי הַצְּדָדִים צְרִיכִים לִהְיוֹת רִשׁוּמִים בְּמִשְׂרַד הַפְּנִים כְּבַעַל וְאִישָּׁה.

הַצְּדָדִים יְקַבְּלוּ הַחְלָטָה מְקוֹרִית מְאוּשֶּׁרֶת בְּחוֹתֶמַת בֵּית הַדִּין בְּאֶמְצָעוּת הַדּוֹאַר אוֹ בְּמַזְכִּירוֹת בֵּית הַדִּין הַשְּׁרְעִי בּוֹ הִתְקַבְּלָה הַחְלָטָה.

כְמוֹ כֵּן מוּמְלָץ לְפָרֵט בִּפְנִיָּיה נְתוּנִים כְּגוֹן: פִּרָטֵי הַיַּצְרָן, אֶרֶץ הַיְּצוּר וּפִרָטֵי הַיְּבוּאָן וְכֵן לְצָרֵף לְתּוֹכֶן הַפְּנִיָּיה מִסְמָכִים רֵלֵוַונְטִיִּים, כְּגוֹן: קַבָּלוֹת, חֶשְׁבּוֹנִיּוֹת, חוֹזֶה, צִילּוּמִים אוֹ כָּל מוּצָּג אַחֵר הֶעָשׂוּי לְסַיֵּיעַ.

כְּדֵי שֶׁהַבַּקָּשָׁה שֶׁלָּכֶם תְּקַבֵּל תַּאֲרִיךְ בֵּינְלְאוּמִּי, עֲלֵיכֶם לְצָרֵף אֶת כָּל הַמִּסְמָכִים הַנִּדְרָשִׁים.

עֲלֵיכֶם לְשַׁלֵּם אֶת הָאֲגָרוֹת הַיִּשְׂרְאֵלִיּוֹת וְגַם אֶת הָאִגָרוֹת הַבֵּינְלְאוּמִּיּוֹת.

בְּמְקְרִים בָּהֶם הָאֲתָר הַמְּקוּוָׂן לֹא פָּעִיל, בְּאֶפְשָׁרוּתְכֶם לְהַגִּישׁ בַּקָּשָׁה בְּאֶמְצָעוּת הַטּוֹפֶס הַבָּא.

ישׁ לְמַלֵּא אֶת טוֹפֶּס הַבַּקָּשָׁה הַמְּקוּוָׂן לְאִישּׁוּר צִיּוּד אַלְחוֹטִי, לְצָרֵף אֶת הַמִּסְמָכִים הַנִּדְרָשִׁים וּלְבַצֵּעַ שְׁלִיחָה.

ישׁ לְמַלֵּא אֶת הַתַּצְהִיר עַל פִּי דְּרִישׁוֹת הָרָשׁוּת הָאַרְצִית לִכַבָּאוֹת וְהַצָּלָה, וּלְהַגִּישׁוֹ לִרְשׁוּת הַמְּקוֹמִית, לְאַחַר חֲתִימָה וְאֵימּוּת שֶׁל עוֹרֵךְ דִּין.

לְמְעֵט אִם צוּיַּן אַחֶרֶת, עֲלֵיכֶם לְהַגִּישׁ מִסְמָכִים מְקוֹרִיִּים בִּלְבַד וּבַשָּׂפָה רִשְׁמִית.

הַפָנִיָה תְּטוּפֵל עַל בְּסִיס בְּחִינַת הַדִּיוּוּחַ שֶׁהוּגַּשׁ, אֲשֶׁר מָצוּי בַּתִּיק הַחֶבְרָה, וְאֵין צוֹרֶךְ לְהַגִּישׁ דִּיוּוּחִים נוֹסָפִים.

ישׁ לְמַלֵּא אֶת הַטּוֹפֶס הַמְּקוּוָׂן לְדִיוּוחַ עַל סִרְבָנִי וְסַרְבָנִיּוֹת גֶּט בִּפְּרָטִים הַיְּדוּעִים לָכֶם, וּלְהַקִּישׁ עַל כַּפְתּוֹר "שָׁלַח".

נִיתֶן לִשְׁלוֹחַ הוֹדָעָה בְּדוֹאַר אֵלֶקְטְרוֹנִי וּלְצָרֵף כַּמָּה שֶׁיּוֹתֵר פְּרָטִים כְּמוֹ: הַמָּקוֹם בּוֹ נִרְאָהַ/ נִרְאֲתָה הַסָּרְבָּןָ/ הַסַּרְבָנִית, תַּאֲרִיךְ וְשָׁעָה, תֵּיאוּר הַלְּבוּשׁ, מְקוֹם עֲבוֹדָה, אֲנָשִׁים הַנִּמְצָאִים אִיתּוּ/ אִיתָּהּ וְעוֹד.

נִיתָן לְשַׁלֵּם עֲבוּר חוֹב סְפֵּצִיפִי אוֹ בְּגִין חוֹבוֹת עָבָר.

לְאַחַר הַגָּשַׁת בַּקָּשָׁה נִיתָּן לְהַשְׁלִים מִסְמָכִים (כְּמוֹ: מַאֲזָן שָׂכָר לִימּוּד, מִכְתַּב אִישִׁי) בְּדֶרֶךְ מְקַוֶּוֶנֶת בְּאֶמְצָעוּת טוֹפְס מְקוּוָן לְהַגָּשַׁת מִסְמָכִים נוֹסָפִים לְמִינְהָל.

ישׁ לְמַלֵּא אֶת הַפְּרָטִים הָאִישִׁיִּים וּלְאַחַר מִכֵּן לְצָרֵף אֶת הַמִּסְמָכִים וְהָאִישּׁוּרִים הַנִּדְרָשִׁים.

לְחִישׁוּב הַדְרָגָה לְזַכָּאוֹת לְסַבְּסוֹד נִיתָּן לְהִיכָּנֵס לְסִימּוּלָאטוֹר לְחִישּׁוּב דַּרְגָה עַל בְּסִיס הַכְנָסָה לְנֶפֶשׁ בְּדַף הָאִינְטֶרְנֵט שֶׁל מִשְׂרַד הָעֲבוֹדָה, הָרְוָוחָה וְהַשֵּׁירוּתִים הַחֶבְרָתִיִּים, וּלְמַלֵּא אֶת הַפְּרָטִים.

לְאַחַר שְׁלִיחַת הַבַּקָּשָׁה, יִישָּׁלְחוּ עִדְכּוּנִים לִכְתוֹבֶת הַדּוֹאַר הָאֵלֶקְטְרוֹנִי שֶׁהַגְדָּרָתָם.

לְקַבֹּלַת מֵידָע מוּרְחָב בִּדְבַר הוֹדָעָה שֶׁנִּשְׁלְחָה אֲלֵיכֶם, נִיתָּן לִפְנוֹת לְמֶרְכַּז WTO-TBT, וּלְקַבֵּל אֶת מִסְמַךְ הַתַּקָנָה הַמָּלֵא.
