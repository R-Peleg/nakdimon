‫הרשויות הפדרליות הֵחֵלּוּ מָצוֹד אַרְצִי אַחֵר ויליאם בוצ'ר.‬
‫אלה וְוָט. זוֹ הֲפִיכָה דְּפוּקָה מבפנים.‬
‫אני יוֹדֵעַ מֵי הָרַג אֶת ריינר.‬
‫משהו שֶׁקָּשׁוּר לַמְּחַבֵּל-עַל שֶׁהוּבְרַח לג'רזי.‬
‫אחיה.‬
‫אני רַק רוֹצֶה אֶת אִשְׁתִּי בחזרה.‬
‫תביא אֶת הַמְּחַבֵּל ללנגלי וּתְקַבֵּל אותה.‬
‫נאמר לִי שֶׁהוּא לָעוֹלָם לֹא יָדַע עַל ריאן אוֹ עליי.
זֶה הָיָה ההסכם.‬
‫אתם חַיָּיבִים לִגְרוֹם לוֹ לְעָזוּב, כִּי אַחֶרֶת אֶקַּח אֶת ריאן,‬ ‫ואברח.‬
‫גיבורי-עַל לֹא נוֹלָדִים אֶלָּא נוֹצָרִים בְּעֶזְרַת תִּרְכּוֹבֶת וי.‬
‫יש לָנוּ מֵחֶבֶל-עַל מָתַחְתָּ לסיפון,‬ ‫והמצב יָכוֹל לְהִשְׁתַּבֵּשׁ בְּדִיּוּק ככה.‬
‫ממש שטני.‬
‫אמרתי שֶׁהוּא שלי.‬
‫מי שֶׁמְּנַמְנֵם מַפְסִיד, סבא'לה.‬
‫הגיבורה הַחֲדָשָׁה שֶׁלָּנוּ, סטורמפרונט, עָצְרָה אֶת המחבל‬ ‫לפני שֶׁגַּבָּה חַיִּים נוֹסָפִים שֶׁל חַפִים מפשע.‬
‫תקשורת הִיא הכול.‬
‫סטיב וַאֲנִי יָצָאנוּ בְּמֶשֶׁךְ שְׁנָתַיִים כְּשֶׁהוּא הִצִּיעַ לִי נישואים.‬
‫בסעודה שֶׁלִּפְנֵי הַחֲתוּנָּה, אָבִי אמר‬ ‫שכדאי שֶׁנִּבְחַן תָּוכְנִית חִיסָּכוֹן לילדינו.‬
‫סטיב צָחַק וְאָמַר, "זֶה הָיָה מְטוֹרָף," וש"לעולם לֹא יִהְיוּ לָנוּ ילדים."‬
‫מה שֶׁהָיָה חַדֵּשׁ לִי. כן.‬
‫בקצרה, המסעיד הָיָה סָבְבָה עִם הכול.‬
‫אבל ה"מריוט" לֹא הֶחְזִירוּ אֶת הַמִּקְדָּמָה שֶׁשִּׁילַּמְנוּ עַל אוּלָם הַנְּשָׁפִים, אז...‬ ‫כן, תקשורת.‬
