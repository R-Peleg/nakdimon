אַנְדַּרְטַת הַקִּבּוּצִים
אַנְדַּרְטַת הַקִּבּוּצִים לְזֵכֶר חַבְרֵי הַקִּבּוּצִים שֶׁנָּפְלוּ בְּמַעַרְכוֹת יִשְׂרָאֵל, נִמְצֵאת בְּיַעַר הַקִּבּוּצִים סָמוּךְ לְמִשְׁמַר הָעֵמֶק.
הַקָּמָה
הָרַעְיוֹן לַהֲקָמַת הָאַנְדַּרְטָה נִתְקַל בְּהִתְנַגְּדֻיּוֹת רַבּוֹת. הַדִּילֶמָה הָעִקָּרִית הָיְתָה בְּעֶצֶם נְחִיצוּת הֲקָמָתָהּ הֱיוֹת שֶׁבִּמְקוֹמוֹת רַבִּים בָּאָרֶץ יֵשׁ אַנְדַּרְטָאוֹת לְחַבְרֵי הַקִּבּוּצִים שֶׁנָּפְלוּ בְּמַעַרְכוֹת יִשְׂרָאֵל. לְבַסּוֹף נִתְקַבְּלָה הַהַחְלָטָה וְנִבְחַר הַמָּקוֹם - גִּבְעָה נִשְּׂאָה בְּיַעַר מִשְׁמַר הָעֵמֶק. הַמָּקוֹם נִבְחַר בִּגְלַל הֱיוֹתוֹ צוֹפֶה עַל עֵמֶק יִזְרְעֶאל, עֶרֶשׂ הַהִתְיַשְּׁבוּת, וְקִרְבָתוֹ לג'וערה וְלִמְעָרַת הפלמ"ח בְּקִבּוּץ שֶמִּשְׁמַר הָעֵמֶק, דָּבָר הַמְּאַפְשֵׁר לִיצֹר אֶת הַמְּשֻׁלָּשׁ הַוִּירְטוּאָלִי שֶׁל מוֹרֶשֶׁת הַהֲגָנָה.
מִשְׂרַד הַבִּטָּחוֹן הִתְנָה אֶת אִשּׁוּרוֹ וּתְמִיכָתוֹ בִּתְנַאי שֶׁלֹּא יִהְיֶה צִיּוּן שְׁמוֹת הַחֲלָלִים בָּאֲתָר. הָעֳלָה רַעְיוֹן לִבְנוֹת עֶמְדַּת מַחְשֵׁב בַּמָּקוֹם וּבָהּ רְשִׁימַת הַחֲלָלִים. לְבַסּוֹף הֻחְלַט לַחְקֹק בָּאֲתָר אֶת שְׁמוֹת הַנּוֹפְלִים מֵהֶחָלָל הָרִאשׁוֹן מֹשֶׁה בֶּרְסְקִי חֲבֵר קִבּוּץ דְּגַנְיָה א עַד יָמֵינוּ, סה"כ 3050 שֵׁמוֹת.
הָאַנְדַּרְטָה תֻּכְנְנָה עַל יְדֵי הַפֶּסֶל סַלּוֹ שָׁאוּל חֲבֵר קִבּוּץ עֵין גֵּב. הִיא בְּנוּיָה בַּצּוּרָה שֶׁל מִגְדַּל תַּצְפִּית עָשׂוּי מִבֶּטוֹן חָשׂוּף. גֶּרֶם מַדְרֵגוֹת מַגִּיעַ לְרֹאשׁ הַמִּגְדָּל וּמִמֶּנּוּ תַּצְפִּית מַרְהִיבָה עַל קִבּוּץ מִשְׁמַר הָעֵמֶק וְעֵמֶק יִזְרְעֶאל. עַל קִיר בֶּטוֹן בָּאֲתָר חֲקוּקוֹת מִלּוֹתָיו שֶׁל ס. יִזְהָר: נִזְכֹּר אַחִים, בָּנִים וְרָעִים אֲשֶׁר יָצְאוּ עִמָּנוּ בִּפְלֻגּוֹת הַלּוֹחֲמִים. אֲנַחְנוּ חָזַרְנוּ וְהֵמָּה לֹא יָשׁוּבוּ עוֹד. קֶרֶן קַיֶּמֶת לְיִשְׂרָאֵל הִכְשִׁירָה לָאַנְדַּרְטָה דֶּרֶךְ כֻּרְכַּר מְשֻׁלֶּטֶת, בְּאֹרֶךְ 6 ק"מ מג'וערה.
בִּתְחִלַּת הַדֶּרֶךְ, מִצַּד שְׂמֹאל, אֶפְשָׁר לְהַבְחִין בִּשְׁנֵי עַמּוּדֵי אֶבֶן גְּדוּמִים - גָּבוֹהַּ וְנָמוּךְ לְזָכָר שְׁנֵי חַבְרֵי קִבּוּץ עֵין הַשּׁוֹפֵט. הַגָּבוֹהַּ לְזֵכֶר שְׁלֹמֹה קוּרְץ, הַנָּמוּךְ לְזֵכֶר נֶכְדּוֹ לִיאוֹר צ'רן. בְּ-7 בְּאוֹקְטוֹבֶּר 1989 יָרְדוּ הַשְּׁנַיִם לַחְקֹר מְעָרָה בָּאֵזוֹר וְנִקְבְּרוּ חַיִּים.
הָאֲתָר נֶחְנַךְ רִשְׁמִית בְּ-29 בְּאוֹגוּסְט 2007 בַּטֶּקֶס בְּהִשְׁתַּתְּפוּת שַׂר הַבִּטָּחוֹן אֵהוּד בָּרָק וּנְשִׂיא הַמְּדִינָה שִׁמְעוֹן פֶּרֶס.
