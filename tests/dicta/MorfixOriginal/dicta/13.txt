אֵלִיָּהוּ מִזְרָחִי
רַבִּי אֵלִיָּהוּ מִזְרָחִי (הרא"ם; 1435-1526,רפ"ו) הָיָה פּוֹסֵק הֲלָכָה, אִישׁ מַדָּע, וּמְדִינַאי יְהוּדִי-טוּרְקִי. רַבָּה הָרָאשִׁי שֶׁל יַהֲדוּת טוּרְקִיָּה וּמַנְהִיגָה, רֹאשׁ יְשִׁיבָה וְאֶחָד מִשְּׁלֹשֶׁת פַּרְשָׁנָיו הָעִקָּרִיִּים שֶׁל פֵּרוּשׁ רש"י לַתּוֹרָה.
תּוֹלְדוֹת חַיָּיו
הרא"ם נוֹלַד בְּקוּשְׁטָא בֵּין הַשָּׁנִים 1435 - 1437 (וְיֵשׁ הַגּוֹרְסִים 1450), לְמִשְׁפָּחָה רומניוטית מִן הַבַּלְקָן. לָמַד תּוֹרָהּ אֵצֶל רַבִּי אֵלִיָּהוּ הַלֵּוִי בֶּן רַבִּי מֹשֶׁה הַלֵּוִי הָאַשְׁכְּנַזִּי, וְאֵצֶל מָרְדְּכַי כומטינו לָמַד מָתֶמָטִיקָה וְאַסְטְרוֹנוֹמְיָה.
הֶעֱמִיד תַּלְמִידִים רַבִּים, בְּחָכְמַת הַתּוֹרָה וּבְחָכְמוֹת חִיצוֹנִיּוֹת (הרא"ם שָׁלַט גַּם בַּעֲרָבִית וּבִיוָנִית וְנֶאֱמַר עָלָיו כִּי הָיָה הָרִאשׁוֹן שֶׁגִּילָה שִׁיטָה לִמְצִיאַת שֹׁרֶשׁ רִבּוּעִי). בֵּין הַתַּלְמִידִים שֶׁהֶעֱמִיד הָיוּ גַּם תַּלְמִידִים קָרָאִים וְזֹאת לְמוֹרַת רוּחוֹ שֶׁל הָרַב מֹשֶׁה אֵלִיָּהוּ קפשאלי, שֶׁהָיָה הַסַּמְכוּת הַתּוֹרָנִית הָעֶלְיוֹנָה בָּאִימְפֶּרְיָה הָעוֹתְ'מָאנִית בְּאוֹתָהּ עֵת. בְּכָךְ וּבְנוֹשְׂאִים נוֹסָפִים הִגִּיעַ לְעִמּוּת עַם הָרַב קאפסלי. עִם זֹאת, עַל אַף הוֹרָאָתוֹ שֶׁל הָרַב קוֹלוֹן (המהרי"ק) שֶׁלֹּא לְצַיֵּת לָרַב קאפסלי בְּתַפְקִידוֹ כֶּחָכָם בָּאשִׁי (תַּפְקִיד שֶׁכָּלַל גַּם סַמְכֻיּוֹת לִגְבִיַּת כְּסָפִים שׁוֹנִים מֵהַיְּהוּדִים עֲבוּר הַשֻּׂלְטָאן), נִמְנַע הרא"ם בַּעֲצַת רַבּוֹ, הָרַב יְהוּדָה מינץ, מִלְּהַבִּיעַ עֶמְדָּה בַּסּוּגְיָה.
חָכָם בָּאשִׁי[1]
לְאַחַר מוֹת הָרַב מֹשֶׁה אֵלִיָּהוּ קפשאלי בְּ-1495 מוֹנֶה בִּמְקוֹמוֹ הרא"ם לְחָכָם בָּאשִׁי, וּלְמַעֲשֶׂה הָיָה לַחֲכַם הבאשי הָאַחֲרוֹן. בִּטֵּל אֶת הַחֲרָמוֹת שֶׁהוּטְלוּ עַל לִמּוּד תּוֹרָה לְקָרָאִים וְנִמֵּק אֶת הַדָּבָר בְּכָךְ שֶׁקִּנְאַת תַּלְמִידִים תַּרְבֶּה חָכְמָה[2]. בְּסוֹבְלָנוּתוֹ הָיָה הרא"ם יוֹצֵא מִן הַכְּלָל בִּזְמַנּוֹ וּמְקוֹמוֹ. אֲבָל לַמְרוֹת זֶה בִּסֵּס אֶת אִסּוּר הַחִתּוּן עִם הַקָּרָאִים[3].
הרא"ם גִּילָה יַחַס מֵקֵל לְהַתָּרַת עֲגוּנוֹת מִטַּעֲמֵי צְנִיעוּת. הרא"ם קָבַע בְּדַעַת מִעוּט כִּי גַּם לְאַחַר חֵרֶם דְּרַבֵּנוּ גֵּרְשֹׁם, מָקוֹם בּוֹ יַפְקִיד בַּעַל אֶת הַגֵּט וְהַכְּתֻבָּה בִּידֵי בֵּית הַדִּין, יִהְיֶה פָּטוּר מִתַּשְׁלוּם דְּמֵי מְזוֹנוֹת לְאִשְׁתּוֹ. הִלְכַת מִעוּט זוֹ אָמְנָם לֹא אֻמְּצָה עַל יְדֵי מַרְבִּית הָרַבָּנִים, אַךְ אֻמְצָה בְּבָתֵּי הַמִּשְׁפָּט לְעִנְיְנֵי מִשְׁפָּחָה, בְּמָקוֹם בּוֹ קַיָּם פֵּרוּד אָרֹךְ בֵּין בְּנֵי זוּג, וְהָאִשָּׁה דּוֹחָה אֶפְשָׁרוּת חֲזָרָה לְחַיֵּי מִשְׁפָּחָה, רְאוּ לְמָשָׁל כָּאן.
הרא"ם הִתְנַגֵּד בַּחֲרִיפוּת לְמִנְהָגִים שֶׁנִּתְפְּסוּ בְּעֵינָיו כִּגְרוּעִים וְאָמַר כִּי "מִנְהָג גָּרוּעַ שֶׁאֵינוֹ עַל פִּי הָעִקָּר, מִנְהָג בָּטֵל הוּא אֵצֶל כָּל אָדָם וְאֵין מְחַיְּבִים לָלֶכֶת אַחֲרָיו אַף עַל גַּב דאתפשט אִסּוּרֶיהָ לְכוֹלֵא מִתָּא (=אַף עַל גַּב שֶׁהָאִסּוּר פָּשַׁט בְּכָל אוֹתָהּ הָעִיר) וְאַף עַל גַּב דְּהוּי גַּבֵּי אסורא...". וְעוֹד כָּתַב "דאין כֹּחַ הַמִּנְהָג יָכוֹל לְהִתְפַּשֵּׁט עַל מִי שֶׁאֵינוֹ רוֹצֶה בְּאוֹתוֹ מִנְהַג בְּעַל כָּרְחוֹ".
הרא"ם הִתְפַּרְסֵם בְּיַדְעָנוּתוֹ, וְעָסַק הַרְבֵּה בְּצָרְכֵי צִבּוּר לַמְרוֹת חֻלְשָׁתוֹ הַגּוּפָנִית, וְעִם מִנּוּיוֹ לְחָכָם בָּאשִׁי הָיָה מִיּוֹעֲצֵי הַשֻּׂלְטָאן בְּאִיזְמִיר, וְהָיָה לוֹ מָקוֹם בַּדִּיוָאן עַם הַפַּטְרִיאַרְךְ וְהַמֻּפְתִּי. בְּיָמָיו הִתְחִילָה כְּנִיסַת הֲמוֹנֵי יְהוּדִים לְטוּרְקִיָּה בְּעִקְבוֹת גֵּרוּשׁ סְפָרַד, וְהָרְאֵם עָשָׂה מַאֲמַצִּים לְהָקֵל עַל הַבָּאִים אֶת כְּנִיסָתָם לָאָרֶץ הַחֲדָשָׁה.
הרא"ם נִפְטַר בְּגִיל מֻפְלָג בְּקוּשְׁטָא בִּשְׁנַת 1526. חֲתָנוֹ, רַבִּי מְשֻׁלָּם מוולטירה הֵבִיא לְבִטּוּל הָרַבָּנוּת הָרָאשִׁית בתוגרמה (טוּרְקִיָּה) לְאַחַר פְּטִירָתוֹ. הָרִאשׁוֹן לְצִיּוֹן רַבִּי נִסִּים חַיִּים מֹשֶׁה מִזְרָחִי וְיִשְׂרָאֵל מֵאִיר מִזְרָחִי מִיְּרוּשָׁלַיִם הֵם מִצֶּאֱצָאָיו שֶׁל הרא"ם.
סְפָרָיו
הַמִּזְרָחִי - פֵּרוּשׁ עַל פרש"י שֶׁעַל הַתּוֹרָה (ויניציא 1526, 1545) - סָפַר שְׁבוֹ הֶעֱמִיק לְפָרֵשׁ אֶת דִּבְרֵי רש"י, לְהָגֵן עֲלֵיהֶם מֵהַשָּׂגוֹת הַמִּפְרָשִׂים הָאַחֲרוֹנִים (וּבִמְיֻחָד מֵהַשָּׂגוֹת הרמב"ן) אַךְ גַּם לַחְלֹק עָלָיו בִּמְקוֹמוֹת רַבִּים. הַסֵּפֶר נֶחְשַׁב לְאַחַד הַפֵּרוּשִׁים הַמֶּרְכָּזִיִּים לרש"י. הַסֵּפֶר מֵבִיא לְמַעֲשֶׂה אֶת הַמְּקוֹרוֹת מֵהַתַּלְמוּד וּמֵהַמִּדְרָשׁ עֲלֵיהֶם בִּסֵּס הרש"י אֶת פֵּרוּשׁוֹ. תַּמְצִיתוֹ מוֹפִיעָה בַּסֵּפֶר "שְׂפָתִי חֲכָמִים" (עִם פֵּרוּשׁוֹ שֶׁל המהר"ל מִפְּרָג).
מְלֶאכֶת הַמִּסְפָּר - עַל חָכְמַת הַמִּסְפָּר הַחֶשְׁבּוֹן והתשבורת וְהַהַנְדָּסָה (קוּשְׁטָא 1534) - מַטְּרַת סֵפֶר זֶה הִיא לְלַמֵּד כָּל אָדָם אֶת יְסוֹדוֹת הַחֶשְׁבּוֹן וְאֶת אַרְבַּע פְּעֻלּוֹת הַחֶשְׁבּוֹן: קִבּוּץ (חִבּוּר), חִסּוּר, הַכָּאָה (כֶּפֶל) וְחִלּוּק, בַּמִּסְפָּרִים שְׁלֵמִים, בִּשְׁבָרִים וּבְשִׁבְרֵי הַתְּכוּנָה. אֶת הַסֵּפֶר הוֹצִיא לָאוֹר בְּנוֹ שֶׁל הרא"ם, רַבֵּי יִשְׂרָאֵל[4], לְאַחַר פְּטִירָתוֹ. הַסֵּפֶר כְּתוּב עִבְרִית וְתֻרְגַּם לְלָטִינִית, הִתְפַּרְסֵם בַּחֲשִׁיבוּתוֹ וְשִׁמֵּשׁ כַּסְפַּר יְסוֹד לְהוֹרָאַת הַחֶשְׁבּוֹן. בְּסֵפֶר כָּלוּל גַּם פֶּרֶק עַל לִמּוּדֵי הַשַּׁח, רָאוּ כָּאן.
תּוֹסְפֵי סמ"ג (קוּשְׁטָא 1541) - חִדּוּשִׁים בְּסֵפֶר מִצְווֹת גָּדוֹל.
צוּרַת הָאָרֶץ - חִבּוּר עַל צוּרַת הָאָרֶץ וּתְמוּנָתָהּ.
שְׁאֵלוֹת וּתְשׁוּבוֹת (קוּשְׁטָא 1546) - מֵאָה שְׁאֵלוֹת וּתְשׁוּבוֹת בְּשִׁבְתּוֹ עַל כֵּס הַהוֹרָאָה.
מַיִם עֲמֻקִּים (ויניציה 1647) - שְׁאֵלוֹת וּתְשׁוּבוֹת חֵלֶק ב'.
פְּסָקִים שֶׁפֻּרְסְמוּ בְּסֵפֶר בֵּית יוֹסֵף לְרַבִּי יוֹסֵף קארו
סְפָרִים שֶׁלֹּא נִדְפְּסוּ
בִּיאוֹר עַל סֵפֶר אלמגיסטי - סֵפֶר עַל תְּנוּעַת גַּרְמֵי הַשָּׁמַיִם - הַסֵּפֶר לֹא נִדְפַּס אַךְ קַיָּם כְּתַב יָד בְּסֶנַט פֶּטֶרְבּוּרְג.
פֵּרוּשׁ לַיְּסוֹדוֹת שֶׁל אוקלידס - הַסֵּפֶר אָבַד.
לִקְרִיאָה נוֹסֶפֶת
א' עוֹבְדֶיהָ, כְּתָבִים נִבְחָרִים כָּרַךְ א, יְרוּשָׁלַיִם תש"ב, עמ' 98-63.
