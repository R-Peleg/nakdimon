אוּזְבֶּקִיסְטָן בַּמִּשְׂחָקִים הָאוֹלִימְפִּיִּים
אוּזְבֶּקִיסְטָן מִשְׁתַּתֶּפֶת בַּמִּשְׂחָקִים הָאוֹלִימְפִּיִּים מֵאָז אוֹלִימְפְּיָאדַת אַטְלַנְטָה (1996). עַד 1991 הָיְתָה אוּזְבֶּקִיסְטָן חֵלֶק מִבְּרִית הַמּוֹעָצוֹת, וּסְפּוֹרְטָאֶיהָ הִתְחָרוּ בְּמִסְגֶּרֶת הַמִּשְׁלַחַת הַסּוֹבְיֶטִית. בְּאוֹתָהּ שָׁנָה הִתְפָּרְקָה בְּרִית הַמּוֹעָצוֹת לְ-15 מְדִינוֹת נִפְרָדוֹת, וְ-12 מֵהֶן, בָּהֶן אוּזְבֶּקִיסְטָן, שָׁלְחוּ מִשְׁלַחַת מְאֻחֶדֶת לְאוֹלִימְפְּיָאדַת בַּרְצֶלוֹנָה (1992). בְּאַרְבַּע הָאוֹלִימְפְּיָאדוֹת בָּהֶן הִשְׁתַּתְּפָה אוּזְבֶּקִיסְטָן, צָבְרוּ הַסְּפּוֹרְטָאִים הָאוּזְבֶּקִים 32 מֶדַלְיוֹת, מֵהֶן 9 מֶדַלְיוֹת זָהָב. הָעֲנָפִים הַבּוֹלְטִים בְּיוֹתֵר הֵם הֵאָבְקוּת וְאִגְרוּף.
הִיסְטוֹרְיָה
הַמִּשְׁלַחַת הָעַצְמָאִית הָרִאשׁוֹנָה שֶׁל אוּזְבֶּקִיסְטָן, בְּאוֹלִימְפְּיָאדַת אַטְלַנְטָה, כָּלְלָה 71 סְפּוֹרְטָאִים. שְׁנַיִם מֵהֶם זָכוּ בַּמֶּדַלְיוֹת הָרִאשׁוֹנוֹת בְּתוֹלְדוֹתֶיהָ: הַגּ'וּדוֹקָא אֲרַמֵּן בגדסרוב זָכָה בְּמֶדַלְיַת כֶּסֶף וְהַמִּתְאַגְרֵף כָּרִים טולגנוב זָכָה בְּמֶדַלְיַת אָרָד. בְּאוֹלִימְפְּיָאדַת סִידְנִי (2000) נָטְלוּ חֵלֶק 70 סְפּוֹרְטָאִים אוּזְבֶּקִים. הַמִּתְאַגְרֵף מחמטקודיר עבדולאייב זָכָה בְּמֶדַלְיַת הַזָּהָב הָרִאשׁוֹנָה בְּתוֹלְדוֹת אוּזְבֶּקִיסְטָן (בְּמִשְׂחֲקֵי הַקַּיִץ). שְׁנֵי מִתְאַגְרְפִים נוֹסָפִים, סרגיי מיחאילוב ורוסטם סאידוב, זָכוּ בְּמֶדַלְיוֹת אָרָד, וּמִתְאַגְרֵף רְבִיעִי, רוסלאן צ'גאייב, הֶעְפִּיל לִשְׁלַב רֶבַע גְּמָר. בְּהֵאָבְקוּת בָּלֶט ארתור טאימזוב, שֶׁזָּכָה בְּמֶדַלְיַת כֶּסֶף.
גַּם בְּאוֹלִימְפְּיָאדַת אָתוּנָה (2004) הִשְׁתַּתְּפוּ 70 סְפּוֹרְטָאִים אוּזְבֶּקִים. טאימזוב זָכָה בְּמֶדַלְיַת זָהָב, וְכָמוֹהוּ גַּם הַמִּתְאַבֵּק אָלֶכְּסַנְדֶּר דוקטורשווילי. מִתְאַבֵּק שְׁלִישִׁי, מְגֻמָּד איברגימוב זָכָה בְּמֶדַלְיַת כֶּסֶף. וְהַמִּתְאַגְרְפִים בחודירג'ון סולטונוב ואוטקירבק חיידרוב זָכוּ בְּמֶדַלְיוֹת אָרָד, וּשְׁלוֹשָׁה מִתְאַגְרְפִים אוּזְבֶּקִים נוֹסָפִים, טולאשבוי דוניורוב, דילשוד מחמודוב ושרזוד חוסאנוב, הֶעְפִּילוּ לִשְׁלַב רֶבַע הַגְּמָר. בַּהֲרָמַת מִשְׁקוֹלוֹת בָּלֶט אָלֶכְּסַנְדֶּר אורינוב, שֶׁסִּיֵּם בַּמָּקוֹם הַשְּׁמִינִי. עַל הִשְׁתַּתְּפוּתָהּ שֶׁל אוּזְבֶּקִיסְטָן בְּמִשְׂחָקִים אֵלֶּה הֵעִיבָה פְּסִילָתָהּ שֶׁל מְיַדֶּה הַפַּטִּישׁ אוֹלְגָה שצ'וקינה, בְּשֶׁל שִׁמּוּשׁ בַּסַּמִּים.
לְאוֹלִימְפְּיָאדַת בֵּיגִ'ינְג (2008) שָׁלְחָה אוּזְבֶּקִיסְטָן מִשְׁלַחַת קְטַנָּה יוֹתֵר, בַּת 58 סְפּוֹרְטָאִים. הַמִּתְאַבֵּק ארתור טאימזוב זָכָה בְּמֶדַלְיַת זָהָב שְׁנִיָּה בִּרְצִיפוּת, וַעֲמִיתוֹ סוסלן טיגייב זָכָה בְּמֶדַלְיַת כֶּסֶף, אַךְ זוֹ נִשְׁלְלָה מִמֶּנּוּ בְּ-2016 בְּשֶׁל הָרְשָׁעָה באמרוץ. דילשוד מנסורוב סִיֵּם בַּמָּקוֹם הַחֲמִישִׁי וְקָרְבַּן קורבנוב בַּמָּקוֹם הַשְּׁבִיעִי. הַגּ'וּדוֹקָא עבדולו טנגרייב זָכָה בְּמֶדַלְיַת כֶּסֶף, בְּעוֹד שרישוד סובירוב זָכָה בְּמֶדַלְיַת אָרָד. ג'וּדוֹקָא נוֹסַף, מֵרֵאָלִי שריפוב, סִיֵּם בַּמָּקוֹם הַחֲמִישִׁי. בְּנוֹסָף, שְׁנֵי מִתְעַמְּלִים אוּזְבֶּקִים זָכוּ לָרִאשׁוֹנָה בְּמֶדַלְיוֹת אָרָד - אנטון פוקין בְּתַרְגִּיל הַמַּקְבִּילִים ויקטרינה חִלְּקוֹ, שֶׁהָיְתָה לַסְּפּוֹרְטָאִית הָאוּזְבֶּקִית הָרִאשׁוֹנָה, וְהַיְּחִידָה עַד כֹּה, שֶׁזּוֹכֶה בַּמֶּדַלְיָה הָאוֹלִימְפִּית בְּמִשְׂחֲקֵי הַקַּיִץ - בַּטְּרַמְפּוֹלִינָה. חוֹתֵר הַקָּנוּ וְאֵדִים מִנִּקּוּב הָיָה קָרוֹב לִזְכִיָּה רִאשׁוֹנָה בַּמֶּדַלְיָה גַּם בְּעָנָף זֶה, אַךְ סִיֵּם בַּמָּקוֹם הָרְבִיעִי בְּמִקְצֵה הַסְּפְּרִינְט לְ-1,000 מֶטֶר. בַּעֲנַף הַטַּקְוַנְדוֹ בָּלַט אקמאל אירגשב, שֶׁסִּיֵּם בַּמָּקוֹם הַחֲמִישִׁי. הַמִּתְאַגְרְפִים הָאוּזְבֶּקִים לֹא הִצְלִיחוּ בְּמִשְׂחָקִים אֵלּוּ לִזְכּוֹת בַּמֶּדַלְיוֹת, אַךְ שְׁנַיִם מֵהֶם, דילשוד מחמודוב ואלשוד רסולוב, הֶעְפִּילוּ לִשְׁלַב רֶבַע הַגְּמָר.
בְּאוֹלִימְפְּיָאדַת לוֹנְדוֹן (2012) הִשְׁתַּתְּפוּ 54 סְפּוֹרְטָאִים אוּזְבֶּקִים. טאימזוב זָכָה בְּמֶדַלְיַת זָהָב שְׁלִישִׁית בִּרְצִיפוּת, שֶׁהָיְתָה גַּם מֶדַלְיַת הַזָּהָב הַיְּחִידָה לִזְכוּתָהּ שֶׁל הַמִּשְׁלַחַת. מִתְאַבֵּק נוֹסַף, סוסלאן טיגייב, זָכָה בְּמֶדַלְיַת אָרָד, אַךְ זוֹ נִשְׁלְלָה מִמֶּנּוּ חֳדָשִׁים סְפוּרִים לְאַחַר הַמִּשְׂחָקִים, לְאַחַר שֶׁנִּמְצָא כִּי נֵטֶל סַמִּים. שְׁנֵי סְפּוֹרְטָאִים אוּזְבֶּקִים נוֹסָפִים שֶׁזָּכוּ בַּמֶּדַלְיוֹת הָיוּ הַגּ'וּדוֹקָא רישוד סובירוב וְהַמִּתְאַגְרֵף אֵבוּס אטויב, שֶׁזָּכוּ בְּמֶדַלְיוֹת אָרָד. אַרְבָּעָה מִתְאַגְרְפִים נוֹסָפִים הֻדְּחוּ בִּשְׁלַב רֶבַע הַגְּמָר. סְפּוֹרְטָאִים אוּזְבֶּקִים נוֹסָפִים שֶׁבָּלְטוּ בְּלוֹנְדוֹן כָּלְלוּ אֶת חוֹתֵר הֵקֵאנוּ וְאֵדִים מִנִּקּוּב (מָקוֹם רְבִיעִי בַּמִּקְצֶה לְ-1000 מֶטְרִים), מֵרִים הַמִּשְׁקוֹלוֹת רוסלן נורודינוב (מָקוֹם הָרְבִיעִי), רוֹכֵב הָאוֹפַנַּיִם סרגיי לגוטין (מָקוֹם חֲמִישִׁי בְּמֵרוֹץ אוֹפַנֵּי כְּבִישׁ), הַגּ'וּדוֹקָא רמזידין (מָקוֹם חֲמִישִׁי), הַמִּתְאַבֵּק קָרְבַּן קורבנוב (מָקוֹם חֲמִישִׁי), מֵרִים הַמִּשְׁקוֹלוֹת איוואן אפרמוב (מָקוֹם חֲמִישִׁי); הָאַתְלֶטִית סבטלנה רדזיוויל (מָקוֹם שְׁבִיעִי בַּקְּפִיצָה לַגֹּבַהּ) וְהַגּ'וּדוֹקָא דילשוד צ'ורייב (מָקוֹם שְׁבִיעִי).
אוֹלִימְפְּיָאדַת ריו דֶּה ז'ניירו (2016) הָיְתָה הַמֻּצְלַחַת בְּיוֹתֵר בְּתוֹלְדוֹת אוּזְבֶּקִיסְטָן, עִם 13 מֶדַלְיוֹת, מֵהֶן 4 מֶדַלְיוֹת זָהָב, הֶשֵּׂג דּוֹמֶה לְהֶשֵּׂגֵי הַמִּשְׁלָחוֹת הָאוּזְבֶּקִיּוֹת בִּשְׁלוֹשׁ הָאוֹלִימְפְּיָאדוֹת הַקּוֹדְמוֹת גַּם יַחַד. הָאִגְרוּף הָיָה הֶעָנָף הַבּוֹלֵט בְּיוֹתֵר. הַמִּתְאַגְרְפִים הִתְחָרוּ בְּכָל עֶשֶׂר קָטֵגוֹרְיוֹת הַמִּשְׁקָל לַגְּבָרִים, וּבְשֶׁבַע מֵהֶן זָכוּ בַּמֶּדַלְיוֹת. חסנבוי דוסמטוב, שאחובידין זוירוב ופזלידין גאיבנזארוב זָכוּ בְּמֶדַלְיוֹת זָהָב, שחראם גיאסוב ובקטמיר מליקוזייב זָכוּ בְּמֶדַלְיוֹת כֶּסֶף, ומורוג'ון אחמדלייב ורוסטם טולגנוב זָכוּ בְּמֶדַלְיוֹת אָרָד. בְּנוֹסָף, סִיְּמוּ הורשיד טוז'יבייב ובחודיר ז'לולוב בַּמָּקוֹם הַחֲמִישִׁי. בְּמֶדַלְיַת הַזָּהָב הָרְבִיעִית לִזְכוּת הָאוּזְבֶּקִים זָכָה מֵרִים הַמִּשְׁקוֹלוֹת רוסלאן נורודינוב. איוואן אפרמוב סִיֵּם בַּמָּקוֹם הַחֲמִישִׁי ורוסטם ג'נגבייב בַּמָּקוֹם הַשִּׁשִּׁי. בְּהֵאָבְקוּת זָכוּ שְׁלוֹשָׁה אוּזְבֶּקִים בְּמֶדַלְיוֹת אָרָד: איחטיור נברוזוב, מְגֻמָּד איברגימוב ואלמורט טסמורדוב. בקזוד עבדורחמונוב סִיֵּם בַּמָּקוֹם הַחֲמִישִׁי. הַגּ'וּדוֹקָא רישוד סובירוב זָכָה בְּמֶדַלְיַת אָרָד שְׁלִישִׁית בִּרְצִיפוּת, וְגַם דיורבק אורוזבוייב זָכָה בְּמֶדַלְיַת אָרָד. עבדולו טנגרייב סִיֵּם בַּמָּקוֹם הַחֲמִישִׁי. סְפּוֹרְטָאִים אוּזְבֶּקִים נוֹסָפִים שֶׁבָּלְטוּ בָּרְיוֹ דֶּה ז'ניירו כָּלְלוּ אֶת לוֹחֵם הַטַּקְוַנְדוֹ דמיטרי שוקין שֶׁסִּיֵּם בַּמָּקוֹם הַחֲמִישִׁי, אֵת חוֹתֵר הֵקֵאנוּ גראסים קוצ'נב שֶׁסִּיֵּם בַּמָּקוֹם הַשְּׁבִיעִי בַּמִּקְצֶה לְ-1000 מֶטֶר וּבַמָּקוֹם הַשְּׁמִינִי בְּמִקְצֵה הַזּוּגוֹת לְ-1000 מֶטֶר, יַחַד עִם סריק מירבקוב, וְאֶת הַמִּתְעַמֶּלֶת הַוָּתִיקָה אוקסנה צ'וסוביטינה בַּת הַ-41, שֶׁסִּיְּמָה בַּמָּקוֹם הַשְּׁבִיעִי בְּתַרְגִּיל הַקְּפִיצוֹת.
מִשְׂחֲקֵי הַחֹרֶף
אוֹלִימְפְּיָאדַת הַחֹרֶף הָרִאשׁוֹנָה בָּהּ הִשְׁתַּתְּפָה אוּזְבֶּקִיסְטָן הָיְתָה אוֹלִימְפְּיָאדַת לילהאמר (1994). הָיְתָה זוֹ אוֹלִימְפְּיָאדַת הַחֹרֶף הַיְּחִידָה עַד כֹּה בָּהּ זָכָה סְפּוֹרְטַאי אוּזְבֶּקִי בַּמֶּדַלְיָה - לִינָה צ'ריאזובה זָכְתָה בְּמֶדַלְיַת זָהָב בִּסְקִי חָפְשִׁי. פְּרָט לְהֶשֵּׂג זֶה, הַהֶשֵּׂג הַגָּבוֹהַּ הַיָּחִיד הַנּוֹסָף שֶׁל הַמִּשְׁלַחַת הָאוּזְבֶּקִית נִרְשַׁם בְּאוֹלִימְפְּיָאדַת נאגנו (1998), אָז דֹּרְגָה הַמַּחְלִיקָה הָאָמָּנוּתִית טטיאנה מלינינה בַּמָּקוֹם הַשְּׁמִינִי.
