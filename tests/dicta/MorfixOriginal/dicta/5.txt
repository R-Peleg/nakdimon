אַדְרִיכָלוּת קוֹנְסְטְרוּקְטִיבִיסְטִית
אַדְרִיכָלוּת קוֹנְסְטְרוּקְטִיבִיסְטִית הוּא סִגְנוֹן בָּאַדְרִיכָלוּת הַמּוֹדֶרְנִית שֶׁשִּׂגְשֵׂג בִּבְרִית הַמּוֹעָצוֹת בִּשְׁנוֹת הַ-20 וּבִשְׁנוֹת הַ-30 הַמֻּקְדָּמוֹת, כְּחֵלֶק מִתְּנוּעַת הַקּוּנְסְטְרוּקְטִיבִיזְם. סִגְנוֹן זֶה שִׁלֵּב טֶכְנוֹלוֹגְיָה וְהַנְדָּסָה מִתְקַדְּמִים יַחַד עִם מַטָּרָה חֶבְרָתִית קוֹמוּנִיסְטִית מֻצְהֶרֶת. הַסִּגְנוֹן הִתְחַלֵּק לְמִסְפַּר זִרְמֵי מְשַׁנֶּה שֶׁהִתְחָרוּ אֶחָד בִּשְׁנֵי וְנוֹצְרוּ בּוֹ פְּרוֹיֶקְטִים עִצּוּבִיִּים חֲלוּצִיִּים שֶׁלֹּא הִגִּיעוּ לִידֵי מִמּוּשׁ, כְּמוֹ גַּם מִבְנִים מֻגְמָרִים. תְּחִלָּתוֹ שֶׁל הַסִּגְנוֹן הָיְתָה לַכִּוּוּן הַתֵּאוֹרֶטִי וְהָאָוַנְגַּרְדִּי אוּלָם בְּמַחֲצִית הַשְּׁנִיָּה שֶׁל שְׁנוֹת הַ-20, כְּשֶׁנִּדְרְשׁוּ הָאַדְרִיכָלִים לִבְנוֹת בִּנְיְנֵי מְגוּרִים, בִּנְיְנֵי מִשְׂרָדִים וּמִפְעָלִים, קִבֵּל הַסִּגְנוֹן אֹפִי מַעֲשִׂי יוֹתֵר וְחֵלֶק קַוִּים מְשֻׁתָּפִים עִם הַסִּגְנוֹן הַבֵּינְלְאֻמִּי.[1]
בִּסְבִיבוֹת 1932 יָרְדָה קַרְנוֹ שֶׁל הַסִּגְנוֹן וְסִגְנוֹן הַקְּלָסִיצִיזְם הַסְּטָלִינִיסְטִי הָפַךְ לְשׁוֹלֵט בַּבְּנִיָּה בִּבְרִית הַמּוֹעָצוֹת. בְּבִנְיְנֵי מְגוּרִים שֶׁל הָאַדְרִיכָלוּת הַקּוֹנְסְטְרוּקְטִיבִיסְטִית, דֻּגְמַת הנרקומפין (בְּמוֹסְקְבָה) נַעֲשֶׂה נִסָּיוֹן לְעַצֵּב אֶת חַיֵּי הַדַּיָּרִים בְּעֶזְרַת תִּכְנוּן הַבִּנְיָן: הָיָה בּוֹ שִׁתּוּף בְּשֵׁרוּתִים כְּגוֹן: מִטְבָּח, כְּבִיסָה, שִׁטְחֵי הִתְכַּנְּסוּת וְחִנּוּךְ מְשֻׁתָּף שֶׁל הַיְּלָדִים וְהוּשַׂם דָּגֵשׁ עַל שִׁוְיוֹן בֵּין הַמִּינִים. כָּל אֵלּוּ נִזְנְחוּ בִּשְׁנוֹת הַ-30 וְהֻחְלְפוּ בִּתְחִיַּת עֶרְכֵי הַמִּשְׁפָּחָה שֶׁהָיְתָה חֵלֶק מֵהַמְּשַׁנָּה הַסְּטָלִינִיסְטִית.
תִּכְנוּן עָרִים בָּעִדָּן הַקּוֹנְסְטְרוּקְטִיבִיסְטִי
לַמְרוֹת שְׁאַפְתָּנוּתָן שֶׁל רַבּוֹת מֵהַצָּעוֹת הָאַדְרִיכָלִים הַקּוֹנְסְטְרוּקְטִיבִיסְטִיִּים לְתִכְנוּן שְׁכוּנוֹת וְעָרִים מִן הַיְּסוֹד, מְעַטּוֹת הֵן הַדֻּגְמָאוֹת לְתִכְנוּן עָרִים עִקְבִי וּמְגֻבָּשׁ שֶׁנּוֹצַר עַל יָדָם. רֹבַע נרווסקיה זסטווה שֶׁבְּלֶנִינְגְּרָד הָיָה אֶחָד מֵהַמּוֹקְדִים שֶׁל הָאַדְרִיכָלוּת הַקּוֹנְסְטְרוּקְטִיבִיסְטִית. הֵחֵל מְ-1925 נִבְנוּ בְּלֶנִינְגְּרָד בָּתֵּי מְגוּרִים לַפּוֹעֲלִים וְגַם מִבְנֵי צִבּוּר. רַבִּים מֵהַקּוֹנְסְטְרוּקְטִיבִיסְטִיִּים קִוּוּ לְכָךְ שֶׁבְּתָכְנִית הַחֹמֶשׁ הָרִאשׁוֹנָה (1928) וּבַ"מַּהְפֵּכָה הַתַּרְבּוּתִית" שֶׁנִּלְוְתָה אֵלֶיהָ, תִּנָּתֵן לָהֶם אֶפְשָׁרוּת לְבַטֵּא אֶת רַעְיוֹנוֹתֵיהֶם הָאַדְרִיכָלִיִּים, אַךְ בְּאוֹתָהּ עֵת נֶחְלָק הַסִּגְנוֹן בֵּין חֲסִידֵי הַתִּכְנוּן הָאוּרְבָּנִי וּבֵין אֵלּוּ שֶׁנֶּטוֹ אַחֲרֵי הַתִּכְנוּן הַלֹּא אוּרְבָּנִי, שֶׁהָרַעְיוֹנוֹת הַבּוֹלְטִים בּוֹ הָיוּ עִיר הַגַּנִּים וְהָעִיר הַלִּינֵאָרִית. נוֹסָף לְכָךְ, בְּרֹב הַמִּקְרִים הָעָדְפוּ תָּכְנִיּוֹת פְּרַגְמָטִיּוֹת יוֹתֵר, כְּמוֹ לַמָּשָׁל, בִּבְנִיַּת הָעִיר הַחֲדָשָׁה מגניטוגורסק שֶׁבְּהָרֵי אוּרָל בָּהּ הָעָדְפוּ תָּכְנִיּוֹתֵיהֶם שֶׁל קְבוּצַת אַדְרִיכָלִים גֶּרְמָנִים, בְּרָאשׁוֹת ארנסט מַאי.
