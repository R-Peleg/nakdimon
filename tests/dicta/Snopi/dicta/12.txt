אֶלְדָּד פלד
אֶלְדָּד פלד הוּא קָצִין צַהַ"ל בְּמִילּוּאִים בְּדַרְגַּת אַלּוּף מִשְׁנֶה.
שֵׁירוּת צְבָאִי
עִם גיוסו לַצַּהַ"ל שׁוּבַּץ פלד בַּחֲטִיבַת גבעתי וְעָבָר מַסְלוּל הַכְשָׁרָה כְּלוֹחֵם, קוּרְס מַ"כִּים חי"ר וְקוֹרֵס קְצִינֵי חי"ר. עִם סִיּוּם הַקּוּרְס שִׁימֵּשׁ כִּמְפַקֵּד מַחְלָקָה וּפְלוּגָּה בַּחֲטִיבָה. בַּהֶמְשֵׁךְ שִׁימֵּשׁ כִּמְפַקֵּד פלחה"ן גבעתי וכמפקד סִיַּירְתָּ גבעתי. בַּשָּׁנִים 1999-2001 שִׁימֵּשׁ כִּמְפַקֵּד גְּדוּד שָׁקַד וְלֶקַח חָלָק בַּפְּעִילוּת המבצעית כְּנֶגֶד הַטֵּרוֹר הפלסטיני בִּרְצוּעַת עַזָּה וּבַלְּחִימָה בַּדָּרוֹם לְבָנוֹן. לְאַחַר מִכֶּן שִׁימֵּשׁ כִּמְפַקֵּד בא"ח גבעתי[1] וכסגן מְפַקֵּד הַחֲטִיבָה[2]. בַּשָּׁנִים 2006-2009 שִׁימֵּשׁ כִּמְפַקֵּד מַעֲרָךְ מִגַּל[3]. לְאַחֵר שִׁחְרוּרוֹ פָּנָה לַעֲבוֹד כְּעוֹרֵךְ דִּין[4].
