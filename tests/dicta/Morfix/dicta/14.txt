אָלֶכְּס היגינס
אָלֶכְּסַנְדֶּר גורדון "אָלֶכְּס" היגינס (18 בְּמֶרְץ 1949 - 24 בְּיוּלִי 2010), יָדוּעַ גַּם בְּכִינּוּי הוּרִיקַן היגנס, הָיָה שַׂחְקַן סְנוּקֶר מִקְצוֹעָנִי צָפוֹן אִירִי. היגינס זָכָה פַּעֲמַיִים בְּאַלִּיפוּת הָעוֹלָם בַּסְּנוּקֶר בְּשָׁנִים 1972 וְ-1982, וְהָיָה פַּעֲמַיִים סְגַן אַלּוּף הָעוֹלָם בַּסְּנוּקֶר בְּשָׁנִים 1976 וְ-1980. בְּנוֹסָף, היגינס זָכָה בְּאַלִּיפוּת הָעוֹלָם בַּסְּנוּקֶר לַזּוּגוֹת, יַחַד עִם ג'ימי וייט הָאַנְגְּלִי בִּשְׁנַת 1984, וְכֵן זָכָה שָׁלוֹשׁ פְּעָמִים בִּרְצִיפוּת בִּגְבִיעַ הָעוֹלָם בַּסְּנוּקֶר לַנִּבְחָרוֹת, עִם נִבְחֶרֶת אִירְלַנְד בְּשָׁנִים 1985, 1986 וְ-1987. בְּשִׂיאוֹ דּוֹרַג היגינס בַּמָּקוֹם הַשֵּׁנִי בָּעוֹלָם בְּדֵירוּג שַׂחְקָנֵי הַסְּנוּקֶר הָעוֹלָמִי. זָכָה לַכִּינּוּי "הוּרִיקָן" בִּזְכוּת סִגְנוֹן מִשְׂחָקוֹ הַמָּהִיר.
קַרְיֶירָה
היגינס הֵחֵל לְשַׂחֵק סְנוּקֶר מִקְצוֹעָנִי בְּעוֹנַת 1971/1972, בְּגִיל 22. הוּא זָכָה בְּאַלִּיפוּת הָעוֹלָם בִּסְנוּקֶר כְּבָר בְּנִיסְיוֹנוֹ הָרִאשׁוֹן בִּשְׁנַת 1972, בְּגִיל 22, לְאַחַר שֶׁנִּיצֵּחַ בַּגְּמָר אֶת ג'ון ספנסר הָאַנְגְּלִי בְּתוֹצָאָה 32-37, וּבְכָךְ נִהְיָה לְאַלּוּף הָעוֹלָם בַּסְּנוּקֶר הַצָּעִיר בְּיוֹתֵר אֵי פַּעַם (שִׂיא שֶׁנִּשְׁבַּר כַּעֲבוֹר 18 שָׁנִים עַל יְדֵי סטיבן הנדרי כְּשֶׁזָּכָה בְּאַלִּיפוּת הָעוֹלָם בְּגִיל 21). לְאַחַר מִכֵּן הִגִּיעַ היגינס לִגְמַר אַלִּיפוּת הָעוֹלָם בְּשָׁנִים 1976 וְ-1980, אַךְ הִפְסִיד. בִּשְׁנַת 1982, זָכָה היגינס בַּפַּעַם הַשְּׁנִיָּיה בְּאַלִּיפוּת הָעוֹלָם, לְאַחַר שֶׁנִּיצֵּחַ בַּגְּמָר אֶת ריי רירדון הַוֶּלְשִׁי בְּתוֹצָאָה 15-18, וְאִם לֹא הָיָה נִקְנָס בְּהוֹרָדַת נְקוּדּוֹת דֵּירוּג עֵקֶב עֲבֵירוֹת מִשְׁמַעַת, הָיָה מְדוֹרָג בְּעוֹנַת 1982/1983 בַּמָּקוֹם הָרִאשׁוֹן בָּעוֹלָם.
היגינס, עִם סִגְנוֹן מִשְׂחָקוֹ, שֶׁהָיָה מַלְהִיב, הָיָה אַחְרַאי בְּמִידָּה רַבָּה לָעֲלִיָּיה בְּפּוֹפּוּלָרִיּוּת שֶׁל עֲנַף הַסְּנוּקֶר בְּסוֹף שְׁנוֹת הַשִּׁבְעִים. הוּא הָיָה שַׂחְקָן מָהִיר מְאוֹד, וּבָרַאֲיוֹנוֹת הוּא אַף הִסְבִּיר שֶׁהַסִּיבָּה לְכָךְ הָיְיתָה פָּשׁוּט חוֹסֶר סַבְלָנוּת. רַבִּים מַחְשִׁיבִים אוֹתוֹ לְאֶחָד מִשַּׂחְקָנֵי הַסְּנוּקֶר הַטּוֹבִים בְּיוֹתֵר בְּכָל הַזְּמַנִּים.
עִם זֹאת, נוֹדַע גַּם בָּאוֹפִי וּמָזַג בְּעָיָיתִיִּים. עַל פִּי הַעֲרָכוֹת, כְּ-50 תְּלוּנוֹת רִשְׁמִיּוֹת הוּגְּשׁוּ נֶגֶד אָלֶכְּס היגינס עַל יְדֵי הִתְאַחֲדוּת הַסְּנוּקֶר הָעוֹלָמִית - יוֹתֵר מִכֹּל שַׂחְקַן סְנוּקֶר אַחֵר בַּהִיסְטוֹרְיָה שֶׁל הַמִּשְׂחָק.[1]
בְּמַהֲלַךְ הַקַּרְיֶירָה, זָכָה היגינס בְּ-30 תְּאָרִים מִקְצוֹעָנִיִּים (26 תּוֹאֳרִי יְחִידִים וְ-4 תְּאָרִים קְבוּצָתִיִּים), בָּהֶם אַלִּיפוּת בְּרִיטַנְיָה בִּשְׁנַת 1983, טוּרְנִיר המאסטרס בְּשָׁנִים 1978 וְ-1981, והמאסטרס הָאִירִי בִּשְׁנַת 1989, בּוֹ נִיצֵּחַ בְּגִיל 40 אֶת סטיבן הנדרי הַצָּעִיר.
