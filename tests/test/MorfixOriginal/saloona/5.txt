הַסִּפּוּר הַזֶּה תָּפַס אוֹתִי לַחֲלוּטִין לֹא מוּכָנָה וּבִנְקֻדַּת זְמַן מְאוֹד רְגִישָׁה.
יֵשׁ לָנוּ שְׁתֵּי בָּנוֹת, וּבְאוֹתָהּ תְּקוּפָה חָשַׁבְנוּ עַל יֶלֶד נוֹסָף וְהֵנָּה פִּתְאוֹם אֲנִי כְּבָר לֹא "אִילָנֵי מָגֵן" אֶלָּא: "אִילָנֵי מָגֵן סִכּוּן גָּבוֹהַּ".
וּמִי בִּכְלָל מוּכָן לְדַבֵּר אִתְּךָ עַל יְלָדִים.
הִתְחִילָה שִׁגְרָה שֶׁל בְּדִיקוֹת וּמַעֲקָב: כָּל 3 חֳדָשִׁים בְּדִיקוֹת אֵצֶל כִירוּרְג, בְּדִיקוֹת MRI, אוּלְטְרָסָאוּנְד, וְעִם כָּל זֶה, תְּקוּפָה אֲרֻכָּה עֲדַיִן חָיִיתִי בַּהַדְחָקָה וּבַהַכְחָשָׁה כְּשֶׁהַכֹּל מְרַחֵף בָּאֲוִיר, לֹא נוֹגֵעַ בִּי.
עִם הַזְּמַן נֶחְשַׂפְתִּי לַמֵּידָע.
שָׁמַעְתִּי שֶׁסַּרְטַן הַשָּׁד נִתֵּן לָרִפּוּי בְּגִלּוּי מֻקְדָּם, שֶׁיֵּשׁ לוֹ טִפּוּל.
לְגַבֵּי סַרְטַן הַשְׁחָלוֹת אֵין גִּלּוּי מֻקְדָּם, וּמֵרֶגַע הַגִּלּוּי יֵשׁ מַמָּשׁ זְמַן קָצָר עַד שֶׁ"מִזְדַּכִּים עַל הַחַיִּים".
הֶחְלַטְתִּי לַעֲבֹר נִתּוּחַ כְּרִיתַת שְׁחָלוֹת וַחֲצוֹצְרוֹת, וּבְדַקָּה הַ-90 קִבַּלְתִּי רַגְלַיִם קָרוֹת וּבָרַחְתִּי..
נְקֻדַּת הַמִּפְנֶה מִבְּחִינָתִי הָיְתָה סַדְנָא בָּהּ הִשְׁתַּתַּפְתִּי בְּהַמְלָצַת פְּרוֹפֶסוֹר פְרִידְמַן, מְנַהֵל הַיְּחִידָה האונקוגנטית בְּתֵל הַשּׁוֹמֵר וד"ר שְׁחֹר לֵב אָרִי, מִבֵּית הַחוֹלִים איכילוב שֶׁהֵקִימוּ קְבוּצַת תְּמִיכָה לַנַּשָּׂאִיּוֹת.
הַסַּדְנָא פָּעֲלָה לְפִי עֶקְרוֹנוֹת "הָעֲבוֹדָה" שֶׁל ביירון קייטי.
נִפְגַּשְׁנוּ בְּמֶשֶׁךְ שְׁלוֹשָׁה חֳדָשִׁים, בְּכָל יוֹם שִׁשִּׁי לְמִפְגָּשִׁים בְּנֵי חָמֵשׁ שָׁעוֹת.
עֶצֶם הַמִּפְגָּשׁ עִם נַשָּׂאִיּוֹת כָּמוֹנִי, שֶׁרֻבָּן כְּבָר עָבְרוּ נִתּוּחִים מניעתיים בֵּין שֶׁל כְּרִיתַת שְׁחָלוֹת וַחֲצוֹצְרוֹת וּבֵין כְּרִיתַת שָׁדַיִם הָיָה חָשׁוּב וּמַרְגִּיעַ - נוֹכַחְתִּי שֶׁיֵּשׁ חַיִּים אַחֲרֵי הַכְּרִיתָה.
מַטְּרַת הַסַּדְנָה הַמֻּצְהֶרֶת הָיְתָה לְנַטְרֵל פְּחָדִים וַחֲרָדוֹת, שֶׁהֵם כְּשֶׁלְּעַצְמָם יְכוֹלִים לְזָרֵז הוֹפָעַת הַמַּחֲלָה.
יָצָאתִי מֵהַסַּדְנָה מְחֻזֶּקֶת וְהַפַּעַם נִגַּשְׁתִּי בָּאֹמֶץ לְנִתּוּחַ כְּרִיתַת שְׁחָלוֹת וַחֲצוֹצְרוֹת.
לֹא אוֹמֵר שֶׁזֶּה הָיָה קַל, אֲבָל תּוֹדָה לְאֵל הַיּוֹם זֶה מֵאֲחוֹרַי.
הַנִּתּוּחַ מְלַוֶּה בְּתוֹפְעוֹת לְוַאי שֶׁל גִּיל הַמַּעֲבָר, שֶׁתּוֹפְסוֹת אוֹתְךָ בְּבַת אַחַת - גַּלֵּי חֹם מְטֹרָפִים, וְעִמָּן הַפָּן הָרִגְשִׁי: הַפְסָקַת הַמַּחְזוֹר הַחָדְשִׁי, וְהַהַשְׁלָמָה עִם כָּךְ שֶׁהִפְסַקְתִּי לִהְיוֹת פּוֹרִיָּה בְּגִיל כָּל כָּךְ צָעִיר לֹא הָיְתָה קַלָּה.
וּבְכָל זֹאת, הַיּוֹם אֲנִי בְּמָקוֹם אַחֵר לְגַמְרֵי.
הֶחְלַטְתִּי שֶׁאֵין טַעַם לְהִתְוַכֵּחַ עִם הַמְּצִיאוּת.
הִיא פֹּה, וְצָרִיךְ לְהִסְתַּכֵּל לָהּ בָּעֵינַיִם.
