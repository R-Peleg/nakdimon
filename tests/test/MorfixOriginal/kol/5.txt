שִׁחְרוּר עַל תְּנַאי מִמַּאֲסַר
תִּרְגּוּם: الصفحة بالعربية
הַקְדָּמָה:
אָסִיר עָשׂוּי לִהְיוֹת מְשֻׁחְרָר לִפְנֵי שֶׁהִשְׁלִים אֶת מְלוֹא תְּקוּפַת הַמַּאֲסָר שֶׁנִּגְזְרָה עָלָיו, בִּתְנַאי שֶׁלֹּא יַעֲבֹר עֲבֵרָה נוֹסֶפֶת לְאַחַר שִׁחְרוּרוֹ וּבִתְנָאִים נוֹסָפִים שֶׁיִּקָּבְעוּ
הַחְלָטָה עַל שִׁחְרוּר עַל תְּנַאי תִּנָּתֵן עַל סְמַךְ שִׁקּוּלִים הַקְּשׁוּרִים, בֵּין הַיֶּתֶר, לְהִתְנַהֲגוּת הָאָסִיר וְכֵן לַשְּׁאֵלָה אִם שִׁלֵּם בַּמֶּרְכָּז לִגְבִיַּת קְנָסוֹת אֶת הַקְּנָס אוֹ הַפִּצּוּי לַקָּרְבָּן שֶׁהוּטַל עָלָיו (אִם הוּטַל) בְּמִסְגֶּרֶת גְּזַר הַדִּין
וְעֵדַת הַשִּׁחְרוּרִים רַשָּׁאִית לְשַׁנּוֹת אֶת תְּנָאֵי הַשִּׁחְרוּר שֶׁל אָסִיר בְּכָל עֵת וְגַם לְבַטֵּל אֶת הַהַחְלָטָה עַל הַשִּׁחְרוּר
אֲסִיר יָכוֹל לְבַקֵּשׁ דִּיּוּן חוֹזֵר בְּנוֹשֵׂא שִׁחְרוּר עַל תְּנַאי וְגַם לַעֲתֹר לְבֵית הַמִּשְׁפָּט כְּנֶגֶד הַחְלָטַת וַעֲדַת הַשִּׁחְרוּרִים
לְמֵידָע נוֹסָף רָאוּ סְעִיפִים 9-2 בְּחֹק שִׁחְרוּר עַל-תְּנַאי מִמַּאֲסָר, תשס"א-2001
שִׁחְרוּר עַל תְּנַאי הוּא שִׁחְרוּר מִמַּאֲסָר לִפְנֵי שֶׁהָאָסִיר הִשְׁלִים אֶת מְלוֹא תְּקוּפַת הַמַּאֲסָר שֶׁנִּגְזְרָה עָלָיו.
הַתְּנַאי שֶׁבּוֹ מֻתְנֶה הַשִּׁחְרוּר הוּא שֶׁהָאָסִיר הַמְּשֻׁחְרָר לֹא יַעֲבֹר עֲבֵרָה נוֹסֶפֶת, שֶׁהָעֹנֶשׁ עָלֶיהָ הוּא לְמַעְלָה מִשְּׁלוֹשָׁה חָדְשֵׁי מַאֲסָר, בְּמֶשֶׁךְ תְּקוּפַת הַתְּנַאי (הַתְּקוּפָה שֶׁמֵּרֶגַע שִׁחְרוּרוֹ וְעַד לְסִיּוּם תְּקוּפַת הַמַּאֲסָר שֶׁהָיָה נוֹשֵׂא לְפִי גְּזַר הַדִּין אִלּוּלֵא שֻׁחְרַר).
בְּמִקְרֶה שֶׁמְּדֻבָּר בַּשִּׁחְרוּר עַל תְּנַאי כְּתוֹצָאָה מֵהֲקָלָה בָּעֹנֶשׁ עֵקֶב בַּקָּשָׁה שֶׁל שַׂר הַמִּשְׁפָּטִים אוֹ נְשִׂיא הַמְּדִינָה אוֹ כְּתוֹצָאָה מִקְּצִיבַת עֹנֶשׁ שֶׁל אֲסִיר עוֹלָם בִּידֵי הַנָּשִׂיא תְּחֻשַּׁב תְּקוּפַת הַתְּנַאי בְּאֹפֶן אַחֵר.
וַעֲדַת הַשִּׁחְרוּרִים רַשָּׁאִית לְהוֹסִיף תְּנָאִים נוֹסָפִים.
אָסִיר שֶׁשֻּׁחְרַר וְהֵפֵר אֶת תְּנָאֵי הַשִּׁחְרוּר, יֻחְזַר לְרִצּוּי הֶמְשֵׁךְ תְּקוּפַת הַמַּאֲסָר הַמְּקוֹרִית.
שִׁחְרוּר עַל תְּנַאי אֵינוֹ נִתָּן בְּאֹפֶן אוֹטוֹמָטִי. וַעֲדַת הַשִּׁחְרוּרִים מַחְלִיטָה עָלָיו עַל סְמַךְ שִׁקּוּלִים הַקְּשׁוּרִים בֵּין הַיֶּתֶר לְחֻמְרַת הָעֲבֵרָה, הַרְשָׁעוֹת קוֹדְמוֹת, הִתְנַהֲגוּת הָאָסִיר, חַוַּת דַּעַת שֶׁל הָרָשׁוּת לְשִׁקּוּם הָאָסִיר, וְכֵן לַשְּׁאֵלָה אִם שִׁלֵּם הָאָסִיר אֶת הַקְּנָס אוֹ הַפִּצּוּי לַקָּרְבָּן שֶׁהוּטַל עָלָיו בְּמִסְגֶּרֶת גְּזַר הַדִּין (אִם הוּטַל עָלָיו).
נִתָּן לְשַׁלֵּם אֶת הַקְּנָס אוֹ אֶת הַפִּצּוּי לַקָּרְבָּן הֶעֱבִירָה, שֶׁהוּטְלוּ עַל הָאָסִיר בְּמִסְגֶּרֶת גְּזַר הַדִּין, בַּמֶּרְכָּז לִגְבִיַּת קְנָסוֹת, וְכָךְ לְהַגְדִּיל אֶת הַסִּכּוּי שֶׁוַּעֲדַת הַשִּׁחְרוּרִים תַּחְלִיט עַל שִׁחְרוּרוֹ עַל תְּנַאי שֶׁל הָאָסִיר. לַמֵּידָע עַל דַּרְכֵי תַּשְׁלוּם רָאוּ תַּשְׁלוּם חוֹב בַּמֶּרְכָּז לִגְבִיַּת קְנָסוֹת, אֲגָרוֹת וְהוֹצָאוֹת.
אֻכְלוּסִיַּת יַעַד וּתְנָאִים מַקְדִּימִים
אֲסִיר הַמְּרַצֶּה עֹנֶשׁ מַאֲסָר שֶׁבֵּין שְׁלוֹשָׁה חֳדָשִׁים לְשִׁשָּׁה חֳדָשִׁים -
נְצִיב שב"ס רַשַּׁאי לְשַׁחְרֵר עַל תְּנַאי אָסִיר הַנּוֹשֵׂא עֹנֶשׁ מַאֲסָר לַתְּקוּפָה הָעוֹלֶה עַל שְׁלוֹשָׁה חֳדָשִׁים וְאֵינָהּ עוֹלֶה עַל שִׁשָּׁה חֳדָשִׁים, הָעוֹנָה עַל שְׁנֵי הַתְּנָאִים הַבָּאִים:
הָאָסִיר נָשָׂא לְפָחוֹת שְׁנֵי שְׁלִישִׁים מִתְּקוּפַת הַמַּאֲסָר שֶׁעָלָיו לָשֵׂאת.
נְצִיב שב"ס שֻׁכְנַע כִּי הָאָסִיר רָאוּי לַשִּׁחְרוּר וְכִי שִׁחְרוּרוֹ אֵינוֹ מְסַכֵּן אֶת שְׁלוֹם הַצִּבּוּר.
אֲסִיר הַמְּרַצֶּה עֹנֶשׁ מַאֲסָר שֶׁל לְמַעְלָה מִשִּׁשָּׁה חֳדָשִׁים -
וַעֲדַת שִׁחְרוּרִים בְּעִנְיַן אֲסִירִים רַשָּׁאִית לְשַׁחְרֵר עַל תְּנַאי אָסִיר הַנּוֹשֵׂא עֹנֶשׁ מַאֲסָר לַתְּקוּפָה הָעוֹלֶה עַל שִׁשָּׁה חֳדָשִׁים, הָעוֹנָה עַל שְׁלֹשֶׁת הַתְּנָאִים הַבָּאִים:
הָאָסִיר אֵינוֹ אֲסִיר עוֹלָם.
הָאָסִיר נָשָׂא לְפָחוֹת שְׁנֵי שְׁלִישִׁים מִתְּקוּפַת הַמַּאֲסָר שֶׁעָלָיו לָשֵׂאת.
וַעֲדַת הַשִּׁחְרוּרִים שֻׁכְנְעָה כִּי הָאָסִיר רָאוּי לַשִּׁחְרוּר וְכִי שִׁחְרוּרוֹ אֵינוֹ מְסַכֵּן אֶת שְׁלוֹם הַצִּבּוּר.
אָסִיר שֶׁלֹּא הִשְׁלִים שְׁנֵי שְׁלִישִׁים מִתְּקוּפַת מַאֲסָרוֹ -
וַעֲדַת שִׁחְרוּרִים מְיֻחֶדֶת רַשָּׁאִית לְשַׁחְרֵר עַל תְּנַאי אָסִיר שֶׁעוֹד לֹא הִשְׁלִים שְׁנֵי שְׁלִישִׁים מִמַּאֲסָרוֹ אִם הוּא עוֹנָה עַל שְׁלֹשֶׁת הַתְּנָאִים הַבָּאִים:
הָאָסִיר אֵינוֹ אֲסִיר עוֹלָם.
הָאָסִיר שׁוֹהֶה בְּמַאֲסָר 25 שָׁנִים לְפָחוֹת.
וַעֲדַת הַשִּׁחְרוּרִים הַמְּיֻחֶדֶת שֻׁכְנְעָה כִּי הָאָסִיר רָאוּי לַשִּׁחְרוּר וְכִי שִׁחְרוּרוֹ אֵינוֹ מְסַכֵּן אֶת שְׁלוֹם הַצִּבּוּר.
אֲסִיר עוֹלָם -
וַעֲדַת שִׁחְרוּרִים מְיֻחֶדֶת רַשָּׁאִית לְשַׁחְרֵר אֲסִיר עוֹלָם אִם הוּא עוֹנָה עַל שְׁלֹשֶׁת הַתְּנָאִים הַבָּאִים:
עָנְשׁוֹ שֶׁל הָאָסִיר נִקְצַב.
הָאָסִיר נָשָׂא לְפָחוֹת שְׁנֵי שְׁלִישִׁים מֵהַתְּקוּפָה שֶׁנִּקְצְבָה (כּוֹלֵל הַתְּקוּפָה שֶׁבָּהּ שָׁהָה בַּמַּעֲצָר בְּעִקְבוֹת הָעֲבֵרוֹת שֶׁבְּגִינָן הוּא נוֹשֵׂא אֶת עֹנֶשׁ הַמַּאֲסָר).
וַעֲדַת הַשִּׁחְרוּרִים הַמְּיֻחֶדֶת שֻׁכְנְעָה כִּי הָאָסִיר רָאוּי לַשִּׁחְרוּר וְכִי שִׁחְרוּרוֹ אֵינוֹ מְסַכֵּן אֶת שְׁלוֹם הַצִּבּוּר.
שְׁלַבֵּי הַהֲלִיךְ:
עִנְיָנוֹ שֶׁל הָאָסִיר יוּבָא בִּפְנֵי וַעֲדַת הַשִּׁחְרוּרִים, וְהִיא תָּדוּן בְּעִנְיָנוֹ וְתִתֵּן הַחְלָטָתָהּ.
הַחְלָטַת הַוַּעֲדָה תִּהְיֶה מְנֻמֶּקֶת וְתֵחָתֵם בִּידֵי חַבְרֵי הַוַּעֲדָה.
הֶעְתֵּק הַהַחְלָטָה יִמָּסֵר לָאָסִיר, לְבָא כּוֹחַ הַיּוֹעֵץ הַמִּשְׁפָּטִי לַמֶּמְשָׁלָה, וְאִם הֻתְנָה הַשִּׁחְרוּר עַל-תְּנַאי בְּהִשְׁתַּתְּפוּת הָאָסִיר בְּתָכְנִית שִׁקּוּמִית, יָעֳבַר הֶעְתֵּק הַהַחְלָטָה גַּם לְשֵׁרוּת הַמִּבְחָן אוֹ לָרָשׁוּת לְשִׁקּוּם הָאָסִיר, לְפִי הָעִנְיָן.
וַעֲדָה שֶׁהֶחְלִיטָה לְשַׁחְרֵר אָסִיר עַל-תְּנַאי, תִּתֵּן לַאֲסִיר רִשָּׁיוֹן וּבוֹ יְפֹרְטוּ תְּנָאֵי הַשִּׁחְרוּר. אָסִיר שֶׁשֻּׁחְרַר עַל-תְּנַאי יִשָּׂא עִמּוֹ אֶת רִשְׁיוֹנוֹ בְּכָל תְּקוּפַת הַתְּנַאי, וְיַצִּיגוּ בִּפְנֵי שׁוֹטֵר, עַל פִּי דְּרִישָׁתוֹ.
נוֹכְחוּת וּזְכוּת טִעוּן בְּדִיּוּנֵי הַוַּעֲדָה
דִּיּוּן בַּוַּעֲדָה, (מִלְּבַד דִּיּוּן בְּעִנְיָנוֹ שֶׁל אֲסִיר עוֹלָם שֶׁעָנְשׁוֹ נִקְצַב אוֹ אָסִיר שֶׁעָנְשׁוֹ הוּקַל ע"י נְשִׂיא הַמְּדִינָה אוֹ דִּיּוּן חוֹזֵר בְּעִנְיָנוֹ שֶׁל אָסִיר כָּלְשֶׁהוּ) יִתְקַיֵּם בְּנוֹכְחוּת הָאָסִיר וְעוֹרֵךְ דִּינוֹ (אִם הוּא מְיֻצָּג), וְכֵן בְּנוֹכְחוּת נְצִיג הַיּוֹעֵץ הַמִּשְׁפָּטִי לַמֶּמְשָׁלָה. אֵם הִתְקַבְּלָה הַסְכָּמַת נְצִיג הַיּוֹעֵץ הַמִּשְׁפָּטִי לַמֶּמְשָׁלָה לְשִׁחְרוּרוֹ הַמֻּקְדָּם שֶׁל הָאָסִיר אוֹ לְשִׁנּוּי תְּנַאי שִׁחְרוּרוֹ שֶׁל אָסִיר עַל פִּי בַּקָּשָׁתוֹ, נִתָּן לְקַיֵּם אֶת הַדִּיּוּן בַּוַּעֲדָה אַף בְּלֹא נוֹכְחוּת הָאָסִיר אוֹ עוֹרֵךְ דִּינוֹ.
דִּיּוּן בַּוַּעֲדָה בְּבַקָּשָׁה לְקַיֵּם דִּיּוּן חוֹזֵר בְּעִנְיָנוֹ שֶׁל אָסִיר יִתְקַיֵּם שֶׁלֹּא בְּנוֹכְחוּת הַצְּדָדִים.
הַוַּעֲדָה תַּחְלִיט עַל שִׁחְרוּרוֹ עַל-תְּנַאי שֶׁל אָסִיר, כּוֹלֵל עַל שִׁחְרוּרוֹ מִטְּעָמִים רְפוּאִיִּים אוֹ עַל שִׁנּוּי תְּנַאי שִׁחְרוּרוֹ שֶׁל אָסִיר שֶׁשֻּׁחְרַר עַל-תְּנַאי, רַק לְאַחַר שֶׁנִּתְּנָה לְבָא כּוֹחַ הַיּוֹעֵץ הַמִּשְׁפָּטִי לַמֶּמְשָׁלָה הִזְדַּמְּנוּת לִטְעֹן אֶת טַעֲנוֹתָיו.
