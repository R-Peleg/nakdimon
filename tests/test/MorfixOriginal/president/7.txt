לַשֻּׁתָּפִים הָרַבִּים כָּל כָּךְ, הַנְּכוֹנִים לְהַיְשִׁיר מַבָּט אַמִּיץ לַשִּׁנּוּיִים הָאַדִּירִים שֶׁעוֹבֶרֶת הַחֶבְרָה הַיִּשְׂרְאֵלִית, לְאֵלֶּה הָרוֹצִים בֶּחָזוֹן שֶׁל תִּקְוָה יִשְׂרְאֵלִית לְכֻלָּנוּ, מִקֶּרֶב כָּל הַמִּגְזָרִים, הַמַּחֲנוֹת וְהַקְּהִלּוֹת, אֲנִי מְבַקֵּשׁ לוֹמַר: הַחֶבְרָה הַיִּשְׂרְאֵלִית נִמְצֵאת הַיּוֹם בְּצֹמֶת דְּרָכִים.
אָנוּ רוֹאִים אֶת הַשִּׁנּוּיִים לַנַּגָּד עִנִּינוּ, וְזוֹהִי הַשָּׁעָה לְהַכְרִיעַ לְאָן פָּנֵינוּ מוּעָדוֹת.
כּוֹתֶרֶת הַכֶּנֶס הַזֶּה הִיא "תִּקְוָה יִשְׂרְאֵלִית מְשֻׁתֶּפֶת- חָזוֹן אוֹ חֲלוֹם".
אֲבָל בֶּאֱמֶת, כּוֹתֶרֶת הַכֶּנֶס הַזֶּה הָיְתָה צְרִיכָה לִהְיוֹת: "תִּקְוָה יִשְׂרְאֵלִית: לִהְיוֹת אוֹ לַחְדֹּל".
מַסְפִּיק מַבָּט קָצָר בָּעִיר יְרוּשָׁלַיִם, הָעִיר הָעֲרָבִית הֲכִי גְּדוֹלָה בָּאָרֶץ, הָעִיר הַחֲרֵדִית הֲכִי גְּדוֹלָה בָּאָרֶץ, הָעִיר הַדָּתִית הֲכִי גְּדוֹלָה בָּאָרֶץ וְהָעִיר הַחִלּוֹנִית הַשְּׁנִיָּה בְּגָדְלָהּ בָּאָרֶץ.
הָעִיר הָאֲהוּבָה אֲבָל גַּם הָעֲנִיָּה, הַקְּרוּעָה וְהַמְּתוּחָה כָּל כָּךְ- כְּדֵי לְהָבִין שֶׁכּוֹחוֹת הַשּׁוּק לְבַדָּם, וְלַחֲלוּפִין הָרָצוֹן הַטּוֹב לְבַדּוֹ שֶׁל אִרְגּוּנִים וִיחִידִים, מְסוּרִים כְּכָל שֶׁיִּהְיוּ, לֹא יוּכְלוּ לְהִתְמוֹדֵד עִם גֹּדֶל הָאֶתְגָּר.
לִפְנֵי שָׁנָה הָיוּ כָּאֵלֶּה שֶׁהֵבִינוּ אֶת דְּבָרַי כְּעוֹד קְרִיאָה מֵהַסּוּג הַחֲגִיגִי הָאָפְיָנִי לַנָּשִׂיא- קְרִיאָה לְ"אַחְדוּת" נוֹטֶפֶת דְּבַשׁ, לְ"מַעְגְּלֵי שִׂיחַ" אוֹ לִכְבוֹד הַדֶּדִי.
אֲבָל, בָּרֹאשׁ וּבָרִאשׁוֹנָה, דְּבָרַי נוֹעֲדוּ לְשֶׁמֶשׁ קְרִיאַת הַשְׁכָּמָה לַפַּעַר וּלְאִי הַהֲלִימָה שֶׁבֵּין הַמְּצִיאוּת הַחֶבְרָתִית הַיִּשְׂרְאֵלִית לְבֵין הַמַּעֲרֶכֶת הַמּוֹסָדִית הַיִּשְׂרְאֵלִית.
הַחֶבְרָה הַיִּשְׂרְאֵלִית אוּלַי מְשַׁנָּה אֶת פָּנֶיהָ, אֲבָל לֹא אֶת הַתּוֹדָעָה שֶׁלָּהּ.
אֲנִי חוֹשֵׁשׁ, שֶׁהַמַּעֲרָכוֹת הַיְּסוֹדִיּוֹת הַמְּאַרְגְּנוֹת אֶת חַיֵּינוּ- הַפּוֹלִיטִיקָה, הַתִּקְשֹׁרֶת, הָאָקָדֶמְיָה, הַמֶּשֶׁק, הַחִנּוּךְ וְעוֹד, טֶרֶם הִפְנִימוּ אֶת הַשִּׁנּוּיִים הַלָּלוּ לְעָמְקָם, טֶרֶם הִתְאִימוּ אֶת עַצְמָן לַמְּצִיאוּת הַחֶבְרָתִית הַמִּתְהַוָּה.
בַּשְּׁנָתַיִם הַחוֹלְפוֹת פָּגַשְׁתִּי עֶשְׂרוֹת עֲמֻתּוֹת וְאִרְגּוּנִים, אַנְשֵׁי אָקָדֶמְיָה וְחִנּוּךְ מַסּוֹרִים, הִתְרַשַּׁמְתִּי מֵאֵינְסְפוֹר מִפְגְּשֵׁי שִׂיחַ, וְאוּלָם, אִם אֲנַחְנוּ מְעֻנְיָנִים בְּשִׁנּוּי אֲמִתִּי וְעָמֹק, אֲנַחְנוּ חַיָּבִים לַחְתֹּר לְשִׁנּוּיִים מוֹסָדִיִּים וּמַעֲרַכְתִּיִּים אוֹתָם יֵשׁ לְנַהֵל כְּמַאֲמָץ לְאֻמִּי, לְהַצִּיב יְעָדִים שֶׁאֶפְשָׁר לְהַסְכִּים עֲלֵיהֶם וְלַעֲקֹב אַחֲרֵיהֶם.
יְתֵרָה מִזֹּאת, עָלֵינוּ לְהַכִּיר בְּכָךְ שֶׁהַיּוֹם קַיָּמִים חֲסָמִים מַהוּתִיִּים וּמִבְנִיִּים, הַמַּקָּשִׁים עָלֵינוּ לְבַסֵּס אֶת כְּלָלֵי הַמִּשְׂחָק הַמְּשֻׁתָּפִים בֵּין הַמִּגְזָרִים הַשּׁוֹנִים: פּוֹלִיטִיקָה סֶקְטוֹרְיָאלִית, שֶׁלֹּא פּוֹסַחַת לְדַאֲבוֹנֵנוּ עַל אַף שֵׁבֶט, פּוֹלִיטִיקָה שֶׁנִּזּוֹנָה וּמִתְפַּרְנֶסֶת מִקִּיצוֹנוֹת וְהַקְצָנָה, וּמְשַׁמֶּשֶׁת תַּמְרִיץ לְהַעֲמָקַת הַשִּׁבְטִיּוֹת; מַעֲרֶכֶת צִבּוּרִית שֶׁטֶּרֶם הִפְנִימָה וְלָמְדָה אֶת הַשִּׁנּוּיִים שֶׁעוֹבֶרֶת הַחֶבְרָה הַיִּשְׂרְאֵלִית, אֶת הָאֶתְגָּרִים המגזריים הַשּׁוֹנִים, וְשֶׁטֶּרֶם עָבְרָה בְּעַצְמָהּ תַּהֲלִיךְ שֶׁל גִּוּוּן; וּלְצַד אֵלֶּה, מַעֲרֶכֶת חֶבְרָתִית שְׁלֵמָה, שֶׁמֻּשְׁתֶּתֶת עַל הַפִּצּוּל בֵּין הַמִּגְזָרִים: בֶּעָרִים, בַּתִּקְשֹׁרֶת, בַּכַּלְכָּלָה, וּמֵעַל לַכֹּל, בַּחִנּוּךְ.
בְּמַעֲרֶכֶת מְפֻצֶּלֶת כָּל כָּךְ גַּם כּוֹחוֹת הַשּׁוּק לֹא יַצְלִיחוּ לִיצֹר אִינְטֶרֶסִים מְשֻׁתָּפִים.
