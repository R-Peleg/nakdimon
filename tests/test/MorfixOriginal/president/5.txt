לְצַעֲרֵנוּ, אָנוּ חַיִּים בַּמְּצִיאוּת שֶׁל סִכְסוּךְ עָקֹב מִדָּם.
לְקֵץ הַסִּכְסוּךְ הַזֶּה, הַטְּרָגֶדְיָה הַזֹּאת, אֵין לוּחוֹת זְמַנִּים בְּרוּרִים.
מָה שֶׁבָּרוּר הוּא שֶׁהָאַחְרָיוּת לְחַיֵּינוּ כָּאן, לְחַיֵּינוּ הַמְּשֻׁתָּפִים כָּאן, יְהוּדִים וַעֲרָבִים, מוּטֶלֶת עַל כִּתְפֵי כֻּלָּנוּ.
מָה שֶׁבָּרוּר הוּא שֶׁהַתַּשְׁתִּית לְכָל פִּתְרוֹן שֶׁיִּהְיֶה הוּא בִּסּוּס אָמוּן בֵּין הַצְּדָדִים.
בַּזְּמַן הָאַחֲרוֹן אֲנַחְנוּ עֶדְיָם בַּיֶּתֶר שְׂאֵת לַנִּסָּיוֹן שֶׁל מַנְהִיגִים פּוֹלִיטִיִּים וַאֲחֵרִים לַהֲפֹךְ אֶת הַסִּכְסוּךְ הַיִּשְׂרְאֵלִי פָלַסְטִינִי לְסִכְסוּךְ דָּתִי.
לְכָזֶה שֶׁאֵין מִמֶּנּוּ דֶּרֶךְ חָזְרָה.
מִלְחֶמֶת גּוֹג וּמָגוֹג, מִלְחֶמֶת יוֹם הַדִּין, גִ'הַאד, אוֹ קֵץ הַיָּמִים.
הַנִּסָּיוֹן לִשְׁפֹּךְ דֶּלֶק וְלַהַט פוּנְדָּמֶנְטָלִיסְטִיִּים וּמְשִׁיחִיִּים עַל מְדוּרַת הַסִּכְסוּךְ, הוּא נִסָּיוֹן לְהַבְעִיר אֶת הָאֲדָמָה הַזֹּאת עָלֶיהָ יוֹשְׁבִים כֻּלָּנוּ.
וְכָל מִי שֶׁעוֹשֶׂה זֹאת- דָּמֵנוּ, דַּם כֻּלָּנוּ, בְּרֹאשׁוֹ.
דֵּי לָנוּ אִם נַבִּיט סְבִיבֵנוּ בַּמִּעוּטִים הַנִּשְׁחָטִים בְּסוּרְיָה וּבְעִירָאק כְּדֵי שֶׁנָּבִין לְאֵיזוֹ תְּהוֹם אֲפֵלָה עֲשׂוּיָה לִגְרֹר אֶת כֻּלָּנוּ מִלְחֶמֶת דָּת.
לָעָם הַיְּהוּדִי וְלִמְדִינַת יִשְׂרָאֵל לֹא הָיְתָה וְלֹא תִּהְיֶה מִלְחָמָה עִם הָאִסְלָאם.
הַשֶּׁקֶר הַנּוֹאָל וְהַמְּרֻשָּׁע לְפִיו מְדִינַת יִשְׂרָאֵל חוֹתֶרֶת לְשַׁנּוֹת אֶת הַסְּטָטוּס קְווֹ בְּהַר הַבַּיִת אֵינוֹ רַק שֶׁקֶר גַּס, אֶלָּא גַּם עֲלִילָה מְסֻכֶּנֶת שֶׁעוֹלָה כְּבָר כָּעֵת בְּדָמָם שֶׁל חַפִּים מִפֶּשַׁע.
מְדִינַת יִשְׂרָאֵל וּמֶמְשֶׁלֶת יִשְׂרָאֵל נְחוּשׁוֹת בְּכַוָּנָתָן לִשְׁמֹר וּלְהָגֵן עַל הַסְּטָטוּס קְווֹ בְּהַר הַבַּיִת.
בְּנִגּוּד לַקַּמְפֵּיְן הַמְּרֻשָּׁע וְהַשִּׁקְרִי שֶׁמְּנֹהָל נֶגְדֵּנוּ מֵעַזָּה וּמֵרָמָאלָה וּלְדַאֲבוֹנֵנוּ- גַּם מִתּוֹךְ מְדִינַת יִשְׂרָאֵל.
יִשְׂרָאֵל אֵינָהּ מְנַסֶּה לִמְנֹעַ מִמֻּסְלְמִים לְהִתְפַּלֵּל בַּמְּקוֹמוֹת הַקְּדוֹשִׁים לָהֶם, כְּפִי שֶׁיִּשְׂרָאֵל לְעוֹלָם לֹא תִּפְגַּע בַּמִּסְגָּדִים בְּהַר הַבַּיִת.
וַאֲנִי מַאֲמִין שֶׁגַּם מְפִיצֵי הַשֶּׁקֶר הַזֶּה יוֹדְעִים זֹאת הֵיטֵב.
גָּדַלְתִּי בִּירוּשָׁלַיִם בְּיָמִים בָּהֶם נֶאֶסְרָה עָלֵינוּ בְּכוֹחַ הַזְּרוֹעַ הִגִּישָׁה לַמְּקוֹמוֹת הַקְּדוֹשִׁים, לַמְרוֹת הֶסְכֵּם בֵּינְלְאֻמִּי חָתוּם.
עוֹד קֹדֶם לָכֵן, אֲנִי זוֹכֵר כְּיֶלֶד קְטַן בֵּן חָמֵשׁ כֵּיצַד תְּקִיעָה בַּשּׁוֹפָר בְּרַחֲבַת הַכֹּתֶל הַצָּרָה בְּמוֹצָאֵי יוֹם כִּפּוּר הָיְתָה עִלָּה לַמַּעֲצָר בִּידֵי חַיָּלֵי הַמַּנְדָּט הַבְּרִיטִי.
כְּמִי שֶׁהָיוּ קָרְבָּנוֹת בְּעַצְמָם שֶׁל הַגְבָּלַת חֹפֶשׁ הַפֻּלְחָן בָּעִיר יְרוּשָׁלַיִם, אָנוּ לְעוֹלָם לֹא נִנְהַג כָּךְ.
הַמְּחֻיָּבוּת שֶׁלָּנוּ כִּיהוּדִים וּכְרִבּוֹנִים לַשְּׁמִירָה עַל הַסְּטָטוּס קְווֹ, אֵינָהּ נוֹבַעַת רַק מִמְּחֻיָּבוּת לְהֶסְכֵּמִים פוֹרְמָלִיִּים.
הִיא נוֹבַעַת מִתּוֹךְ הַהֲבָנָה הָעֲמֻקָּה שֶׁל הָאַחְרָיוּת הַכְּרוּכָה בְּרִבּוֹנוּת עַל יְרוּשָׁלַיִם, הָאַחְרָיוּת לְבִטְחוֹנָם וּשְׁלוֹמָם שֶׁל כָּל תּוֹשָׁבֶיהָ שֶׁל הָעִיר.
הָאַחְרָיוּת לְהַבְטָחַת חֹפֶשׁ הַפֻּלְחָן שֶׁל כָּל הַמַּאֲמִינִים הַנּוֹשְׂאִים אֵלֶיהָ אֶת עֵינֵיהֶם.
וְזֹאת גַּם בִּמְחִיר הַגְבָּלָה קָשָׁה וְכוֹאֶבֶת שֶׁל יְכָלְתֵּנוּ הַיְּהוּדִים לְהִתְפַּלֵּל וַאֲנִי מַדְגִּישׁ- לְהִתְפַּלֵּל, בְּמָקוֹם כֹּה קָדוֹשׁ עֲבוּרֵנוּ.
מְדִינַת יִשְׂרָאֵל לֹא תֶּאֱסֹר עַל כְּנִיסַת יְהוּדִים לְהַר הַבַּיִת וְאֵין בְּכָךְ מִשּׁוּם פְּגִיעָה אוֹ הֲפָרָה שֶׁל הַסְּטָטוּס קְווֹ.
זִקָּתוֹ שֶׁל הָעָם הַיְּהוּדִי לְהַר הַבַּיִת אֵינָהּ נִתֶּנֶת לַהַכְחָשָׁה אוֹ לַסִּלּוּף.
הָאֲמִירָה לְפִיהָ יְהוּדִים מְטַנְּפִים בְּרַגְלֵיהֶם אֶת הַר הַבַּיִת הִיא אֲמִירָה בִּלְתִּי נִסְבֶּלֶת, שֶׁעִמָּהּ לֹא נוּכַל לְהַשְׁלִים.
יֵשׁ לִזְכֹּר, עַד כֹּה וּלְאֹרֶךְ הַהִיסְטוֹרְיָה רַק יְרוּשָׁלַיִם בְּרִבּוֹנוּת יִשְׂרָאֵל יָדְעָה לָתֵת שֶׁקֶט, שָׁלוֹם וְחֹפֶשׁ פֻּלְחָן לְכָל תּוֹשָׁבֶיהָ.
וְכָךְ- גַּם יִשָּׁאֲרוּ פְּנֵי הַדְּבָרִים.
