עָמְרִי טָעַן שֶׁבִּמְקוֹם שֶׁמֹּשֶׁה יְבָרֵר אֶת טַעֲנוֹתָיו בָּאֲגֻדָּה לְתַרְבּוּת הַדִּיּוּר, בְּשָׂפָה מְקֻבֶּלֶת, בָּחַר לָצֵאת בְּמַסַּע הַשְׁמָצוֹת מְרֻשָּׁעוֹת, נִבְזִיּוֹת, שִׁקְרִיּוּת, שֶׁכָּל מַטְּרָתָן לְבַזּוֹת אֶת עָמְרִי, מוּל שְׁכֵנָיו, מַכָּרָיו וּבְנֵי מִשְׁפַּחְתּוֹ, תּוֹךְ שֶׁהוּא מְכַנֶּה אוֹתוֹ בִּשְׁלַל כִּנּוּיִים לְרַבּוֹת גַּנָּב, נוּכַל, כֶּלֶב, חֲזִיר, וְכֵן רְמִיזוֹת בְּנוֹגֵעַ לִנְטִיּוֹתָיו הַמִּינִיּוֹת.
עוֹד טָעַן שֶׁבְּאַרְבַּע פְּעָמִים שׁוֹנוֹת, טָעַן הַוַּעַד הַנִּכְנָס כִּי חַבְרֵי הַוַּעַד הַיּוֹצֵא, וּבִפְרַט עָמְרִי, אַחְרָאִים לַחֹסֶר שֶׁיֵּשׁ בְּקֻפַּת הַוַּעַד.
הֲגַם שֶׁהַדְּבָרִים אֵינָם נְכוֹנִים וְקַל לְהוֹכִיחָם, חֲבֵר הַוַּעַד הַנִּכְנָס (מְנַחֵם מכאל) וּמֹשֶׁה גַּם יַחַד, מָצְאוּ שֶׁהַדֶּרֶךְ הַנְּכוֹנָה הִיא לָצֵאת בַּהַאֲשָׁמוֹת נֶגֶד עָמְרִי, רַק כִּי אֶפְשָׁר.
מֹשֶׁה טָעַן שֶׁעָמְרִי בָּחַר לְצַטֵּט חֵלֶק מִזְעָרִי מֵהַשִּׂיחַ בִּקְבוּצַת הווטסאפ הַמְּשֻׁתֶּפֶת שֶׁל בַּעֲלֵי הַדִּירוֹת בַּבִּנְיָן.
עִיּוּן בְּכָל הִשְׁתַּלְשְׁלוּת הַשִּׂיחַ בִּקְבוּצַת הווטסאפ מְגַלֶּה כִּי דַּוְקָא עָמְרִי הוּא זֶה אֲשֶׁר הוֹצִיא דִּבָּתוֹ שֶׁל מֹשֶׁה רָעָה, הֶאֱשִׁים אוֹתוֹ בַּקְּבוּצָה בְּהַאֲשָׁמוֹת שָׁוְא, הֶעֱלִיב אוֹתוֹ, פָּגַע בּוֹ וְהַזִּיק לִשְׁמוֹ הַטּוֹב.
עָמְרִי בָּחַר לְהִשְׁתַּלֵּחַ בְּדַיָּרֵי הַבִּנְיָן וּבְמֹשֶׁה דַּוְקָא כְּמִי שֶׁעָשָׂה יָד אַחַת לִכְאוֹרָה עִם חַבְרֵי הַנְּצִיגוּת, לְטוֹבָתוֹ הָאִישִׁית בְּגִין בִּצּוּעַ עֲבוֹדוֹת בַּגָּדֵר הָאֲחוֹרִית שֶׁל הַבִּנְיָן בָּרְכוּשׁ הַמְּשֻׁתָּף, תּוֹךְ גְּרִיפַת רְוָחִים לִכְאוֹרָה.
