בַּמַּהֲלָךְ אוֹתָהּ תְּקוּפָה, הָאָשְׁמוּ עַל יְדֵי מקארתי אֲלָפִים מֵאֶזְרְחֵי אַרְצוֹת הַבְּרִית, אוֹ בַּחֲבֵרוֹת בַּמִּפְלָגָה הַקּוֹמוּנִיסְטִית שֶׁל אַרְצוֹת הַבְּרִית אוֹ בִּתְמִיכָה וְאַהֲדָה בָּהּ וּבָרַעְיוֹן הַקּוֹמוּנִיסְטִי.
נֶחְשְׁדוּ בְּכָךְ, בֵּין הַשְּׁאָר, מֵאוֹת אָמָּנִים, מוֹרִים, אַנְשֵׁי קוֹלְנוֹעַ וּדְמֻיּוֹת מוֹבִילוֹת בְּהוֹלִיווּד, פְּעִילִים שֶׁל אִגּוּדֵי עוֹבְדִים, עִתּוֹנָאִים וְאַנְשֵׁי תִּקְשֹׁרֶת, אַנְשֵׁי צָבָא וְעוֹבְדִי וְאַף בְּכִירִים יַחֲסִית בַּמִּמְשָׁל הָאָמֶרִיקָאִי.
הַדָּבָר הֵבִיא רַבִּים מֵהֶם לַמַּצָּב בּוֹ אוֹתָם נֶאֱשָׁמִים פֻּטְּרוּ וְאִבְּדוּ אֶת הַקַּרְיֶרָה שֶׁלָּהֶם, רַבִּים נִכְלְאוּ וּבוֹדְדוּ חֶבְרָתִית.
אַנְשֵׁי קוֹלְנוֹעַ שֶׁנִּכְלְלוּ בְּ"רְשִׁימָה הַשְּׁחֹרָה שֶׁל הוֹלִיווּד" לֹא זָכוּ לְהִכָּלֵל בַּהֲפָקוֹת כָּלְשֶׁהֵן וְחַיֵּיהֶם נֶהֶרְסוּ.
טַעֲנַת הַסֶּנָטוֹר הָיְתָה שֶׁאוֹתָם אֲנָשִׁים הִסְתַּנְּנוּ לַמִּמְשָׁל הָאָמֶרִיקָאִי וּפוֹגְעִים בָּאֻמָּה מִבִּפְנִים.
הוּא גָּרַם לַצִּבּוּר לְהַאֲמִין שֶׁיֵּשׁ סִכּוּי לְכָךְ, מָה שֶׁהֵבִיא לַחֲשָׁשׁ עָצוּם מֵהַשְׁפָּעָה קוֹמוּנִיסְטִית וּמִמְּרַגְּלִים סוֹבְיֶטִיִּים בְּכָל פִּנָּה.
אֲפִלּוּ הַ-FBI וְרֹאשׁוֹ הַמִּיתוֹלוֹגִי ג'ון אֶדְגַר הוּבֶר פָּצְחוּ בַּשּׁוּרָה שֶׁל פְּעֻלּוֹת אַנְטִי-קוֹמוּנִיסְטִיּוֹת, שֶׁרַק חִזְּקוּ אֶת אֲמִינוּתָם שֶׁל ג'וזף מקארתי וּמַסַּע הַהַפְחָדָה וְהָרְדִיפָה שֶׁלּוֹ.
גַּם מִתְנַגְּדָיו הוֹדוּ אָז שמקארתי הָיָה מֻכְשָׁר בַּתִּקְשֹׁרֶת וְתָמִיד הִקְדִּים אוֹתָם.
לֹא פַּעַם הוּא נָקַט בְּשִׁיטוֹת מְפֻקְפָּקוֹת, שֶׁלָּקַח זְמַן רַב עַד שֶׁהִתְגַּלּוּ בִּמְלֵא הֶקֵּפָם.
אָז נוֹלַד הַסֶּדֶק הָרִאשׁוֹן בַּמֶּקַרְתִּיזְם.
הַסֶּדֶק הַזֶּה נוֹלַד כְּשֶׁבְּאֶחָד מִנְּאוּמָיו הַיְּדוּעִים הוּא זָכָה לְאֵמוּן הַמַּאֲזִינִים לוֹ, כְּשֶׁנִּפְנֵף בְּדַף נְיָר וְהִכְרִיז: "בְּיָדַי 205 שֵׁמוֹת שֶׁל קוֹמוּנִיסְטִים שֶׁחָדְרוּ לְמִשְׂרַד הַחוּץ שֶׁלָּנוּ".
עִתּוֹנַאי חָרוּץ חִטֵּט לְאַחַר הַנְּאוּם הַזֶּה בְּדוּכַן הַנּוֹאֲמִים וְחִפֵּשׂ אֶת שְׁמוֹתֵיהֶם שֶׁל הַקּוֹמוּנִיסְטִים הַלָּלוּ.
לְהַפְתָּעָתוֹ הוּא גִּלָּה שֶׁהַנְּיָר שֶׁבּוֹ נִפְנֵף הַסֶּנָטוֹר הָיָה רֵיק לַחֲלוּטִין.
לֹא הָיָה כָּתוּב בּוֹ דָּבָר.
